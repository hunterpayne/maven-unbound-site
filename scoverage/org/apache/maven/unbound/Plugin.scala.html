73 case class Plugin(
74   groupId: String = SL.DefaultPluginGroup,
75   artifactId: String = null, version: String = null,
76   extensions: Boolean = false,
77   executions: Seq[Execution] = Seq[Execution](),
78   dependencies: Seq[Dependency] = Seq[Dependency](),
79   inherited: Boolean = true,
80   configuration: Config = null) {
81 
82   def this(elem: Elem) = this(
83     emptyToDefault((elem \ SL.GroupId).text, SL.DefaultPluginGroup),
84     emptyToNull((elem \ SL.ArtifactId).text),
85     emptyToNull((elem \ SL.Version).text),
86     emptyToDefault(
87       (elem \ SL.Extensions).text.toLowerCase(Locale.ROOT), SL.FalseStr) ==
88       SL.TrueStr.toString,
89     (elem \ SL.Executions \ SL.ExecutionStr).map { case e: Elem =>
90       new Execution(e) },
91     (elem \ SL.Dependencies \ SL.DependencyStr).map { case e: Elem =>
92       new Dependency(e) },
93     emptyToDefault(
94       (elem \ SL.Inherited).text.toLowerCase(Locale.ROOT), SL.TrueStr) ==
95       SL.TrueStr.toString,
96     (elem \ SL.Configuration).headOption.map { case e: Elem =>
97       elemToConfig(e) }.getOrElse(null))
98 
99   lazy val xml = <plugin>
100                    <groupId>{groupId}</groupId>
101                    <artifactId>{artifactId}</artifactId>
102                    { if (version != null) <version>{version}</version> }
103                    { if (extensions) <extensions>true</extensions> }
104                    { if (!executions.isEmpty) <executions>
105                        { executions.map { _.xml } }
106                      </executions> }
107                    { if (!dependencies.isEmpty) <dependencies>
108                        { dependencies.map { _.xml } }
109                      </dependencies> }
110                    { if (!inherited) <inherited>false</inherited> }
111                    { if (configuration != null) configToElem(configuration) }
112                  </plugin>
113 
114   def makeModelObject(): org.apache.maven.model.Plugin = {
115     val plugin = new org.apache.maven.model.Plugin()
116     plugin.setGroupId(groupId)
117     plugin.setArtifactId(artifactId)
118     plugin.setVersion(version)
119     plugin.setExtensions(extensions)
120     executions.foreach { ex => plugin.addExecution(ex.makeModelObject()) }
121     dependencies.foreach { d => plugin.addDependency(d.makeModelObject()) }
122     plugin.setInherited(inherited)
123     if (configuration != null) {
124       val xmlStr =
125         (new Writeable {
126           val xml: Elem = configToElem(configuration)
127         }).toXmlString
128       val sReader = new StringReader(xmlStr)
129       plugin.setConfiguration(Xpp3DomBuilder.build(sReader))
130     }
131     plugin
132   }
133 }
