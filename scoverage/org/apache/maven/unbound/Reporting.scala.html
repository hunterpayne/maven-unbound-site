60 case class Reporting(
61   excludeDefaults: Boolean = false,
62   outputDirectory: String = SL.Target + "/" + SL.SiteStr,
63   plugins: Seq[ReportPlugin] = Seq[ReportPlugin]()) {
64 
65   def this(elem: Elem) = this(
66     emptyToDefault(
67       (elem \ SL.ExcludeDefaults).text.toLowerCase(Locale.ROOT), SL.FalseStr) ==
68       SL.TrueStr.toString,
69     emptyToDefault(
70       (elem \ SL.OutputDirectory).text, SL.Target + "/" + SL.SiteStr),
71     (elem \ SL.Plugins \ SL.PluginStr).map { case e: Elem =>
72       new ReportPlugin(e) })
73 
74   lazy val xml =
75     <reporting>
76       { if (excludeDefaults) <excludeDefaults>true</excludeDefaults> }
77       { if (outputDirectory != null &&
78         outputDirectory != (SL.Target + "/" + SL.SiteStr).toString)
79         <outputDirectory>{outputDirectory}</outputDirectory> }
80       { if (!plugins.isEmpty) <plugins>
81         { plugins.map { _.xml } }
82         </plugins> }
83     </reporting>
84 
85   def makeModelObject(): org.apache.maven.model.Reporting = {
86     val reporting = new org.apache.maven.model.Reporting()
87     reporting.setExcludeDefaults(
88       if (excludeDefaults) SL.TrueStr else SL.FalseStr)
89     reporting.setOutputDirectory(outputDirectory)
90     plugins.foreach { plugin => reporting.addPlugin(plugin.makeModelObject()) }
91     reporting
92   }

135 case class ReportPlugin(
136   groupId: String = SL.DefaultPluginGroup,
137   artifactId: String = null, version: String = null,
138   reportSets: Seq[ReportSet] = Seq[ReportSet](),
139   inherited: Boolean = true,
140   configuration: Config = null) {
141 
142   def this(elem: Elem) = this(
143     emptyToDefault((elem \ SL.GroupId).text, SL.DefaultPluginGroup),
144     emptyToNull((elem \ SL.ArtifactId).text),
145     emptyToNull((elem \ SL.Version).text),
146     (elem \ SL.ReportSets \ SL.ReportSetStr).map { case e: Elem =>
147       new ReportSet(e)},
148     emptyToDefault(
149       (elem \ SL.Inherited).text.toLowerCase(Locale.ROOT), SL.TrueStr) ==
150       SL.TrueStr.toString,
151     (elem \ SL.Configuration).headOption.map { case e: Elem =>
152       elemToConfig(e) }.getOrElse(null))
153 
154   lazy val xml =
155     <plugin>
156       { if (groupId != null && groupId != SL.DefaultPluginGroup.toString)
157         <groupId>{groupId}</groupId> }
158       { if (artifactId != null) <artifactId>{artifactId}</artifactId> }
159       { if (version != null) <version>{version}</version> }
160       { if (!reportSets.isEmpty) <reportSets>
161         { reportSets.map { _.xml } }
162         </reportSets> }
163       { if (!inherited) <inherited>false</inherited> }
164       { if (configuration != null) configToElem(configuration) }
165     </plugin>
166 
167   def makeModelObject(): org.apache.maven.model.ReportPlugin = {
168     val plugin = new org.apache.maven.model.ReportPlugin()
169     plugin.setGroupId(groupId)
170     plugin.setArtifactId(artifactId)
171     plugin.setVersion(version)
172     reportSets.foreach { rs => plugin.addReportSet(rs.makeModelObject()) }
173     if (configuration != null) {
174       val xmlStr =
175         (new Writable {
176           val xml: Elem = configToElem(configuration)
177         }).toXmlString
178       val sReader = new StringReader(xmlStr)
179       plugin.setConfiguration(Xpp3DomBuilder.build(sReader))
180     }
181     plugin.setInherited(if (inherited) SL.TrueStr else SL.FalseStr)
182     plugin
183   }
184 }
