86   /**
87     * Converts an XML Element into a Typesafe Config object.  Generally it
88     * converts Elements into Maps keyed by the names of the child tags.  It
89     * also looks for special patterns of XML to do more specific type
90     * conversions. Supported type conversions: <ul>
91     * <li>Boolean the strings true or false</li>
92     * <li>Number both Integers and Doubles</li>
93     * <li>String Text nodes of the XML document</li>
94     * <li>Seq[Any] Elements whose child labels are the singlar form of elem's
95     * label
96     * <li>Properties Elements whose grand child labels are all name or value
97     * <li>Seq[Dependency] Special case where child elements all have the
98     * label &quot;dependency&quot;
99     * <li>Archiver Special case where elem's label is &quot;archive&quot;
100     * <li>Resource Transformer Special case where the elem's label is
101     * transformers, the child's label is transformer and there is an attribute
102     * named implementation defined
103     * <li>Map[String, Any] general/default case
104     * <ul>
105     * @see Configuring Plugins
106     * [[https://maven.apache.org/guides/mini/guide-configuring-plugins.html]]
107     * @see Archiver [[http://maven.apache.org/shared/maven-archiver/index.html]]
108     * @see Resource Transformers
109     * [[https://maven.apache.org/plugins-archives/maven-shade-plugin-2.0/examples/resource-transformers.html]]
110     */
111   def elemToConfig(elem: Elem): Config = {
112     import scala.collection.JavaConverters._
113 
114     // handles Booleans and Numbers
115     def toAnyRef(s: String): Any = s match {
116       case "true" => true
117       case "false" => false
118       case s =>
119         if (s.startsWith("0")) {
120           unencodeXml(s)
121         } else {
122           try { s.toInt } catch {
123             case nfe: NumberFormatException =>
124               try { s.toDouble } catch {
125                 case nfe2: NumberFormatException => unencodeXml(s)
126               }
127           }
128         }
129     }
130 
131     // converts a node el at path path to a ConfigValue inside of parent
132     def appendNode(path: String, parent: Config, el: Node): ConfigValue =
133       if (el != null) {
134         el match {
135           case e: Elem =>
136             // a simple string value
137             val attrs = e.attributes.asAttrMap
138             if (e.child.forall { _.isInstanceOf[Text] } && attrs.isEmpty) {
139               ConfigValueFactory.fromAnyRef(toAnyRef(e.text.trim))
140 
141               // an Archiver
142             } else if (e.label == SL.Archive.toString) {
143               val a = new Archiver(e)
144               ConfigFactory.parseString(JsonWriter.writeArchiver(a)).root()
145 
146               // a Dependency object
147             } else if (attrs.isEmpty && e.child.forall {
148               case ele: Elem => ele.label == SL.DependencyStr.toString
149               case t: Text => t.text.trim == ""
150               case _ => false
151             }) {
152               val listInJava = e.child.map {
153                 appendNode(path, parent, _) }.filter { _ != null}.asJava
154               ConfigValueFactory.fromIterable(listInJava)
155 
156               // a property map (ie Map[String, String])
157             } else if (attrs.isEmpty && e.child.forall {
158               case el: Elem => el.label == SL.PropertyStr.toString
159               case t: Text => t.text.trim == ""
160               case _ => false
161             }) {
162               val pairSeqInScala = (e \ SL.PropertyStr).map { el =>
163                 ((el \ SL.Name).text, (el \ SL.ValueStr).text) }
164               val mapInJava = (pairSeqInScala :+ (
165                 (SL.PropertiesStr.toString, SL.TrueStr.toString))).
166                 toMap.asJava
167               ConfigValueFactory.fromMap(mapInJava)
168 
169               // a list
170             } else if (attrs.isEmpty && e.child.forall {
171               case el: Elem => e.label.startsWith(el.label)
172               case t: Text => t.text.trim == ""
173               case _ => false
174             }) {
175               val listInJava = e.child.map {
176                 appendNode(path, parent, _) }.filter { _ != null }.asJava
177               ConfigValueFactory.fromIterable(listInJava)
178 
179             } else {
180               // check if this is a resource transformer
181               val implAttr = (e \ ("@" + SL.Implementation))
182               val mapInScala: Map[String, ConfigValue] =
183                 e.child.map {
184                   case ele: Elem =>
185                     val newPath =
186                       if (path != "") path + "." + ele.label else ele.label
187                     (ele.label, appendNode(newPath, parent, ele))
188                   case t: Text => (SL.TextStr.toString, null)
189                 }.filter { case(k, v) => v != null }.toMap
190               val map = mapInScala.foldLeft(parent) { case(c, (k, v)) =>
191                 c.withValue(k, v) }.root()
192 
193               // resource transformer
194               if (e.label == SL.Transformer.toString && !implAttr.isEmpty) {
195                 map.withValue(
196                   SL.Implementation,
197                   ConfigValueFactory.fromAnyRef(implAttr.text.trim))
198                 // handle attributes here
199               } else if (!attrs.isEmpty) {
200                 val keys = attrs.keySet.toSeq
201                 attrs.foldLeft(map) { case(c, (n, v)) =>
202                   c.withValue(n, ConfigValueFactory.fromAnyRef(v))
203                 }.withValue(
204                   "attributeKeys",
205                   ConfigValueFactory.fromIterable(keys.asJava))
206               } else {
207                 map
208               }
209             }
210           case t: Text =>
211             if (t.text.trim != "")
212               ConfigValueFactory.fromAnyRef(toAnyRef(t.text.trim))
213             else null
214           case c: Comment =>
215             val epty = ConfigFactory.empty()
216             val list = new java.util.ArrayList[java.lang.String]()
217             list.add(c.text)
218             epty.root().withOrigin(epty.origin().withComments(list))
219             // unhandled
220           case n: Node =>
221             println("n " + n)
222             assert(false)
223             null
224         }
225       } else {
226         null
227       }
228 
229     // check if null and start recursing down the heirarchy of children
230     if (elem != null) {
231       val empty = ConfigFactory.empty()
232       // build each child's ConfigObject
233       val childConfs =
234         elem.child.map { ch => (ch.label, appendNode(ch.label, empty, ch)) }
235       // fold them together into 1 big Config at the paths
236       childConfs.foldLeft(ConfigFactory.empty()) { case(c, (k, v)) =>
237         if (v != null) c.withValue(k, v) else c }
238     } else {
239       null
240     }
241   }

249   // translate from json string to a jvm string
250   def unencodeJson(s: String): String = {
251     val sb = new StringBuilder()
252     var i = 0
253     // the scala compiler didn't want to do this with a flatMap and a flag so
254     // I had to do it with a while loop, the compiler kept moving the flag
255     // before the flatMap when it was being used in an if condition after the
256     // flatMap even though it was a var
257     while (i < s.length - 1) {
258       s.substring(i, i + 2) match {
259         case "\\b" => sb.append('\b'); i = i + 2
260         case "\\f" => sb.append('\f'); i = i + 2
261         case "\\n" => sb.append('\n'); i = i + 2
262         case "\\r" => sb.append('\r'); i = i + 2
263         case "\\t" => sb.append('\t'); i = i + 2
264         case "\\\"" => sb.append('"'); i = i + 2
265         case "\\\\" => sb.append('\\'); i = i + 2
266         case ch => sb.append(ch(0)); i = i + 1
267       }
268     }
269     if (i == s.length - 1) sb.append(s.last)
270     sb.toString
271   }

295   /**
296     * Translates a Typesafe Config object to a XML Element.  Handles all the
297     * same special cases as elemToConfig.
298     * Supported type conversions: <ul>
299     * <li>Boolean &lt;key&gt;true/false&lt;/key&gt;</li>
300     * <li>Number &lt;key&gt;number&lt;/key&gt;</li>
301     * <li>String &lt;key&gt;value&lt;/key&gt;</li>
302     * <li>Seq[Any] &lt;key(plural)&gt;&lt;key(singular)&gt;value(0)
303     * &lt;/key(singular)&gt;&lt;key(singular)&gt;value(1)&lt;/key(singular)&gt;
304     * ...&lt;/key(plural)&gt;
305     * <li>Properties Identified by the presence of a child at path
306     * &quot;properties&quot; with the value true, otherwise represented as a
307     * Map[String, String] in the Scala code.  It returns the Maven property
308     * XML format which is: &lt;key&gt;&lt;property&gt;&lt;name&gt;property
309     * key1&lt;/name&gt;&lt;value&gt;property
310     * value1&lt;/value&gt;&lt;/property&gt;
311     * &lt;property&gt;&lt;name&gt;property key2&lt;/name&gt;
312     * &lt;value&gt;property value2&lt;/value&gt;&lt;/property&gt;...&lt;/key&gt;
313     * <li>Seq[Dependency] Special case where child elements all have the
314     * keys of a valid &quot;dependency&quot; object in the POM
315     * <li>Archiver Special case where a child key is &quot;archive&quot;
316     * <li>Resource Transformer Special case where a child key is
317     * implementation
318     * <li>Map[String, Any] general/default case, returns whatever
319     * the value translates to inside a XML element labeled by the key
320     * <ul>
321     */
322   def configToElem(config: Config): Elem = {
323     import scala.collection.JavaConverters._
324 
325     // convert the Config value at key into a XML Element
326     def makeElem(key: String, value: ConfigValue): Elem =
327       // check for a child key called archive
328       if (key == SL.Archive.toString) {
329         HoconReader.readArchiver(value.atKey(SL.Archive)).xml
330         // check for child keys with . so they can be split and encoded into
331         // XML correctly
332       } else if (key.contains(".")) {
333         val tokens: Array[String] = key.split('.').reverse
334         val last = tokens.head
335         tokens.tail.foldLeft(makeElem(last, value)) { case(c, token) =>
336           new Elem(null, token, Null, TopScope, c)
337         }
338       } else if (value != null) {
339         value match {
340           // a list so create a parent element called key with children labeled
341           // key in singular form
342           case l: ConfigList =>
343             val childElems = l.asScala.map { it => makeElem(singular(key), it) }
344             new Elem(null, key, Null, TopScope, childElems: _*)
345             // a map so look for special cases and if they don't match convert
346             // as a generic map built using recursion
347           case m: ConfigObject =>
348             val mS = m.asScala
349             var elemKey = key
350             var attrs: MetaData = Null
351             val childElems =
352               if (mS.keySet.find {
353                 _ == SL.PropertiesStr.toString }.isDefined) {
354                 // a Properties object
355                 mS.filter{
356                   _._1 != SL.PropertiesStr.toString }.map { case(k, v) =>
357                     new Elem(
358                       null, SL.PropertyStr, Null, TopScope,
359                       new Elem(null, SL.Name, Null, TopScope, new Text(k)),
360                       new Elem(
361                         null, SL.ValueStr, Null, TopScope,
362                         new Text(removeQuotes(v.render()))))
363                 }.toSeq
364               } else if (mS.values.forall { v =>
365                 v.valueType() == ConfigValueType.STRING }) {
366                 if (mS.keySet.forall { isDependencyProperty(_) }) {
367                   // a Dependency object (for configurations)
368                   elemKey = SL.DependencyStr
369                   mS.map { case(k, v) => new Elem(
370                     null, k, Null, TopScope, Text(removeQuotes(v.render())))
371                   }.toSeq
372                 } else if (mS.keySet.find {
373                   _ == SL.Implementation.toString }.isDefined) {
374                   // a resource transformer
375                   elemKey = SL.Transformer
376                   val impl =
377                     mS.find { case(k, v) =>
378                       k == SL.Implementation.toString }.map { _._2 }.get
379                   attrs = new UnprefixedAttribute(
380                     SL.Implementation, removeQuotes(impl.render()), Null)
381                   mS.filter { case(k, _) =>
382                     k != SL.Implementation.toString }.map { case(k, v) =>
383                       new Elem(
384                         null, k, Null, TopScope, Text(removeQuotes(v.render())))
385                   }.toSeq
386                 } else if (mS.keySet.find {
387                   _ == SL.Archive.toString }.isDefined) {
388                   // a Maven archiver
389                   val arch = HoconReader.readArchiver(m.toConfig())
390                   Seq(arch.xml) ++
391                   mS.filter { _._1 != SL.Archive.toString }.map { case(k, v) =>
392                     makeElem(k, v) }
393                 } else {
394                   // a normal Map (ie non-string value type)
395                   mS.map { case(k, v) => makeElem(k, v) }.toSeq
396                 }
397               } else {
398                 // look for attributeKeys and if its present make those
399                 // children into attributes instead of elements
400                 val c = m.toConfig
401                 if (c.hasPath("attributeKeys")) {
402                   val attrKeys = c.getStringList("attributeKeys")
403 
404                   if (attrKeys != null && !attrKeys.isEmpty) {
405                     val attrValues: Seq[(String, String)] =
406                       attrKeys.asScala.filter(
407                         _ != SL.Implementation.toString).map { attr =>
408                         (attr, removeQuotes(mS(attr).render())) }
409                     val md: MetaData = Null
410                     attrs =
411                       attrValues.foldLeft(md) { case(n, (k, v)) =>
412                         new UnprefixedAttribute(k, v, n) }
413                     mS.filter { case(k, _) =>
414                       k != "attributeKeys" && !attrKeys.contains(k)
415                     }.map { case(k, v) => makeElem(k, v) }.toSeq
416 
417                   } else {
418                     // a normal Map (ie non-string value type)
419                     mS.map { case(k, v) => makeElem(k, v) }.toSeq
420                   }
421                 } else {
422 
423                   // a normal Map (ie non-string value type)
424                   mS.map { case(k, v) => makeElem(k, v) }.toSeq
425                 }
426               }
427 
428             new Elem(null, elemKey, attrs, TopScope, childElems: _*)
429             // a simple value so convert to a string and wrap into a Element
430             // labeled key
431           case v: ConfigValue =>
432             val value = removeQuotes(v.render())
433             new Elem(null, key, Null, TopScope, new Text(value))
434         }
435       } else {
436         null
437       }
438 
439     // start the recurse by visiting the top level children of the config
440     if (config != null) {
441       val children: Seq[(String, ConfigValue)] = config.root().asScala.toSeq
442       val childElems: Seq[Elem] =
443         children.map { entry => makeElem(entry._1, entry._2) }
444       new Elem(null, SL.Configuration, Null, TopScope, childElems: _*)
445     } else {
446       null
447     }
448   }

450   /**
451     * Converts a Json4s object into a Typesafe config in a very direct way
452     * Handles no special cases as there is an isomorphism between Json and
453     * Hocon.  Hocon is really just a special case of Json afterall.
454     */
455   def jsonToConfig(jobject: JObject): Config = {
456     import scala.collection.JavaConverters._
457 
458     def fromJson(v: JValue): Any =
459       if (v != null) {
460         v match {
461           case JNull => null
462           case s: JString => s.s
463           case b: JBool => b.value
464           case d: JDecimal => d.num
465           case d: JDouble => d.num
466           case i: JInt => i.num
467           case o: JObject => ConfigValueFactory.fromMap(fromJsonObject(o))
468           case a: JArray =>
469             val iter = a.arr.map { fromJson(_) }.asJava
470             ConfigValueFactory.fromIterable(iter)
471           case v: JValue => assert(false); null
472         }
473       } else {
474         null
475       }
476 
477     def fromJsonObject(jobj: JObject): java.util.Map[String, Any] =
478       jobj.obj.map { case((key, v)) =>
479         v match {
480           case arr: JArray => (key, fromJson(arr))
481           case obj: JObject => (key, fromJsonObject(obj))
482           case v: JValue => (key, fromJson(v))
483         }
484       }.toMap.asJava
485 
486     if (jobject != null)
487       ConfigValueFactory.fromMap(fromJsonObject(jobject)).toConfig()
488     else null
489   }

491   /**
492     * Converts a Typesafe Config object to a Json4s Json object usually for
493     * serialization.  Does a direct translation.
494     */
495   def configToJson(conf: Config): JObject = {
496     import scala.collection.JavaConverters._
497 
498     def makeJValuePrimitive(value: ConfigValue): JValue =
499       value.unwrapped() match {
500         case null => JNull
501         case b: java.lang.Boolean => JBool(b.booleanValue())
502         case b: java.lang.Byte => JInt(b.toInt)
503         case s: java.lang.Short => JInt(s.toInt)
504         case i: java.lang.Integer => JInt(i.toInt)
505         case l: java.lang.Long => JDecimal(new java.math.BigDecimal(l.toLong))
506         case f: java.lang.Float => JDouble(f.toDouble)
507         case d: java.lang.Double => JDouble(d.toDouble)
508         case s: String => JString(s)
509         case _ => assert(false); null
510       }
511 
512     def makeJValue(value: ConfigValue): JValue = {
513       value match {
514         case l: ConfigList =>
515           JArray(l.asScala.map { it => makeJValue(it) }.toList)
516         case m: ConfigObject =>
517           JObject(m.asScala.map { it => (it._1, makeJValue(it._2)) }.toList)
518         case _ => makeJValuePrimitive(value)
519       }
520     }
521 
522     if (conf != null) {
523       val root = conf.root()
524       val childKeys: Seq[String] = root.keySet().asScala.toSeq
525       val childElems: Seq[JField] =
526         childKeys.map { key => (key, makeJValue(root.get(key))) }
527       new JObject(childElems.toList)
528     } else {
529       null
530     }
531   }
