124   /**
125     * Converts an XML Element into a Typesafe Config object.  Generally it
126     * converts Elements into Maps keyed by the names of the child tags.  It
127     * also looks for special patterns of XML to do more specific type
128     * conversions. Supported type conversions: <ul>
129     * <li>Boolean the strings true or false</li>
130     * <li>Number both Integers and Doubles</li>
131     * <li>String Text nodes of the XML document</li>
132     * <li>Seq[Any] Elements whose child labels are the singlar form of elem's
133     * label
134     * <li>Properties Elements whose grand child labels are all name or value
135     * <li>Seq[Dependency] Special case where child elements all have the
136     * label &quot;dependency&quot;
137     * <li>Archiver Special case where elem's label is &quot;archive&quot;
138     * <li>Fileset Special case where elem's label is &quot;fileset&quot;
139     * <li>Resource Transformer Special case where the elem's label is
140     * transformers, the child's label is transformer and there is an attribute
141     * named implementation defined
142     * <li>Map[String, Any] general/default case
143     * <ul>
144     * @see Configuring Plugins
145     * [[https://maven.apache.org/guides/mini/guide-configuring-plugins.html]]
146     * @see Archiver [[http://maven.apache.org/shared/maven-archiver/index.html]]
147     * @see Resource Transformers
148     * [[https://maven.apache.org/plugins-archives/maven-shade-plugin-2.0/examples/resource-transformers.html]]
149     */
150   def elemToConfig(elem: Elem): Config = {
151     import scala.collection.JavaConverters._
152 
153     // handles Booleans and Numbers
154     def toAnyRef(s: String): Any = s match {
155       case "true" => true
156       case "false" => false
157       case s =>
158         if (s.startsWith("0")) {
159           unencodeXml(s)
160         } else {
161           try { s.toInt } catch {
162             case nfe: NumberFormatException =>
163               try { s.toDouble } catch {
164                 case nfe2: NumberFormatException => unencodeXml(s)
165               }
166           }
167         }
168     }
169 
170     // converts a node el at path path to a ConfigValue inside of parent
171     def appendNode(path: String, parent: Config, el: Node): ConfigValue =
172       if (el != null) {
173         el match {
174           case e: Elem =>
175             // a simple string value
176             val attrs = e.attributes.asAttrMap
177             if (e.child.forall { _.isInstanceOf[Text] } && attrs.isEmpty) {
178               ConfigValueFactory.fromAnyRef(toAnyRef(e.text.trim))
179 
180               // an Archiver
181             } else if (e.label == SL.Archive.toString && attrs.isEmpty) {
182               val a = new Archiver(e)
183               ConfigFactory.parseString(JsonWriter.writeArchiver(a)).root()
184 
185               // a fileset
186             } else if (e.label == SL.FilesetStr.toString && attrs.isEmpty) {
187               val f = new Fileset(e)
188               ConfigFactory.parseString(JsonWriter.writeFileset(f)).root()
189 
190               // a Dependency object
191             } else if (attrs.isEmpty && e.child.forall {
192               case ele: Elem => ele.label == SL.DependencyStr.toString
193               case t: Text => t.text.trim == ""
194               case c: Comment => true
195               case _ => false
196             }) {
197               val listInJava = e.child.map {
198                 appendNode(path, parent, _) }.filter { _ != null}.asJava
199               ConfigValueFactory.fromIterable(listInJava)
200 
201               // a property map (ie Map[String, String])
202             } else if (attrs.isEmpty && e.child.forall {
203               case el: Elem => el.label == SL.PropertyStr.toString
204               case t: Text => t.text.trim == ""
205               case c: Comment => true
206               case _ => false
207             }) {
208               val pairSeqInScala = (e \ SL.PropertyStr).map { el =>
209                 ((el \ SL.Name).text, (el \ SL.ValueStr).text) }
210               val mapInJava = (pairSeqInScala :+ (
211                 (SL.PropertiesStr.toString, SL.TrueStr.toString))).
212                 toMap.asJava
213               ConfigValueFactory.fromMap(mapInJava)
214 
215               // a list
216             } else if (attrs.isEmpty && e.child.forall {
217               case el: Elem => e.label.startsWith(el.label)
218               case t: Text => t.text.trim == ""
219               case c: Comment => true
220               case _ => false
221             }) {
222               val listInJava = e.child.map {
223                 appendNode(path, parent, _) }.filter { _ != null }.asJava
224               ConfigValueFactory.fromIterable(listInJava)
225 
226             } else {
227               // check if this is a resource transformer
228               val implAttr = (e \ ("@" + SL.Implementation))
229               val mapInScala: Map[String, ConfigValue] =
230                 e.child.map {
231                   case ele: Elem =>
232                     val newPath =
233                       if (path != "") path + "." + ele.label else ele.label
234                     (ele.label, appendNode(newPath, parent, ele))
235                   case t: Text => (SL.TextStr.toString, null)
236                   case _ => (null, null)
237                 }.filter { case(k, v) => v != null }.toMap
238               val map = mapInScala.foldLeft(parent) { case(c, (k, v)) =>
239                 c.withValue(k, v) }.root()
240 
241               // resource transformer
242               if (e.label == SL.Transformer.toString && !implAttr.isEmpty) {
243                 map.withValue(
244                   SL.Implementation,
245                   ConfigValueFactory.fromAnyRef(implAttr.text.trim))
246                 // handle attributes here
247               } else if (!attrs.isEmpty) {
248                 val keys = attrs.keySet.toSeq
249                 attrs.foldLeft(map) { case(c, (n, v)) =>
250                   c.withValue(n, ConfigValueFactory.fromAnyRef(v))
251                 }.withValue(
252                   SL.AttributeKeys,
253                   ConfigValueFactory.fromIterable(keys.asJava))
254               } else {
255                 map
256               }
257             }
258           case t: Text =>
259             if (t.text.trim != "")
260               ConfigValueFactory.fromAnyRef(toAnyRef(t.text.trim))
261             else null
262           case c: Comment =>
263             /*
264             val epty = ConfigFactory.empty()
265             val list = new java.util.ArrayList[java.lang.String]()
266             list.add(c.text)
267             epty.root().withOrigin(epty.origin().withComments(list))
268              */
269             null
270             // unhandled
271           case n: Node =>
272             println("n " + n + " class " + n.getClass.getName)
273             assert(false)
274             null
275         }
276       } else {
277         null
278       }
279 
280     // check if null and start recursing down the heirarchy of children
281     if (elem != null) {
282       // build each child's ConfigObject
283       val childConfs =
284         elem.child.map { ch =>
285           (ch.label, appendNode(ch.label, ConfigFactory.empty(), ch)) }
286       // fold them together into 1 big Config at the paths
287       childConfs.foldLeft(ConfigFactory.empty()) { case(c, (k, v)) =>
288         if (v != null && !k.startsWith("#")) { c.withValue(k, v) } else c }
289     } else {
290       null
291     }
292   }

300   // translate from json string to a jvm string
301   def unencodeJson(s: String): String = {
302     val sb = new StringBuilder()
303     var i = 0
304     // the scala compiler didn't want to do this with a flatMap and a flag so
305     // I had to do it with a while loop, the compiler kept moving the flag
306     // before the flatMap when it was being used in an if condition after the
307     // flatMap even though it was a var
308     while (i < s.length - 1) {
309       s.substring(i, i + 2) match {
310         case "\\b" => sb.append('\b'); i = i + 2
311         case "\\f" => sb.append('\f'); i = i + 2
312         case "\\n" => sb.append('\n'); i = i + 2
313         case "\\r" => sb.append('\r'); i = i + 2
314         case "\\t" => sb.append('\t'); i = i + 2
315         case "\\\"" => sb.append('"'); i = i + 2
316         case "\\\\" => sb.append('\\'); i = i + 2
317         case ch => sb.append(ch(0)); i = i + 1
318       }
319     }
320     if (i == s.length - 1) sb.append(s.last)
321     sb.toString
322   }

346   /**
347     * Translates a Typesafe Config object to a XML Element.  Handles all the
348     * same special cases as elemToConfig.
349     * Supported type conversions: <ul>
350     * <li>Boolean &lt;key&gt;true/false&lt;/key&gt;</li>
351     * <li>Number &lt;key&gt;number&lt;/key&gt;</li>
352     * <li>String &lt;key&gt;value&lt;/key&gt;</li>
353     * <li>Seq[Any] &lt;key(plural)&gt;&lt;key(singular)&gt;value(0)
354     * &lt;/key(singular)&gt;&lt;key(singular)&gt;value(1)&lt;/key(singular)&gt;
355     * ...&lt;/key(plural)&gt;
356     * <li>Properties Identified by the presence of a child at path
357     * &quot;properties&quot; with the value true, otherwise represented as a
358     * Map[String, String] in the Scala code.  It returns the Maven property
359     * XML format which is: &lt;key&gt;&lt;property&gt;&lt;name&gt;property
360     * key1&lt;/name&gt;&lt;value&gt;property
361     * value1&lt;/value&gt;&lt;/property&gt;
362     * &lt;property&gt;&lt;name&gt;property key2&lt;/name&gt;
363     * &lt;value&gt;property value2&lt;/value&gt;&lt;/property&gt;...&lt;/key&gt;
364     * <li>Seq[Dependency] Special case where child elements all have the
365     * keys of a valid &quot;dependency&quot; object in the POM
366     * <li>Archiver Special case where a child key is &quot;archive&quot;
367     * <li>Fileset Special case where a child key is &quot;fileset&quot;
368     * <li>Resource Transformer Special case where a child key is
369     * implementation
370     * <li>Map[String, Any] general/default case, returns whatever
371     * the value translates to inside a XML element labeled by the key
372     * <ul>
373     */
374   def configToElem(config: Config): Elem = {
375     import scala.collection.JavaConverters._
376 
377     // convert the Config value at key into a XML Element
378     def makeElem(key: String, value: ConfigValue): Elem =
379       // check for a child key called archive
380       if (key == SL.Archive.toString) {
381         HoconReader.readArchiver(value.atKey(SL.Archive)).xml
382         // check for child keys with . so they can be split and encoded into
383         // XML correctly
384       } else if (key.contains(".")) {
385         val tokens: Array[String] = key.split('.').reverse
386         val last = tokens.head
387         tokens.tail.foldLeft(makeElem(last, value)) { case(c, token) =>
388           new Elem(null, token, Null, TopScope, c)
389         }
390       } else if (value != null) {
391         value match {
392           // a list so create a parent element called key with children labeled
393           // key in singular form
394           case l: ConfigList =>
395             val childElems = l.asScala.map { it => makeElem(singular(key), it) }
396             new Elem(null, key, Null, TopScope, childElems: _*)
397             // a map so look for special cases and if they don't match convert
398             // as a generic map built using recursion
399           case m: ConfigObject =>
400             val mS = m.asScala
401             var elemKey = key
402             var attrs: MetaData = Null
403             val childElems =
404               if (mS.keySet.find {
405                 _ == SL.PropertiesStr.toString }.isDefined) {
406                 // a Properties object
407                 mS.filter{
408                   _._1 != SL.PropertiesStr.toString }.map { case(k, v) =>
409                     new Elem(
410                       null, SL.PropertyStr, Null, TopScope,
411                       new Elem(null, SL.Name, Null, TopScope, new Text(k)),
412                       new Elem(
413                         null, SL.ValueStr, Null, TopScope,
414                         new Text(removeQuotes(v.render()))))
415                 }.toSeq
416               } else if (mS.values.forall { v =>
417                 v.valueType() == ConfigValueType.STRING }) {
418                 if (!mS.isEmpty &&
419                   mS.keySet.forall { isDependencyProperty(_) }) {
420                   // a Dependency object (for configurations)
421                   elemKey = SL.DependencyStr
422                   mS.map { case(k, v) => new Elem(
423                     null, k, Null, TopScope, Text(removeQuotes(v.render())))
424                   }.toSeq
425                 } else if (mS.keySet.find {
426                   _ == SL.Implementation.toString }.isDefined) {
427                   // a resource transformer
428                   elemKey = SL.Transformer
429                   val impl =
430                     mS.find { case(k, v) =>
431                       k == SL.Implementation.toString }.map { _._2 }.get
432                   attrs = new UnprefixedAttribute(
433                     SL.Implementation, removeQuotes(impl.render()), Null)
434                   mS.filter { case(k, _) =>
435                     k != SL.Implementation.toString }.map { case(k, v) =>
436                       new Elem(
437                         null, k, Null, TopScope, Text(removeQuotes(v.render())))
438                   }.toSeq
439                 } else if (mS.keySet.find {
440                   _ == SL.Archive.toString }.isDefined &&
441                   !mS.keySet.find {
442                     _ == SL.AttributeKeys.toString }.isDefined) {
443                   // a Maven archiver
444                   val arch = HoconReader.readArchiver(m.toConfig())
445                   Seq(arch.xml) ++
446                   mS.filter { _._1 != SL.Archive.toString }.map { case(k, v) =>
447                     makeElem(k, v) }
448 
449                 } else if (mS.keySet.find {
450                   _ == SL.FilesetStr.toString }.isDefined &&
451                    !mS.keySet.find {
452                      _ == SL.AttributeKeys.toString }.isDefined) {
453                   // a Maven fileset
454                   val fs = HoconReader.readFileset(m.toConfig())
455                   Seq(fs.xml) ++
456                   mS.filter { _._1 != SL.FilesetStr.toString
457                   }.map { case(k, v) => makeElem(k, v) }
458 
459                 } else {
460                   // a normal Map (ie non-string value type)
461                   mS.map { case(k, v) => makeElem(k, v) }.toSeq
462                 }
463               } else if (mS.keySet.find {
464                 _ == SL.Archive.toString }.isDefined &&
465                 !mS.keySet.find { _ == SL.AttributeKeys.toString }.isDefined) {
466                 // a Maven archiver
467                 val arch = HoconReader.readArchiver(m.toConfig())
468                 Seq(arch.xml) ++
469                 mS.filter { _._1 != SL.Archive.toString }.map { case(k, v) =>
470                   makeElem(k, v) }
471 
472               } else if (mS.keySet.find {
473                 _ == SL.FilesetStr.toString }.isDefined &&
474                 !mS.keySet.find { _ == SL.AttributeKeys.toString }.isDefined) {
475                 // a Maven fileset
476                 val fs = HoconReader.readFileset(m.toConfig())
477                 Seq(fs.xml) ++
478                 mS.filter { _._1 != SL.FilesetStr.toString }.map { case(k, v) =>
479                   makeElem(k, v) }
480 
481               } else {
482                 // look for attributeKeys and if its present make those
483                 // children into attributes instead of elements
484                 val c = m.toConfig
485                 if (c.hasPath(SL.AttributeKeys)) {
486                   val attrKeys = c.getStringList(SL.AttributeKeys)
487 
488                   if (attrKeys != null && !attrKeys.isEmpty) {
489                     val attrValues: Seq[(String, String)] =
490                       attrKeys.asScala.filter(
491                         _ != SL.Implementation.toString).map { attr =>
492                         (attr, removeQuotes(mS(attr).render())) }
493                     val md: MetaData = Null
494                     attrs =
495                       attrValues.foldLeft(md) { case(n, (k, v)) =>
496                         new UnprefixedAttribute(k, v, n) }
497                     mS.filter { case(k, _) =>
498                       k != SL.AttributeKeys.toString && !attrKeys.contains(k)
499                     }.map { case(k, v) => makeElem(k, v) }.toSeq
500 
501                   } else {
502                     // a normal Map (ie non-string value type)
503                     mS.map { case(k, v) => makeElem(k, v) }.toSeq
504                   }
505                 } else {
506 
507                   // a normal Map (ie non-string value type)
508                   mS.map { case(k, v) => makeElem(k, v) }.toSeq
509                 }
510               }
511 
512             new Elem(null, elemKey, attrs, TopScope, childElems: _*)
513             // a simple value so convert to a string and wrap into a Element
514             // labeled key
515           case v: ConfigValue =>
516             val value = removeQuotes(v.render())
517             new Elem(null, key, Null, TopScope, new Text(value))
518         }
519       } else {
520         null
521       }
522 
523     // start the recurse by visiting the top level children of the config
524     if (config != null) {
525       val children: Seq[(String, ConfigValue)] = config.root().asScala.toSeq
526       val childElems: Seq[Elem] =
527         children.map { entry => makeElem(entry._1, entry._2) }
528       new Elem(null, SL.Configuration, Null, TopScope, childElems: _*)
529     } else {
530       null
531     }
532   }

534   /**
535     * Converts a Json4s object into a Typesafe config in a very direct way
536     * Handles no special cases as there is an isomorphism between Json and
537     * Hocon.  Hocon is really just a special case of Json afterall.
538     */
539   def jsonToConfig(jobject: JObject): Config = {
540     import scala.collection.JavaConverters._
541 
542     def fromJson(v: JValue): Any =
543       if (v != null) {
544         v match {
545           case JNull => null
546           case s: JString =>
547             if (isWindows())
548               s.s.replaceAllLiterally(System.lineSeparator, "\n")
549             else s.s
550           case b: JBool => b.value
551           case d: JDecimal => d.num
552           case d: JDouble => d.num
553           case i: JInt => i.num
554           case o: JObject => ConfigValueFactory.fromMap(fromJsonObject(o))
555           case a: JArray =>
556             val iter = a.arr.map { fromJson(_) }.asJava
557             ConfigValueFactory.fromIterable(iter)
558           case v: JValue => assert(false); null
559         }
560       } else {
561         null
562       }
563 
564     def fromJsonObject(jobj: JObject): java.util.Map[String, Any] =
565       jobj.obj.map { case((key, v)) =>
566         v match {
567           case arr: JArray => (key, fromJson(arr))
568           case obj: JObject => (key, fromJsonObject(obj))
569           case v: JValue => (key, fromJson(v))
570         }
571       }.toMap.asJava
572 
573     if (jobject != null)
574       ConfigValueFactory.fromMap(fromJsonObject(jobject)).toConfig()
575     else null
576   }

578   /**
579     * Converts a Typesafe Config object to a Json4s Json object usually for
580     * serialization.  Does a direct translation.
581     */
582   def configToJson(conf: Config): JObject = {
583     import scala.collection.JavaConverters._
584 
585     def makeJValuePrimitive(value: ConfigValue): JValue =
586       value.unwrapped() match {
587         case null => JNull
588         case b: java.lang.Boolean => JBool(b.booleanValue())
589         case b: java.lang.Byte => JInt(b.toInt)
590         case s: java.lang.Short => JInt(s.toInt)
