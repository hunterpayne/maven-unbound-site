109   /**
110     * Converts an XML Element into a Typesafe Config object.  Generally it
111     * converts Elements into Maps keyed by the names of the child tags.  It
112     * also looks for special patterns of XML to do more specific type
113     * conversions. Supported type conversions: <ul>
114     * <li>Boolean the strings true or false</li>
115     * <li>Number both Integers and Doubles</li>
116     * <li>String Text nodes of the XML document</li>
117     * <li>Seq[Any] Elements whose child labels are the singlar form of elem's
118     * label
119     * <li>Properties Elements whose grand child labels are all name or value
120     * <li>Seq[Dependency] Special case where child elements all have the
121     * label &quot;dependency&quot;
122     * <li>Archiver Special case where elem's label is &quot;archive&quot;
123     * <li>Fileset Special case where elem's label is &quot;fileset&quot;
124     * <li>Resource Transformer Special case where the elem's label is
125     * transformers, the child's label is transformer and there is an attribute
126     * named implementation defined
127     * <li>Map[String, Any] general/default case
128     * <ul>
129     * @see Configuring Plugins
130     * [[https://maven.apache.org/guides/mini/guide-configuring-plugins.html]]
131     * @see Archiver [[http://maven.apache.org/shared/maven-archiver/index.html]]
132     * @see Resource Transformers
133     * [[https://maven.apache.org/plugins-archives/maven-shade-plugin-2.0/examples/resource-transformers.html]]
134     */
135   def elemToConfig(elem: Elem): Config = {
136     import scala.collection.JavaConverters._
137 
138     // handles Booleans and Numbers
139     def toAnyRef(s: String): Any = s match {
140       case "true" => true
141       case "false" => false
142       case s =>
143         if (s.startsWith("0")) {
144           unencodeXml(s)
145         } else {
146           try { s.toInt } catch {
147             case nfe: NumberFormatException =>
148               try { s.toDouble } catch {
149                 case nfe2: NumberFormatException => unencodeXml(s)
150               }
151           }
152         }
153     }
154 
155     // converts a node el at path path to a ConfigValue inside of parent
156     def appendNode(path: String, parent: Config, el: Node): ConfigValue =
157       if (el != null) {
158         el match {
159           case e: Elem =>
160             // a simple string value
161             val attrs = e.attributes.asAttrMap
162             if (e.child.forall { _.isInstanceOf[Text] } && attrs.isEmpty) {
163               ConfigValueFactory.fromAnyRef(toAnyRef(e.text.trim))
164 
165               // an Archiver
166             } else if (e.label == SL.Archive.toString && attrs.isEmpty) {
167               val a = new Archiver(e)
168               ConfigFactory.parseString(JsonWriter.writeArchiver(a)).root()
169 
170               // a fileset
171             } else if (e.label == SL.FilesetStr.toString && attrs.isEmpty) {
172               val f = new Fileset(e)
173               ConfigFactory.parseString(JsonWriter.writeFileset(f)).root()
174 
175               // a Dependency object
176             } else if (attrs.isEmpty && e.child.forall {
177               case ele: Elem => ele.label == SL.DependencyStr.toString
178               case t: Text => t.text.trim == ""
179               case c: Comment => true
180               case _ => false
181             }) {
182               val listInJava = e.child.map {
183                 appendNode(path, parent, _) }.filter { _ != null}.asJava
184               ConfigValueFactory.fromIterable(listInJava)
185 
186               // a property map (ie Map[String, String])
187             } else if (attrs.isEmpty && e.child.forall {
188               case el: Elem => el.label == SL.PropertyStr.toString
189               case t: Text => t.text.trim == ""
190               case c: Comment => true
191               case _ => false
192             }) {
193               val pairSeqInScala = (e \ SL.PropertyStr).map { el =>
194                 ((el \ SL.Name).text, (el \ SL.ValueStr).text) }
195               val mapInJava = (pairSeqInScala :+ (
196                 (SL.PropertiesStr.toString, SL.TrueStr.toString))).
197                 toMap.asJava
198               ConfigValueFactory.fromMap(mapInJava)
199 
200               // a list
201             } else if (attrs.isEmpty && e.child.forall {
202               case el: Elem => e.label.startsWith(el.label)
203               case t: Text => t.text.trim == ""
204               case c: Comment => true
205               case _ => false
206             }) {
207               val listInJava = e.child.map {
208                 appendNode(path, parent, _) }.filter { _ != null }.asJava
209               ConfigValueFactory.fromIterable(listInJava)
210 
211             } else {
212               // check if this is a resource transformer
213               val implAttr = (e \ ("@" + SL.Implementation))
214               val mapInScala: Map[String, ConfigValue] =
215                 e.child.map {
216                   case ele: Elem =>
217                     val newPath =
218                       if (path != "") path + "." + ele.label else ele.label
219                     (ele.label, appendNode(newPath, parent, ele))
220                   case t: Text => (SL.TextStr.toString, null)
221                   case _ => (null, null)
222                 }.filter { case(k, v) => v != null }.toMap
223               val map = mapInScala.foldLeft(parent) { case(c, (k, v)) =>
224                 c.withValue(k, v) }.root()
225 
226               // resource transformer
227               if (e.label == SL.Transformer.toString && !implAttr.isEmpty) {
228                 map.withValue(
229                   SL.Implementation,
230                   ConfigValueFactory.fromAnyRef(implAttr.text.trim))
231                 // handle attributes here
232               } else if (!attrs.isEmpty) {
233                 val keys = attrs.keySet.toSeq
234                 attrs.foldLeft(map) { case(c, (n, v)) =>
235                   c.withValue(n, ConfigValueFactory.fromAnyRef(v))
236                 }.withValue(
237                   SL.AttributeKeys,
238                   ConfigValueFactory.fromIterable(keys.asJava))
239               } else {
240                 map
241               }
242             }
243           case t: Text =>
244             if (t.text.trim != "")
245               ConfigValueFactory.fromAnyRef(toAnyRef(t.text.trim))
246             else null
247           case c: Comment =>
248             /*
249             val epty = ConfigFactory.empty()
250             val list = new java.util.ArrayList[java.lang.String]()
251             list.add(c.text)
252             epty.root().withOrigin(epty.origin().withComments(list))
253              */
254             null
255             // unhandled
256           case n: Node =>
257             println("n " + n + " class " + n.getClass.getName)
258             assert(false)
259             null
260         }
261       } else {
262         null
263       }
264 
265     // check if null and start recursing down the heirarchy of children
266     if (elem != null) {
267       // build each child's ConfigObject
268       val childConfs =
269         elem.child.map { ch =>
270           (ch.label, appendNode(ch.label, ConfigFactory.empty(), ch)) }
271       // fold them together into 1 big Config at the paths
272       childConfs.foldLeft(ConfigFactory.empty()) { case(c, (k, v)) =>
273         if (v != null && !k.startsWith("#")) { c.withValue(k, v) } else c }
274     } else {
275       null
276     }
277   }

285   // translate from json string to a jvm string
286   def unencodeJson(s: String): String = {
287     val sb = new StringBuilder()
288     var i = 0
289     // the scala compiler didn't want to do this with a flatMap and a flag so
290     // I had to do it with a while loop, the compiler kept moving the flag
291     // before the flatMap when it was being used in an if condition after the
292     // flatMap even though it was a var
293     while (i < s.length - 1) {
294       s.substring(i, i + 2) match {
295         case "\\b" => sb.append('\b'); i = i + 2
296         case "\\f" => sb.append('\f'); i = i + 2
297         case "\\n" => sb.append('\n'); i = i + 2
298         case "\\r" => sb.append('\r'); i = i + 2
299         case "\\t" => sb.append('\t'); i = i + 2
300         case "\\\"" => sb.append('"'); i = i + 2
301         case "\\\\" => sb.append('\\'); i = i + 2
302         case ch => sb.append(ch(0)); i = i + 1
303       }
304     }
305     if (i == s.length - 1) sb.append(s.last)
306     sb.toString
307   }

331   /**
332     * Translates a Typesafe Config object to a XML Element.  Handles all the
333     * same special cases as elemToConfig.
334     * Supported type conversions: <ul>
335     * <li>Boolean &lt;key&gt;true/false&lt;/key&gt;</li>
336     * <li>Number &lt;key&gt;number&lt;/key&gt;</li>
337     * <li>String &lt;key&gt;value&lt;/key&gt;</li>
338     * <li>Seq[Any] &lt;key(plural)&gt;&lt;key(singular)&gt;value(0)
339     * &lt;/key(singular)&gt;&lt;key(singular)&gt;value(1)&lt;/key(singular)&gt;
340     * ...&lt;/key(plural)&gt;
341     * <li>Properties Identified by the presence of a child at path
342     * &quot;properties&quot; with the value true, otherwise represented as a
343     * Map[String, String] in the Scala code.  It returns the Maven property
344     * XML format which is: &lt;key&gt;&lt;property&gt;&lt;name&gt;property
345     * key1&lt;/name&gt;&lt;value&gt;property
346     * value1&lt;/value&gt;&lt;/property&gt;
347     * &lt;property&gt;&lt;name&gt;property key2&lt;/name&gt;
348     * &lt;value&gt;property value2&lt;/value&gt;&lt;/property&gt;...&lt;/key&gt;
349     * <li>Seq[Dependency] Special case where child elements all have the
350     * keys of a valid &quot;dependency&quot; object in the POM
351     * <li>Archiver Special case where a child key is &quot;archive&quot;
352     * <li>Fileset Special case where a child key is &quot;fileset&quot;
353     * <li>Resource Transformer Special case where a child key is
354     * implementation
355     * <li>Map[String, Any] general/default case, returns whatever
356     * the value translates to inside a XML element labeled by the key
357     * <ul>
358     */
359   def configToElem(config: Config): Elem = {
360     import scala.collection.JavaConverters._
361 
362     // convert the Config value at key into a XML Element
363     def makeElem(key: String, value: ConfigValue): Elem =
364       // check for a child key called archive
365       if (key == SL.Archive.toString) {
366         HoconReader.readArchiver(value.atKey(SL.Archive)).xml
367         // check for child keys with . so they can be split and encoded into
368         // XML correctly
369       } else if (key.contains(".")) {
370         val tokens: Array[String] = key.split('.').reverse
371         val last = tokens.head
372         tokens.tail.foldLeft(makeElem(last, value)) { case(c, token) =>
373           new Elem(null, token, Null, TopScope, c)
374         }
375       } else if (value != null) {
376         value match {
377           // a list so create a parent element called key with children labeled
378           // key in singular form
379           case l: ConfigList =>
380             val childElems = l.asScala.map { it => makeElem(singular(key), it) }
381             new Elem(null, key, Null, TopScope, childElems: _*)
382             // a map so look for special cases and if they don't match convert
383             // as a generic map built using recursion
384           case m: ConfigObject =>
385             val mS = m.asScala
386             var elemKey = key
387             var attrs: MetaData = Null
388             val childElems =
389               if (mS.keySet.find {
390                 _ == SL.PropertiesStr.toString }.isDefined) {
391                 // a Properties object
392                 mS.filter{
393                   _._1 != SL.PropertiesStr.toString }.map { case(k, v) =>
394                     new Elem(
395                       null, SL.PropertyStr, Null, TopScope,
396                       new Elem(null, SL.Name, Null, TopScope, new Text(k)),
397                       new Elem(
398                         null, SL.ValueStr, Null, TopScope,
399                         new Text(removeQuotes(v.render()))))
400                 }.toSeq
401               } else if (mS.values.forall { v =>
402                 v.valueType() == ConfigValueType.STRING }) {
403                 if (!mS.isEmpty &&
404                   mS.keySet.forall { isDependencyProperty(_) }) {
405                   // a Dependency object (for configurations)
406                   elemKey = SL.DependencyStr
407                   mS.map { case(k, v) => new Elem(
408                     null, k, Null, TopScope, Text(removeQuotes(v.render())))
409                   }.toSeq
410                 } else if (mS.keySet.find {
411                   _ == SL.Implementation.toString }.isDefined) {
412                   // a resource transformer
413                   elemKey = SL.Transformer
414                   val impl =
415                     mS.find { case(k, v) =>
416                       k == SL.Implementation.toString }.map { _._2 }.get
417                   attrs = new UnprefixedAttribute(
418                     SL.Implementation, removeQuotes(impl.render()), Null)
419                   mS.filter { case(k, _) =>
420                     k != SL.Implementation.toString }.map { case(k, v) =>
421                       new Elem(
422                         null, k, Null, TopScope, Text(removeQuotes(v.render())))
423                   }.toSeq
424                 } else if (mS.keySet.find {
425                   _ == SL.Archive.toString }.isDefined &&
426                   !mS.keySet.find {
427                     _ == SL.AttributeKeys.toString }.isDefined) {
428                   // a Maven archiver
429                   val arch = HoconReader.readArchiver(m.toConfig())
430                   Seq(arch.xml) ++
431                   mS.filter { _._1 != SL.Archive.toString }.map { case(k, v) =>
432                     makeElem(k, v) }
433 
434                 } else if (mS.keySet.find {
435                   _ == SL.FilesetStr.toString }.isDefined &&
436                    !mS.keySet.find {
437                      _ == SL.AttributeKeys.toString }.isDefined) {
438                   // a Maven fileset
439                   val fs = HoconReader.readFileset(m.toConfig())
440                   Seq(fs.xml) ++
441                   mS.filter { _._1 != SL.FilesetStr.toString
442                   }.map { case(k, v) => makeElem(k, v) }
443 
444                 } else {
445                   // a normal Map (ie non-string value type)
446                   mS.map { case(k, v) => makeElem(k, v) }.toSeq
447                 }
448               } else if (mS.keySet.find {
449                 _ == SL.Archive.toString }.isDefined &&
450                 !mS.keySet.find { _ == SL.AttributeKeys.toString }.isDefined) {
451                 // a Maven archiver
452                 val arch = HoconReader.readArchiver(m.toConfig())
453                 Seq(arch.xml) ++
454                 mS.filter { _._1 != SL.Archive.toString }.map { case(k, v) =>
455                   makeElem(k, v) }
456 
457               } else if (mS.keySet.find {
458                 _ == SL.FilesetStr.toString }.isDefined &&
459                 !mS.keySet.find { _ == SL.AttributeKeys.toString }.isDefined) {
460                 // a Maven fileset
461                 val fs = HoconReader.readFileset(m.toConfig())
462                 Seq(fs.xml) ++
463                 mS.filter { _._1 != SL.FilesetStr.toString }.map { case(k, v) =>
464                   makeElem(k, v) }
465 
466               } else {
467                 // look for attributeKeys and if its present make those
468                 // children into attributes instead of elements
469                 val c = m.toConfig
470                 if (c.hasPath(SL.AttributeKeys)) {
471                   val attrKeys = c.getStringList(SL.AttributeKeys)
472 
473                   if (attrKeys != null && !attrKeys.isEmpty) {
474                     val attrValues: Seq[(String, String)] =
475                       attrKeys.asScala.filter(
476                         _ != SL.Implementation.toString).map { attr =>
477                         (attr, removeQuotes(mS(attr).render())) }
478                     val md: MetaData = Null
479                     attrs =
480                       attrValues.foldLeft(md) { case(n, (k, v)) =>
481                         new UnprefixedAttribute(k, v, n) }
482                     mS.filter { case(k, _) =>
483                       k != SL.AttributeKeys.toString && !attrKeys.contains(k)
484                     }.map { case(k, v) => makeElem(k, v) }.toSeq
485 
486                   } else {
487                     // a normal Map (ie non-string value type)
488                     mS.map { case(k, v) => makeElem(k, v) }.toSeq
489                   }
490                 } else {
491 
492                   // a normal Map (ie non-string value type)
493                   mS.map { case(k, v) => makeElem(k, v) }.toSeq
494                 }
495               }
496 
497             new Elem(null, elemKey, attrs, TopScope, childElems: _*)
498             // a simple value so convert to a string and wrap into a Element
499             // labeled key
500           case v: ConfigValue =>
501             val value = removeQuotes(v.render())
502             new Elem(null, key, Null, TopScope, new Text(value))
503         }
504       } else {
505         null
506       }
507 
508     // start the recurse by visiting the top level children of the config
509     if (config != null) {
510       val children: Seq[(String, ConfigValue)] = config.root().asScala.toSeq
511       val childElems: Seq[Elem] =
512         children.map { entry => makeElem(entry._1, entry._2) }
513       new Elem(null, SL.Configuration, Null, TopScope, childElems: _*)
514     } else {
515       null
516     }
517   }

519   /**
520     * Converts a Json4s object into a Typesafe config in a very direct way
521     * Handles no special cases as there is an isomorphism between Json and
522     * Hocon.  Hocon is really just a special case of Json afterall.
523     */
524   def jsonToConfig(jobject: JObject): Config = {
525     import scala.collection.JavaConverters._
526 
527     def fromJson(v: JValue): Any =
528       if (v != null) {
529         v match {
530           case JNull => null
531           case s: JString => s.s
532           case b: JBool => b.value
533           case d: JDecimal => d.num
534           case d: JDouble => d.num
535           case i: JInt => i.num
536           case o: JObject => ConfigValueFactory.fromMap(fromJsonObject(o))
537           case a: JArray =>
538             val iter = a.arr.map { fromJson(_) }.asJava
539             ConfigValueFactory.fromIterable(iter)
540           case v: JValue => assert(false); null
541         }
542       } else {
543         null
544       }
545 
546     def fromJsonObject(jobj: JObject): java.util.Map[String, Any] =
547       jobj.obj.map { case((key, v)) =>
548         v match {
549           case arr: JArray => (key, fromJson(arr))
550           case obj: JObject => (key, fromJsonObject(obj))
551           case v: JValue => (key, fromJson(v))
552         }
553       }.toMap.asJava
554 
555     if (jobject != null)
556       ConfigValueFactory.fromMap(fromJsonObject(jobject)).toConfig()
557     else null
558   }

560   /**
561     * Converts a Typesafe Config object to a Json4s Json object usually for
562     * serialization.  Does a direct translation.
563     */
564   def configToJson(conf: Config): JObject = {
565     import scala.collection.JavaConverters._
566 
567     def makeJValuePrimitive(value: ConfigValue): JValue =
568       value.unwrapped() match {
569         case null => JNull
570         case b: java.lang.Boolean => JBool(b.booleanValue())
571         case b: java.lang.Byte => JInt(b.toInt)
572         case s: java.lang.Short => JInt(s.toInt)
573         case i: java.lang.Integer => JInt(i.toInt)
574         case l: java.lang.Long => JDecimal(new java.math.BigDecimal(l.toLong))
575         case f: java.lang.Float => JDouble(f.toDouble)
576         case d: java.lang.Double => JDouble(d.toDouble)
577         case s: String => JString(s)
578         case _ => assert(false); null
579       }
580 
581     def makeJValue(value: ConfigValue): JValue = {
582       value match {
583         case l: ConfigList =>
584           JArray(l.asScala.map { it => makeJValue(it) }.toList)
585         case m: ConfigObject =>
586           JObject(m.asScala.map { it => (it._1, makeJValue(it._2)) }.toList)
587         case _ => makeJValuePrimitive(value)
588       }
