86   /**
87     * Converts an XML Element into a Typesafe Config object.  Generally it
88     * converts Elements into Maps keyed by the names of the child tags.  It
89     * also looks for special patterns of XML to do more specific type
90     * conversions. Supported type conversions: <ul>
91     * <li>Boolean the strings true or false</li>
92     * <li>Number both Integers and Doubles</li>
93     * <li>String Text nodes of the XML document</li>
94     * <li>Seq[Any] Elements whose child labels are the singlar form of elem's
95     * label
96     * <li>Properties Elements whose grand child labels are all name or value
97     * <li>Seq[Dependency] Special case where child elements all have the
98     * label &quot;dependency&quot;
99     * <li>Archiver Special case where elem's label is &quot;archive&quot;
100     * <li>Fileset Special case where elem's label is &quot;fileset&quot;
101     * <li>Resource Transformer Special case where the elem's label is
102     * transformers, the child's label is transformer and there is an attribute
103     * named implementation defined
104     * <li>Map[String, Any] general/default case
105     * <ul>
106     * @see Configuring Plugins
107     * [[https://maven.apache.org/guides/mini/guide-configuring-plugins.html]]
108     * @see Archiver [[http://maven.apache.org/shared/maven-archiver/index.html]]
109     * @see Resource Transformers
110     * [[https://maven.apache.org/plugins-archives/maven-shade-plugin-2.0/examples/resource-transformers.html]]
111     */
112   def elemToConfig(elem: Elem): Config = {
113     import scala.collection.JavaConverters._
114 
115     // handles Booleans and Numbers
116     def toAnyRef(s: String): Any = s match {
117       case "true" => true
118       case "false" => false
119       case s =>
120         if (s.startsWith("0")) {
121           unencodeXml(s)
122         } else {
123           try { s.toInt } catch {
124             case nfe: NumberFormatException =>
125               try { s.toDouble } catch {
126                 case nfe2: NumberFormatException => unencodeXml(s)
127               }
128           }
129         }
130     }
131 
132     // converts a node el at path path to a ConfigValue inside of parent
133     def appendNode(path: String, parent: Config, el: Node): ConfigValue =
134       if (el != null) {
135         el match {
136           case e: Elem =>
137             // a simple string value
138             val attrs = e.attributes.asAttrMap
139             if (e.child.forall { _.isInstanceOf[Text] } && attrs.isEmpty) {
140               ConfigValueFactory.fromAnyRef(toAnyRef(e.text.trim))
141 
142               // an Archiver
143             } else if (e.label == SL.Archive.toString && attrs.isEmpty) {
144               val a = new Archiver(e)
145               ConfigFactory.parseString(JsonWriter.writeArchiver(a)).root()
146 
147               // a fileset
148             } else if (e.label == SL.FilesetStr.toString && attrs.isEmpty) {
149               val f = new Fileset(e)
150               ConfigFactory.parseString(JsonWriter.writeFileset(f)).root()
151 
152               // a Dependency object
153             } else if (attrs.isEmpty && e.child.forall {
154               case ele: Elem => ele.label == SL.DependencyStr.toString
155               case t: Text => t.text.trim == ""
156               case _ => false
157             }) {
158               val listInJava = e.child.map {
159                 appendNode(path, parent, _) }.filter { _ != null}.asJava
160               ConfigValueFactory.fromIterable(listInJava)
161 
162               // a property map (ie Map[String, String])
163             } else if (attrs.isEmpty && e.child.forall {
164               case el: Elem => el.label == SL.PropertyStr.toString
165               case t: Text => t.text.trim == ""
166               case _ => false
167             }) {
168               val pairSeqInScala = (e \ SL.PropertyStr).map { el =>
169                 ((el \ SL.Name).text, (el \ SL.ValueStr).text) }
170               val mapInJava = (pairSeqInScala :+ (
171                 (SL.PropertiesStr.toString, SL.TrueStr.toString))).
172                 toMap.asJava
173               ConfigValueFactory.fromMap(mapInJava)
174 
175               // a list
176             } else if (attrs.isEmpty && e.child.forall {
177               case el: Elem => e.label.startsWith(el.label)
178               case t: Text => t.text.trim == ""
179               case _ => false
180             }) {
181               val listInJava = e.child.map {
182                 appendNode(path, parent, _) }.filter { _ != null }.asJava
183               ConfigValueFactory.fromIterable(listInJava)
184 
185             } else {
186               // check if this is a resource transformer
187               val implAttr = (e \ ("@" + SL.Implementation))
188               val mapInScala: Map[String, ConfigValue] =
189                 e.child.map {
190                   case ele: Elem =>
191                     val newPath =
192                       if (path != "") path + "." + ele.label else ele.label
193                     (ele.label, appendNode(newPath, parent, ele))
194                   case t: Text => (SL.TextStr.toString, null)
195                 }.filter { case(k, v) => v != null }.toMap
196               val map = mapInScala.foldLeft(parent) { case(c, (k, v)) =>
197                 c.withValue(k, v) }.root()
198 
199               // resource transformer
200               if (e.label == SL.Transformer.toString && !implAttr.isEmpty) {
201                 map.withValue(
202                   SL.Implementation,
203                   ConfigValueFactory.fromAnyRef(implAttr.text.trim))
204                 // handle attributes here
205               } else if (!attrs.isEmpty) {
206                 val keys = attrs.keySet.toSeq
207                 attrs.foldLeft(map) { case(c, (n, v)) =>
208                   c.withValue(n, ConfigValueFactory.fromAnyRef(v))
209                 }.withValue(
210                   SL.AttributeKeys,
211                   ConfigValueFactory.fromIterable(keys.asJava))
212               } else {
213                 map
214               }
215             }
216           case t: Text =>
217             if (t.text.trim != "")
218               ConfigValueFactory.fromAnyRef(toAnyRef(t.text.trim))
219             else null
220           case c: Comment =>
221             val epty = ConfigFactory.empty()
222             val list = new java.util.ArrayList[java.lang.String]()
223             list.add(c.text)
224             epty.root().withOrigin(epty.origin().withComments(list))
225             // unhandled
226           case n: Node =>
227             println("n " + n + " class " + n.getClass.getName)
228             assert(false)
229             null
230         }
231       } else {
232         null
233       }
234 
235     // check if null and start recursing down the heirarchy of children
236     if (elem != null) {
237       val empty = ConfigFactory.empty()
238       // build each child's ConfigObject
239       val childConfs =
240         elem.child.map { ch => (ch.label, appendNode(ch.label, empty, ch)) }
241       // fold them together into 1 big Config at the paths
242       childConfs.foldLeft(ConfigFactory.empty()) { case(c, (k, v)) =>
243         if (v != null && !k.startsWith("#")) { c.withValue(k, v) } else c }
244     } else {
245       null
246     }
247   }

255   // translate from json string to a jvm string
256   def unencodeJson(s: String): String = {
257     val sb = new StringBuilder()
258     var i = 0
259     // the scala compiler didn't want to do this with a flatMap and a flag so
260     // I had to do it with a while loop, the compiler kept moving the flag
261     // before the flatMap when it was being used in an if condition after the
262     // flatMap even though it was a var
263     while (i < s.length - 1) {
264       s.substring(i, i + 2) match {
265         case "\\b" => sb.append('\b'); i = i + 2
266         case "\\f" => sb.append('\f'); i = i + 2
267         case "\\n" => sb.append('\n'); i = i + 2
268         case "\\r" => sb.append('\r'); i = i + 2
269         case "\\t" => sb.append('\t'); i = i + 2
270         case "\\\"" => sb.append('"'); i = i + 2
271         case "\\\\" => sb.append('\\'); i = i + 2
272         case ch => sb.append(ch(0)); i = i + 1
273       }
274     }
275     if (i == s.length - 1) sb.append(s.last)
276     sb.toString
277   }

301   /**
302     * Translates a Typesafe Config object to a XML Element.  Handles all the
303     * same special cases as elemToConfig.
304     * Supported type conversions: <ul>
305     * <li>Boolean &lt;key&gt;true/false&lt;/key&gt;</li>
306     * <li>Number &lt;key&gt;number&lt;/key&gt;</li>
307     * <li>String &lt;key&gt;value&lt;/key&gt;</li>
308     * <li>Seq[Any] &lt;key(plural)&gt;&lt;key(singular)&gt;value(0)
309     * &lt;/key(singular)&gt;&lt;key(singular)&gt;value(1)&lt;/key(singular)&gt;
310     * ...&lt;/key(plural)&gt;
311     * <li>Properties Identified by the presence of a child at path
312     * &quot;properties&quot; with the value true, otherwise represented as a
313     * Map[String, String] in the Scala code.  It returns the Maven property
314     * XML format which is: &lt;key&gt;&lt;property&gt;&lt;name&gt;property
315     * key1&lt;/name&gt;&lt;value&gt;property
316     * value1&lt;/value&gt;&lt;/property&gt;
317     * &lt;property&gt;&lt;name&gt;property key2&lt;/name&gt;
318     * &lt;value&gt;property value2&lt;/value&gt;&lt;/property&gt;...&lt;/key&gt;
319     * <li>Seq[Dependency] Special case where child elements all have the
320     * keys of a valid &quot;dependency&quot; object in the POM
321     * <li>Archiver Special case where a child key is &quot;archive&quot;
322     * <li>Fileset Special case where a child key is &quot;fileset&quot;
323     * <li>Resource Transformer Special case where a child key is
324     * implementation
325     * <li>Map[String, Any] general/default case, returns whatever
326     * the value translates to inside a XML element labeled by the key
327     * <ul>
328     */
329   def configToElem(config: Config): Elem = {
330     import scala.collection.JavaConverters._
331 
332     // convert the Config value at key into a XML Element
333     def makeElem(key: String, value: ConfigValue): Elem =
334       // check for a child key called archive

338         // XML correctly

340         val tokens: Array[String] = key.split('.').reverse
341         val last = tokens.head
342         tokens.tail.foldLeft(makeElem(last, value)) { case(c, token) =>
343           new Elem(null, token, Null, TopScope, c)
344         }
345       } else if (value != null) {
346         value match {
347           // a list so create a parent element called key with children labeled
348           // key in singular form

350             val childElems = l.asScala.map { it => makeElem(singular(key), it) }

355             val mS = m.asScala

366                       new Elem(null, SL.Name, Null, TopScope, new Text(k)),

370                 }.toSeq
371               } else if (mS.values.forall { v =>
372                 v.valueType() == ConfigValueType.STRING }) {
373                 if (mS.keySet.forall { isDependencyProperty(_) }) {

375                   elemKey = SL.DependencyStr
376                   mS.map { case(k, v) => new Elem(
377                     null, k, Null, TopScope, Text(removeQuotes(v.render())))
378                   }.toSeq
379                 } else if (mS.keySet.find {
380                   _ == SL.Implementation.toString }.isDefined) {
381                   // a resource transformer

480   /**
481     * Converts a Json4s object into a Typesafe config in a very direct way
482     * Handles no special cases as there is an isomorphism between Json and
483     * Hocon.  Hocon is really just a special case of Json afterall.
484     */
485   def jsonToConfig(jobject: JObject): Config = {
486     import scala.collection.JavaConverters._
487 
488     def fromJson(v: JValue): Any =
489       if (v != null) {
490         v match {
491           case JNull => null
492           case s: JString => s.s
493           case b: JBool => b.value
494           case d: JDecimal => d.num
495           case d: JDouble => d.num
496           case i: JInt => i.num
497           case o: JObject => ConfigValueFactory.fromMap(fromJsonObject(o))
498           case a: JArray =>
499             val iter = a.arr.map { fromJson(_) }.asJava
500             ConfigValueFactory.fromIterable(iter)
501           case v: JValue => assert(false); null
502         }
503       } else {
504         null
505       }
506 
507     def fromJsonObject(jobj: JObject): java.util.Map[String, Any] =
508       jobj.obj.map { case((key, v)) =>
509         v match {
510           case arr: JArray => (key, fromJson(arr))
511           case obj: JObject => (key, fromJsonObject(obj))
512           case v: JValue => (key, fromJson(v))
513         }
514       }.toMap.asJava
515 
516     if (jobject != null)
517       ConfigValueFactory.fromMap(fromJsonObject(jobject)).toConfig()
518     else null
519   }

521   /**
522     * Converts a Typesafe Config object to a Json4s Json object usually for
523     * serialization.  Does a direct translation.
524     */
525   def configToJson(conf: Config): JObject = {
526     import scala.collection.JavaConverters._
527 
528     def makeJValuePrimitive(value: ConfigValue): JValue =
529       value.unwrapped() match {
530         case null => JNull
531         case b: java.lang.Boolean => JBool(b.booleanValue())
532         case b: java.lang.Byte => JInt(b.toInt)
533         case s: java.lang.Short => JInt(s.toInt)
534         case i: java.lang.Integer => JInt(i.toInt)
535         case l: java.lang.Long => JDecimal(new java.math.BigDecimal(l.toLong))
536         case f: java.lang.Float => JDouble(f.toDouble)
537         case d: java.lang.Double => JDouble(d.toDouble)
538         case s: String => JString(s)
539         case _ => assert(false); null
540       }
541 
542     def makeJValue(value: ConfigValue): JValue = {
543       value match {
544         case l: ConfigList =>
545           JArray(l.asScala.map { it => makeJValue(it) }.toList)
546         case m: ConfigObject =>
547           JObject(m.asScala.map { it => (it._1, makeJValue(it._2)) }.toList)
548         case _ => makeJValuePrimitive(value)
549       }
550     }
551 
552     if (conf != null) {
553       val root = conf.root()
554       val childKeys: Seq[String] = root.keySet().asScala.toSeq
555       val childElems: Seq[JField] =
556         childKeys.map { key => (key, makeJValue(root.get(key))) }
557       new JObject(childElems.toList)
558     } else {
559       null
560     }
561   }
