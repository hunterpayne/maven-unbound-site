86   /**
87     * Converts an XML Element into a Typesafe Config object.  Generally it
88     * converts Elements into Maps keyed by the names of the child tags.  It
89     * also looks for special patterns of XML to do more specific type
90     * conversions. Supported type conversions: <ul>
91     * <li>Boolean the strings true or false</li>
92     * <li>Number both Integers and Doubles</li>
93     * <li>String Text nodes of the XML document</li>
94     * <li>Seq[Any] Elements whose child labels are the singlar form of elem's
95     * label
96     * <li>Properties Elements whose grand child labels are all name or value
97     * <li>Seq[Dependency] Special case where child elements all have the
98     * label &quot;dependency&quot;
99     * <li>Archiver Special case where elem's label is &quot;archive&quot;
100     * <li>Fileset Special case where elem's label is &quot;fileset&quot;
101     * <li>Resource Transformer Special case where the elem's label is
102     * transformers, the child's label is transformer and there is an attribute
103     * named implementation defined
104     * <li>Map[String, Any] general/default case
105     * <ul>
106     * @see Configuring Plugins
107     * [[https://maven.apache.org/guides/mini/guide-configuring-plugins.html]]
108     * @see Archiver [[http://maven.apache.org/shared/maven-archiver/index.html]]
109     * @see Resource Transformers
110     * [[https://maven.apache.org/plugins-archives/maven-shade-plugin-2.0/examples/resource-transformers.html]]
111     */
112   def elemToConfig(elem: Elem): Config = {
113     import scala.collection.JavaConverters._
114 
115     // handles Booleans and Numbers
116     def toAnyRef(s: String): Any = s match {
117       case "true" => true
118       case "false" => false
119       case s =>
120         if (s.startsWith("0")) {
121           unencodeXml(s)
122         } else {
123           try { s.toInt } catch {
124             case nfe: NumberFormatException =>
125               try { s.toDouble } catch {
126                 case nfe2: NumberFormatException => unencodeXml(s)
127               }
128           }
129         }
130     }
131 
132     // converts a node el at path path to a ConfigValue inside of parent
133     def appendNode(path: String, parent: Config, el: Node): ConfigValue =
134       if (el != null) {
135         el match {
136           case e: Elem =>
137             // a simple string value
138             val attrs = e.attributes.asAttrMap
139             if (e.child.forall { _.isInstanceOf[Text] } && attrs.isEmpty) {
140               ConfigValueFactory.fromAnyRef(toAnyRef(e.text.trim))
141 
142               // an Archiver
143             } else if (e.label == SL.Archive.toString && attrs.isEmpty) {
144               val a = new Archiver(e)
145               ConfigFactory.parseString(JsonWriter.writeArchiver(a)).root()
146 
147               // a fileset
148             } else if (e.label == SL.FilesetStr.toString && attrs.isEmpty) {
149               val f = new Fileset(e)
150               ConfigFactory.parseString(JsonWriter.writeFileset(f)).root()
151 
152               // a Dependency object
153             } else if (attrs.isEmpty && e.child.forall {
154               case ele: Elem => ele.label == SL.DependencyStr.toString
155               case t: Text => t.text.trim == ""
156               case c: Comment => true
157               case _ => false
158             }) {
159               val listInJava = e.child.map {
160                 appendNode(path, parent, _) }.filter { _ != null}.asJava
161               ConfigValueFactory.fromIterable(listInJava)
162 
163               // a property map (ie Map[String, String])
164             } else if (attrs.isEmpty && e.child.forall {
165               case el: Elem => el.label == SL.PropertyStr.toString
166               case t: Text => t.text.trim == ""
167               case c: Comment => true
168               case _ => false
169             }) {
170               val pairSeqInScala = (e \ SL.PropertyStr).map { el =>
171                 ((el \ SL.Name).text, (el \ SL.ValueStr).text) }
172               val mapInJava = (pairSeqInScala :+ (
173                 (SL.PropertiesStr.toString, SL.TrueStr.toString))).
174                 toMap.asJava
175               ConfigValueFactory.fromMap(mapInJava)
176 
177               // a list
178             } else if (attrs.isEmpty && e.child.forall {
179               case el: Elem => e.label.startsWith(el.label)
180               case t: Text => t.text.trim == ""
181               case c: Comment => true
182               case _ => false
183             }) {
184               val listInJava = e.child.map {
185                 appendNode(path, parent, _) }.filter { _ != null }.asJava
186               ConfigValueFactory.fromIterable(listInJava)
187 
188             } else {
189               // check if this is a resource transformer
190               val implAttr = (e \ ("@" + SL.Implementation))
191               val mapInScala: Map[String, ConfigValue] =
192                 e.child.map {
193                   case ele: Elem =>
194                     val newPath =
195                       if (path != "") path + "." + ele.label else ele.label
196                     (ele.label, appendNode(newPath, parent, ele))
197                   case t: Text => (SL.TextStr.toString, null)
198                   case _ => (null, null)
199                 }.filter { case(k, v) => v != null }.toMap
200               val map = mapInScala.foldLeft(parent) { case(c, (k, v)) =>
201                 c.withValue(k, v) }.root()
202 
203               // resource transformer
204               if (e.label == SL.Transformer.toString && !implAttr.isEmpty) {
205                 map.withValue(
206                   SL.Implementation,
207                   ConfigValueFactory.fromAnyRef(implAttr.text.trim))
208                 // handle attributes here
209               } else if (!attrs.isEmpty) {
210                 val keys = attrs.keySet.toSeq
211                 attrs.foldLeft(map) { case(c, (n, v)) =>
212                   c.withValue(n, ConfigValueFactory.fromAnyRef(v))
213                 }.withValue(
214                   SL.AttributeKeys,
215                   ConfigValueFactory.fromIterable(keys.asJava))
216               } else {
217                 map
218               }
219             }
220           case t: Text =>
221             if (t.text.trim != "")
222               ConfigValueFactory.fromAnyRef(toAnyRef(t.text.trim))
223             else null
224           case c: Comment =>
225             /*
226             val epty = ConfigFactory.empty()
227             val list = new java.util.ArrayList[java.lang.String]()
228             list.add(c.text)
229             epty.root().withOrigin(epty.origin().withComments(list))
230              */
231             null
232             // unhandled
233           case n: Node =>
234             println("n " + n + " class " + n.getClass.getName)
235             assert(false)
236             null
237         }
238       } else {
239         null
240       }
241 
242     // check if null and start recursing down the heirarchy of children
243     if (elem != null) {
244       // build each child's ConfigObject
245       val childConfs =
246         elem.child.map { ch =>
247           (ch.label, appendNode(ch.label, ConfigFactory.empty(), ch)) }
248       // fold them together into 1 big Config at the paths
249       childConfs.foldLeft(ConfigFactory.empty()) { case(c, (k, v)) =>
250         if (v != null && !k.startsWith("#")) { c.withValue(k, v) } else c }
251     } else {
252       null
253     }
254   }

262   // translate from json string to a jvm string
263   def unencodeJson(s: String): String = {
264     val sb = new StringBuilder()
265     var i = 0
266     // the scala compiler didn't want to do this with a flatMap and a flag so
267     // I had to do it with a while loop, the compiler kept moving the flag
268     // before the flatMap when it was being used in an if condition after the
269     // flatMap even though it was a var
270     while (i < s.length - 1) {
271       s.substring(i, i + 2) match {
272         case "\\b" => sb.append('\b'); i = i + 2
273         case "\\f" => sb.append('\f'); i = i + 2
274         case "\\n" => sb.append('\n'); i = i + 2
275         case "\\r" => sb.append('\r'); i = i + 2
276         case "\\t" => sb.append('\t'); i = i + 2
277         case "\\\"" => sb.append('"'); i = i + 2
278         case "\\\\" => sb.append('\\'); i = i + 2
279         case ch => sb.append(ch(0)); i = i + 1
280       }
281     }
282     if (i == s.length - 1) sb.append(s.last)
283     sb.toString
284   }

308   /**
309     * Translates a Typesafe Config object to a XML Element.  Handles all the
310     * same special cases as elemToConfig.
311     * Supported type conversions: <ul>
312     * <li>Boolean &lt;key&gt;true/false&lt;/key&gt;</li>
313     * <li>Number &lt;key&gt;number&lt;/key&gt;</li>
314     * <li>String &lt;key&gt;value&lt;/key&gt;</li>
315     * <li>Seq[Any] &lt;key(plural)&gt;&lt;key(singular)&gt;value(0)
316     * &lt;/key(singular)&gt;&lt;key(singular)&gt;value(1)&lt;/key(singular)&gt;
317     * ...&lt;/key(plural)&gt;
318     * <li>Properties Identified by the presence of a child at path
319     * &quot;properties&quot; with the value true, otherwise represented as a
320     * Map[String, String] in the Scala code.  It returns the Maven property
321     * XML format which is: &lt;key&gt;&lt;property&gt;&lt;name&gt;property
322     * key1&lt;/name&gt;&lt;value&gt;property
323     * value1&lt;/value&gt;&lt;/property&gt;
324     * &lt;property&gt;&lt;name&gt;property key2&lt;/name&gt;
325     * &lt;value&gt;property value2&lt;/value&gt;&lt;/property&gt;...&lt;/key&gt;
326     * <li>Seq[Dependency] Special case where child elements all have the
327     * keys of a valid &quot;dependency&quot; object in the POM
328     * <li>Archiver Special case where a child key is &quot;archive&quot;
329     * <li>Fileset Special case where a child key is &quot;fileset&quot;
330     * <li>Resource Transformer Special case where a child key is
331     * implementation
332     * <li>Map[String, Any] general/default case, returns whatever
333     * the value translates to inside a XML element labeled by the key
334     * <ul>
335     */
336   def configToElem(config: Config): Elem = {
337     import scala.collection.JavaConverters._
338 
339     // convert the Config value at key into a XML Element
340     def makeElem(key: String, value: ConfigValue): Elem =
341       // check for a child key called archive
342       if (key == SL.Archive.toString) {
343         HoconReader.readArchiver(value.atKey(SL.Archive)).xml
344         // check for child keys with . so they can be split and encoded into
345         // XML correctly
346       } else if (key.contains(".")) {
347         val tokens: Array[String] = key.split('.').reverse
348         val last = tokens.head
349         tokens.tail.foldLeft(makeElem(last, value)) { case(c, token) =>
350           new Elem(null, token, Null, TopScope, c)
351         }
352       } else if (value != null) {
353         value match {
354           // a list so create a parent element called key with children labeled
355           // key in singular form
356           case l: ConfigList =>
357             val childElems = l.asScala.map { it => makeElem(singular(key), it) }
358             new Elem(null, key, Null, TopScope, childElems: _*)
359             // a map so look for special cases and if they don't match convert
360             // as a generic map built using recursion
361           case m: ConfigObject =>
362             val mS = m.asScala
363             var elemKey = key
364             var attrs: MetaData = Null
365             val childElems =
366               if (mS.keySet.find {
367                 _ == SL.PropertiesStr.toString }.isDefined) {
368                 // a Properties object

370                   _._1 != SL.PropertiesStr.toString }.map { case(k, v) =>
371                     new Elem(
372                       null, SL.PropertyStr, Null, TopScope,
373                       new Elem(null, SL.Name, Null, TopScope, new Text(k)),
374                       new Elem(
375                         null, SL.ValueStr, Null, TopScope,
376                         new Text(removeQuotes(v.render()))))
377                 }.toSeq
378               } else if (mS.values.forall { v =>
379                 v.valueType() == ConfigValueType.STRING }) {
380                 if (!mS.isEmpty &&
381                   mS.keySet.forall { isDependencyProperty(_) }) {
382                   // a Dependency object (for configurations)
383                   elemKey = SL.DependencyStr
384                   mS.map { case(k, v) => new Elem(
385                     null, k, Null, TopScope, Text(removeQuotes(v.render())))
386                   }.toSeq
387                 } else if (mS.keySet.find {
388                   _ == SL.Implementation.toString }.isDefined) {
389                   // a resource transformer
390                   elemKey = SL.Transformer
391                   val impl =
392                     mS.find { case(k, v) =>
393                       k == SL.Implementation.toString }.map { _._2 }.get
394                   attrs = new UnprefixedAttribute(
395                     SL.Implementation, removeQuotes(impl.render()), Null)
396                   mS.filter { case(k, _) =>
397                     k != SL.Implementation.toString }.map { case(k, v) =>
398                       new Elem(
399                         null, k, Null, TopScope, Text(removeQuotes(v.render())))
400                   }.toSeq
401                 } else if (mS.keySet.find {
402                   _ == SL.Archive.toString }.isDefined) {
403                   // a Maven archiver
404                   val arch = HoconReader.readArchiver(m.toConfig())
405                   Seq(arch.xml) ++
406                   mS.filter { _._1 != SL.Archive.toString }.map { case(k, v) =>
407                     makeElem(k, v) }
408                   /*
409                 } else if (mS.keySet.find {
410                   _ == SL.FilesetStr.toString }.isDefined) {
411                   // a Maven fileset
412                   val fs = HoconReader.readFileset(m.toConfig())
413                   Seq(fs.xml) ++
414                   mS.filter { _._1 != SL.FilesetStr.toString
415                   }.map { case(k, v) => makeElem(k, v) }
416                    */
417                 } else {
418                   // a normal Map (ie non-string value type)
419                   mS.map { case(k, v) => makeElem(k, v) }.toSeq
420                 }
421               } else if (mS.keySet.find {
422                 _ == SL.Archive.toString }.isDefined) {

488   /**
489     * Converts a Json4s object into a Typesafe config in a very direct way
490     * Handles no special cases as there is an isomorphism between Json and
491     * Hocon.  Hocon is really just a special case of Json afterall.
492     */
493   def jsonToConfig(jobject: JObject): Config = {
494     import scala.collection.JavaConverters._
495 
496     def fromJson(v: JValue): Any =
497       if (v != null) {
498         v match {
499           case JNull => null
500           case s: JString => s.s
501           case b: JBool => b.value
502           case d: JDecimal => d.num
503           case d: JDouble => d.num
504           case i: JInt => i.num
505           case o: JObject => ConfigValueFactory.fromMap(fromJsonObject(o))
506           case a: JArray =>
507             val iter = a.arr.map { fromJson(_) }.asJava
508             ConfigValueFactory.fromIterable(iter)
509           case v: JValue => assert(false); null
510         }
511       } else {
512         null
513       }
514 
515     def fromJsonObject(jobj: JObject): java.util.Map[String, Any] =
516       jobj.obj.map { case((key, v)) =>
517         v match {
518           case arr: JArray => (key, fromJson(arr))
519           case obj: JObject => (key, fromJsonObject(obj))
520           case v: JValue => (key, fromJson(v))
521         }
522       }.toMap.asJava
523 
524     if (jobject != null)
525       ConfigValueFactory.fromMap(fromJsonObject(jobject)).toConfig()
526     else null
527   }

529   /**
530     * Converts a Typesafe Config object to a Json4s Json object usually for
531     * serialization.  Does a direct translation.
532     */
533   def configToJson(conf: Config): JObject = {
534     import scala.collection.JavaConverters._
535 
536     def makeJValuePrimitive(value: ConfigValue): JValue =
537       value.unwrapped() match {
538         case null => JNull
539         case b: java.lang.Boolean => JBool(b.booleanValue())
540         case b: java.lang.Byte => JInt(b.toInt)
541         case s: java.lang.Short => JInt(s.toInt)
542         case i: java.lang.Integer => JInt(i.toInt)
543         case l: java.lang.Long => JDecimal(new java.math.BigDecimal(l.toLong))
544         case f: java.lang.Float => JDouble(f.toDouble)
545         case d: java.lang.Double => JDouble(d.toDouble)
546         case s: String => JString(s)
547         case _ => assert(false); null
548       }
549 
550     def makeJValue(value: ConfigValue): JValue = {
551       value match {
552         case l: ConfigList =>
553           JArray(l.asScala.map { it => makeJValue(it) }.toList)
554         case m: ConfigObject =>
555           JObject(m.asScala.map { it => (it._1, makeJValue(it._2)) }.toList)
556         case _ => makeJValuePrimitive(value)
557       }
558     }
559 
560     if (conf != null) {
561       val root = conf.root()
562       val childKeys: Seq[String] = root.keySet().asScala.toSeq
563       val childElems: Seq[JField] =
564         childKeys.map { key => (key, makeJValue(root.get(key))) }
565       new JObject(childElems.toList)
566     } else {
567       null
568     }
569   }
