86   /**
87     * Converts an XML Element into a Typesafe Config object.  Generally it
88     * converts Elements into Maps keyed by the names of the child tags.  It
89     * also looks for special patterns of XML to do more specific type
90     * conversions. Supported type conversions: <ul>
91     * <li>Boolean the strings true or false</li>
92     * <li>Number both Integers and Doubles</li>
93     * <li>String Text nodes of the XML document</li>
94     * <li>Seq[Any] Elements whose child labels are the singlar form of elem's
95     * label
96     * <li>Properties Elements whose grand child labels are all name or value
97     * <li>Seq[Dependency] Special case where child elements all have the
98     * label &quot;dependency&quot;
99     * <li>Archiver Special case where elem's label is &quot;archive&quot;
100     * <li>Fileset Special case where elem's label is &quot;fileset&quot;
101     * <li>Resource Transformer Special case where the elem's label is
102     * transformers, the child's label is transformer and there is an attribute
103     * named implementation defined
104     * <li>Map[String, Any] general/default case
105     * <ul>
106     * @see Configuring Plugins
107     * [[https://maven.apache.org/guides/mini/guide-configuring-plugins.html]]
108     * @see Archiver [[http://maven.apache.org/shared/maven-archiver/index.html]]
109     * @see Resource Transformers
110     * [[https://maven.apache.org/plugins-archives/maven-shade-plugin-2.0/examples/resource-transformers.html]]
111     */
112   def elemToConfig(elem: Elem): Config = {
113     import scala.collection.JavaConverters._
114 
115     // handles Booleans and Numbers
116     def toAnyRef(s: String): Any = s match {
117       case "true" => true
118       case "false" => false
119       case s =>
120         if (s.startsWith("0")) {
121           unencodeXml(s)
122         } else {
123           try { s.toInt } catch {
124             case nfe: NumberFormatException =>
125               try { s.toDouble } catch {
126                 case nfe2: NumberFormatException => unencodeXml(s)
127               }
128           }
129         }
130     }
131 
132     // converts a node el at path path to a ConfigValue inside of parent
133     def appendNode(path: String, parent: Config, el: Node): ConfigValue =
134       if (el != null) {
135         el match {
136           case e: Elem =>
137             // a simple string value
138             val attrs = e.attributes.asAttrMap
139             if (e.child.forall { _.isInstanceOf[Text] } && attrs.isEmpty) {
140               ConfigValueFactory.fromAnyRef(toAnyRef(e.text.trim))
141 
142               // an Archiver
143             } else if (e.label == SL.Archive.toString && attrs.isEmpty) {
144               val a = new Archiver(e)
145               ConfigFactory.parseString(JsonWriter.writeArchiver(a)).root()
146 
147               // a fileset
148             } else if (e.label == SL.FilesetStr.toString && attrs.isEmpty) {
149               val f = new Fileset(e)
150               ConfigFactory.parseString(JsonWriter.writeFileset(f)).root()
151 
152               // a Dependency object
153             } else if (attrs.isEmpty && e.child.forall {
154               case ele: Elem => ele.label == SL.DependencyStr.toString
155               case t: Text => t.text.trim == ""
156               case c: Comment => true
157               case _ => false
158             }) {
159               val listInJava = e.child.map {
160                 appendNode(path, parent, _) }.filter { _ != null}.asJava
161               ConfigValueFactory.fromIterable(listInJava)
162 
163               // a property map (ie Map[String, String])
164             } else if (attrs.isEmpty && e.child.forall {
165               case el: Elem => el.label == SL.PropertyStr.toString
166               case t: Text => t.text.trim == ""
167               case c: Comment => true
168               case _ => false
169             }) {
170               val pairSeqInScala = (e \ SL.PropertyStr).map { el =>
171                 ((el \ SL.Name).text, (el \ SL.ValueStr).text) }
172               val mapInJava = (pairSeqInScala :+ (
173                 (SL.PropertiesStr.toString, SL.TrueStr.toString))).
174                 toMap.asJava
175               ConfigValueFactory.fromMap(mapInJava)
176 
177               // a list
178             } else if (attrs.isEmpty && e.child.forall {
179               case el: Elem => e.label.startsWith(el.label)
180               case t: Text => t.text.trim == ""
181               case c: Comment => true
182               case _ => false
183             }) {
184               val listInJava = e.child.map {
185                 appendNode(path, parent, _) }.filter { _ != null }.asJava
186               ConfigValueFactory.fromIterable(listInJava)
187 
188             } else {
189               // check if this is a resource transformer
190               val implAttr = (e \ ("@" + SL.Implementation))
191               val mapInScala: Map[String, ConfigValue] =
192                 e.child.map {
193                   case ele: Elem =>
194                     val newPath =
195                       if (path != "") path + "." + ele.label else ele.label
196                     (ele.label, appendNode(newPath, parent, ele))
197                   case t: Text => (SL.TextStr.toString, null)
198                   case _ => (null, null)
199                 }.filter { case(k, v) => v != null }.toMap
200               val map = mapInScala.foldLeft(parent) { case(c, (k, v)) =>
201                 c.withValue(k, v) }.root()
202 
203               // resource transformer
204               if (e.label == SL.Transformer.toString && !implAttr.isEmpty) {
205                 map.withValue(
206                   SL.Implementation,
207                   ConfigValueFactory.fromAnyRef(implAttr.text.trim))
208                 // handle attributes here
209               } else if (!attrs.isEmpty) {
210                 val keys = attrs.keySet.toSeq
211                 attrs.foldLeft(map) { case(c, (n, v)) =>
212                   c.withValue(n, ConfigValueFactory.fromAnyRef(v))
213                 }.withValue(
214                   SL.AttributeKeys,
215                   ConfigValueFactory.fromIterable(keys.asJava))
216               } else {
217                 map
218               }
219             }
220           case t: Text =>
221             if (t.text.trim != "")
222               ConfigValueFactory.fromAnyRef(toAnyRef(t.text.trim))
223             else null
224           case c: Comment =>
225             val epty = ConfigFactory.empty()
226             val list = new java.util.ArrayList[java.lang.String]()
227             list.add(c.text)
228             epty.root().withOrigin(epty.origin().withComments(list))
229             // unhandled
230           case n: Node =>
231             println("n " + n + " class " + n.getClass.getName)
232             assert(false)
233             null
234         }
235       } else {
236         null
237       }
238 
239     // check if null and start recursing down the heirarchy of children
240     if (elem != null) {
241       val empty = ConfigFactory.empty()
242       // build each child's ConfigObject
243       val childConfs =
244         elem.child.map { ch => (ch.label, appendNode(ch.label, empty, ch)) }
245       // fold them together into 1 big Config at the paths
246       childConfs.foldLeft(ConfigFactory.empty()) { case(c, (k, v)) =>
247         if (v != null && !k.startsWith("#")) { c.withValue(k, v) } else c }
248     } else {
249       null
250     }
251   }

259   // translate from json string to a jvm string
260   def unencodeJson(s: String): String = {
261     val sb = new StringBuilder()
262     var i = 0
263     // the scala compiler didn't want to do this with a flatMap and a flag so
264     // I had to do it with a while loop, the compiler kept moving the flag
265     // before the flatMap when it was being used in an if condition after the
266     // flatMap even though it was a var
267     while (i < s.length - 1) {
268       s.substring(i, i + 2) match {
269         case "\\b" => sb.append('\b'); i = i + 2
270         case "\\f" => sb.append('\f'); i = i + 2
271         case "\\n" => sb.append('\n'); i = i + 2
272         case "\\r" => sb.append('\r'); i = i + 2
273         case "\\t" => sb.append('\t'); i = i + 2
274         case "\\\"" => sb.append('"'); i = i + 2
275         case "\\\\" => sb.append('\\'); i = i + 2
276         case ch => sb.append(ch(0)); i = i + 1
277       }
278     }
279     if (i == s.length - 1) sb.append(s.last)
280     sb.toString
281   }

305   /**
306     * Translates a Typesafe Config object to a XML Element.  Handles all the
307     * same special cases as elemToConfig.
308     * Supported type conversions: <ul>
309     * <li>Boolean &lt;key&gt;true/false&lt;/key&gt;</li>
310     * <li>Number &lt;key&gt;number&lt;/key&gt;</li>
311     * <li>String &lt;key&gt;value&lt;/key&gt;</li>
312     * <li>Seq[Any] &lt;key(plural)&gt;&lt;key(singular)&gt;value(0)
313     * &lt;/key(singular)&gt;&lt;key(singular)&gt;value(1)&lt;/key(singular)&gt;
314     * ...&lt;/key(plural)&gt;
315     * <li>Properties Identified by the presence of a child at path
316     * &quot;properties&quot; with the value true, otherwise represented as a
317     * Map[String, String] in the Scala code.  It returns the Maven property
318     * XML format which is: &lt;key&gt;&lt;property&gt;&lt;name&gt;property
319     * key1&lt;/name&gt;&lt;value&gt;property
320     * value1&lt;/value&gt;&lt;/property&gt;
321     * &lt;property&gt;&lt;name&gt;property key2&lt;/name&gt;
322     * &lt;value&gt;property value2&lt;/value&gt;&lt;/property&gt;...&lt;/key&gt;
323     * <li>Seq[Dependency] Special case where child elements all have the
324     * keys of a valid &quot;dependency&quot; object in the POM
325     * <li>Archiver Special case where a child key is &quot;archive&quot;
326     * <li>Fileset Special case where a child key is &quot;fileset&quot;
327     * <li>Resource Transformer Special case where a child key is
328     * implementation
329     * <li>Map[String, Any] general/default case, returns whatever
330     * the value translates to inside a XML element labeled by the key
331     * <ul>
332     */
333   def configToElem(config: Config): Elem = {
334     import scala.collection.JavaConverters._
335 
336     // convert the Config value at key into a XML Element
337     def makeElem(key: String, value: ConfigValue): Elem =
338       // check for a child key called archive
339       if (key == SL.Archive.toString) {
340         HoconReader.readArchiver(value.atKey(SL.Archive)).xml
341         // check for child keys with . so they can be split and encoded into
342         // XML correctly
343       } else if (key.contains(".")) {
344         val tokens: Array[String] = key.split('.').reverse
345         val last = tokens.head
346         tokens.tail.foldLeft(makeElem(last, value)) { case(c, token) =>
347           new Elem(null, token, Null, TopScope, c)
348         }
349       } else if (value != null) {
350         value match {
351           // a list so create a parent element called key with children labeled
352           // key in singular form
353           case l: ConfigList =>
354             val childElems = l.asScala.map { it => makeElem(singular(key), it) }
355             new Elem(null, key, Null, TopScope, childElems: _*)
356             // a map so look for special cases and if they don't match convert
357             // as a generic map built using recursion
358           case m: ConfigObject =>
359             val mS = m.asScala
360             var elemKey = key
361             var attrs: MetaData = Null
362             val childElems =
363               if (mS.keySet.find {
364                 _ == SL.PropertiesStr.toString }.isDefined) {
365                 // a Properties object

367                   _._1 != SL.PropertiesStr.toString }.map { case(k, v) =>
368                     new Elem(
369                       null, SL.PropertyStr, Null, TopScope,
370                       new Elem(null, SL.Name, Null, TopScope, new Text(k)),
371                       new Elem(
372                         null, SL.ValueStr, Null, TopScope,
373                         new Text(removeQuotes(v.render()))))
374                 }.toSeq
375               } else if (mS.values.forall { v =>
376                 v.valueType() == ConfigValueType.STRING }) {
377                 if (!mS.isEmpty &&
378                   mS.keySet.forall { isDependencyProperty(_) }) {
379                   // a Dependency object (for configurations)
380                   elemKey = SL.DependencyStr
381                   mS.map { case(k, v) => new Elem(
382                     null, k, Null, TopScope, Text(removeQuotes(v.render())))
383                   }.toSeq
384                 } else if (mS.keySet.find {
385                   _ == SL.Implementation.toString }.isDefined) {
386                   // a resource transformer
387                   elemKey = SL.Transformer
388                   val impl =
389                     mS.find { case(k, v) =>
390                       k == SL.Implementation.toString }.map { _._2 }.get
391                   attrs = new UnprefixedAttribute(
392                     SL.Implementation, removeQuotes(impl.render()), Null)
393                   mS.filter { case(k, _) =>
394                     k != SL.Implementation.toString }.map { case(k, v) =>
395                       new Elem(
396                         null, k, Null, TopScope, Text(removeQuotes(v.render())))
397                   }.toSeq
398                 } else if (mS.keySet.find {
399                   _ == SL.Archive.toString }.isDefined) {
400                   // a Maven archiver
401                   val arch = HoconReader.readArchiver(m.toConfig())
402                   Seq(arch.xml) ++
403                   mS.filter { _._1 != SL.Archive.toString }.map { case(k, v) =>

405                   /*
406                 } else if (mS.keySet.find {
407                   _ == SL.FilesetStr.toString }.isDefined) {
408                   // a Maven fileset
409                   val fs = HoconReader.readFileset(m.toConfig())
410                   Seq(fs.xml) ++
411                   mS.filter { _._1 != SL.FilesetStr.toString

414                 } else {
415                   // a normal Map (ie non-string value type)
416                   mS.map { case(k, v) => makeElem(k, v) }.toSeq
417                 }
418               } else if (mS.keySet.find {

485   /**
486     * Converts a Json4s object into a Typesafe config in a very direct way
487     * Handles no special cases as there is an isomorphism between Json and
488     * Hocon.  Hocon is really just a special case of Json afterall.
489     */
490   def jsonToConfig(jobject: JObject): Config = {
491     import scala.collection.JavaConverters._
492 
493     def fromJson(v: JValue): Any =
494       if (v != null) {
495         v match {
496           case JNull => null
497           case s: JString => s.s
498           case b: JBool => b.value
499           case d: JDecimal => d.num
500           case d: JDouble => d.num
501           case i: JInt => i.num
502           case o: JObject => ConfigValueFactory.fromMap(fromJsonObject(o))
503           case a: JArray =>
504             val iter = a.arr.map { fromJson(_) }.asJava
505             ConfigValueFactory.fromIterable(iter)
506           case v: JValue => assert(false); null
507         }
508       } else {
509         null
510       }
511 
512     def fromJsonObject(jobj: JObject): java.util.Map[String, Any] =
513       jobj.obj.map { case((key, v)) =>
514         v match {
515           case arr: JArray => (key, fromJson(arr))
516           case obj: JObject => (key, fromJsonObject(obj))
517           case v: JValue => (key, fromJson(v))
518         }
519       }.toMap.asJava
520 
521     if (jobject != null)
522       ConfigValueFactory.fromMap(fromJsonObject(jobject)).toConfig()
523     else null
524   }

526   /**
527     * Converts a Typesafe Config object to a Json4s Json object usually for
528     * serialization.  Does a direct translation.
529     */
530   def configToJson(conf: Config): JObject = {
531     import scala.collection.JavaConverters._
532 
533     def makeJValuePrimitive(value: ConfigValue): JValue =
534       value.unwrapped() match {
535         case null => JNull
536         case b: java.lang.Boolean => JBool(b.booleanValue())
537         case b: java.lang.Byte => JInt(b.toInt)
538         case s: java.lang.Short => JInt(s.toInt)
539         case i: java.lang.Integer => JInt(i.toInt)
540         case l: java.lang.Long => JDecimal(new java.math.BigDecimal(l.toLong))
541         case f: java.lang.Float => JDouble(f.toDouble)
542         case d: java.lang.Double => JDouble(d.toDouble)
543         case s: String => JString(s)
544         case _ => assert(false); null
545       }
546 
547     def makeJValue(value: ConfigValue): JValue = {
548       value match {
549         case l: ConfigList =>
550           JArray(l.asScala.map { it => makeJValue(it) }.toList)
551         case m: ConfigObject =>
552           JObject(m.asScala.map { it => (it._1, makeJValue(it._2)) }.toList)
553         case _ => makeJValuePrimitive(value)
554       }
555     }
556 
557     if (conf != null) {
558       val root = conf.root()
559       val childKeys: Seq[String] = root.keySet().asScala.toSeq
560       val childElems: Seq[JField] =
561         childKeys.map { key => (key, makeJValue(root.get(key))) }
562       new JObject(childElems.toList)
563     } else {
564       null
565     }
566   }
