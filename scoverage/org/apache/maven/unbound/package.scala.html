86   /**
87     * Converts an XML Element into a Typesafe Config object.  Generally it
88     * converts Elements into Maps keyed by the names of the child tags.  It
89     * also looks for special patterns of XML to do more specific type
90     * conversions. Supported type conversions: <ul>
91     * <li>Boolean the strings true or false</li>
92     * <li>Number both Integers and Doubles</li>
93     * <li>String Text nodes of the XML document</li>
94     * <li>Seq[Any] Elements whose child labels are the singlar form of elem's
95     * label
96     * <li>Properties Elements whose grand child labels are all name or value
97     * <li>Seq[Dependency] Special case where child elements all have the
98     * label &quot;dependency&quot;
99     * <li>Archiver Special case where elem's label is &quot;archive&quot;
100     * <li>Resource Transformer Special case where the elem's label is
101     * transformers, the child's label is transformer and there is an attribute
102     * named implementation defined
103     * <li>Map[String, Any] general/default case
104     * <ul>
105     * @see Configuring Plugins
106     * [[https://maven.apache.org/guides/mini/guide-configuring-plugins.html]]
107     * @see Archiver [[http://maven.apache.org/shared/maven-archiver/index.html]]
108     * @see Resource Transformers
109     * [[https://maven.apache.org/plugins-archives/maven-shade-plugin-2.0/examples/resource-transformers.html]]
110     */
111   def elemToConfig(elem: Elem): Config = {
112     import scala.collection.JavaConverters._
113 
114     // handles Booleans and Numbers
115     def toAnyRef(s: String): Any = s match {
116       case "true" => true
117       case "false" => false
118       case s =>
119         try { s.toInt } catch {
120           case nfe: NumberFormatException =>
121             try { s.toDouble } catch {
122               case nfe2: NumberFormatException => s
123             }
124         }
125     }
126 
127     // converts a node el at path path to a ConfigValue inside of parent
128     def appendNode(path: String, parent: Config, el: Node): ConfigValue =
129       if (el != null) {
130         el match {
131           case e: Elem =>
132             // a simple string value
133             if (e.child.forall { _.isInstanceOf[Text] }) {
134               ConfigValueFactory.fromAnyRef(toAnyRef(e.text.trim))
135 
136               // an Archiver
137             } else if (e.label == SL.Archive.toString) {
138               val a = new Archiver(e)
139               ConfigFactory.parseString(JsonWriter.writeArchiver(a)).root()
140 
141               // a Dependency object
142             } else if (e.child.forall {
143               case ele: Elem => ele.label == SL.DependencyStr.toString
144               case t: Text => t.text.trim == ""
145               case _ => false
146             }) {
147               val listInJava = e.child.map {
148                 appendNode(path, parent, _) }.filter { _ != null}.asJava
149               ConfigValueFactory.fromIterable(listInJava)
150 
151               // a property map (ie Map[String, String])
152             } else if (e.child.forall {
153               case el: Elem => el.label == SL.PropertyStr.toString
154               case t: Text => t.text.trim == ""
155               case _ => false
156             }) {
157               val pairSeqInScala = (e \ SL.PropertyStr).map { el =>
158                 ((el \ SL.Name).text, (el \ SL.ValueStr).text) }
159               val mapInJava = (pairSeqInScala :+ (
160                 (SL.PropertiesStr.toString, SL.TrueStr.toString))).
161                 toMap.asJava
162               ConfigValueFactory.fromMap(mapInJava)
163 
164               // a list
165             } else if (e.child.forall {
166               case el: Elem => e.label.startsWith(el.label)
167               case t: Text => t.text.trim == ""
168               case _ => false
169             }) {
170               val listInJava = e.child.map {
171                 appendNode(path, parent, _) }.filter { _ != null }.asJava
172               ConfigValueFactory.fromIterable(listInJava)
173 
174             } else {
175               // check if this is a resource transformer
176               val implAttr = (e \ ("@" + SL.Implementation))
177               val mapInScala: Map[String, ConfigValue] =
178                 e.child.map {
179                   case ele: Elem =>
180                     val newPath =
181                       if (path != "") path + "." + ele.label else ele.label
182                     (ele.label, appendNode(newPath, parent, ele))
183                   case t: Text => (SL.TextStr.toString, null)
184                 }.filter { case(k, v) => v != null }.toMap
185               val map = mapInScala.foldLeft(parent) { case(c, (k, v)) =>
186                 c.withValue(k, v) }.root()
187 
188               // resource transformer
189               if (e.label == SL.Transformer.toString && !implAttr.isEmpty) {
190                 map.withValue(
191                   SL.Implementation,
192                   ConfigValueFactory.fromAnyRef(implAttr.text.trim))
193               } else {
194                 map
195               }
196             }
197           case t: Text =>
198             if (t.text.trim != "")
199               ConfigValueFactory.fromAnyRef(toAnyRef(t.text.trim))
200             else null
201           case c: Comment =>
202             val epty = ConfigFactory.empty()
203             val list = new java.util.ArrayList[java.lang.String]()
204             list.add(c.text)
205             epty.root().withOrigin(epty.origin().withComments(list))
206             // unhandled
207           case n: Node =>
208             println("n " + n)
209             assert(false)
210             null
211         }
212       } else {
213         null
214       }
215 
216     // check if null and start recursing down the heirarchy of children
217     if (elem != null) {
218       val empty = ConfigFactory.empty()
219       // build each child's ConfigObject
220       val childConfs =
221         elem.child.map { ch => (ch.label, appendNode(ch.label, empty, ch)) }
222       // fold them together into 1 big Config at the paths
223       childConfs.foldLeft(ConfigFactory.empty()) { case(c, (k, v)) =>
224         if (v != null) c.withValue(k, v) else c }
225     } else {
226       null
227     }
228   }

249   /**
250     * Translates a Typesafe Config object to a XML Element.  Handles all the
251     * same special cases as elemToConfig.
252     * Supported type conversions: <ul>
253     * <li>Boolean &lt;key&gt;true/false&lt;/key&gt;</li>
254     * <li>Number &lt;key&gt;number&lt;/key&gt;</li>
255     * <li>String &lt;key&gt;value&lt;/key&gt;</li>
256     * <li>Seq[Any] &lt;key(plural)&gt;&lt;key(singular)&gt;value(0)
257     * &lt;/key(singular)&gt;&lt;key(singular)&gt;value(1)&lt;/key(singular)&gt;
258     * ...&lt;/key(plural)&gt;
259     * <li>Properties Identified by the presence of a child at path
260     * &quot;properties&quot; with the value true, otherwise represented as a
261     * Map[String, String] in the Scala code.  It returns the Maven property
262     * XML format which is: &lt;key&gt;&lt;property&gt;&lt;name&gt;property
263     * key1&lt;/name&gt;&lt;value&gt;property
264     * value1&lt;/value&gt;&lt;/property&gt;
265     * &lt;property&gt;&lt;name&gt;property key2&lt;/name&gt;
266     * &lt;value&gt;property value2&lt;/value&gt;&lt;/property&gt;...&lt;/key&gt;
267     * <li>Seq[Dependency] Special case where child elements all have the
268     * keys of a valid &quot;dependency&quot; object in the POM
269     * <li>Archiver Special case where a child key is &quot;archive&quot;
270     * <li>Resource Transformer Special case where a child key is
271     * implementation
272     * <li>Map[String, Any] general/default case, returns whatever
273     * the value translates to inside a XML element labeled by the key
274     * <ul>
275     */
276   def configToElem(config: Config): Elem = {
277     import scala.collection.JavaConverters._
278 
279     // convert the Config value at key into a XML Element
280     def makeElem(key: String, value: ConfigValue): Elem =
281       // check for a child key called archive
282       if (key == SL.Archive.toString) {
283         HoconReader.readArchiver(value.atKey(SL.Archive)).xml
284         // check for child keys with . so they can be split and encoded into
285         // XML correctly
286       } else if (key.contains(".")) {
287         val tokens: Array[String] = key.split('.').reverse
288         val last = tokens.head
289         tokens.tail.foldLeft(makeElem(last, value)) { case(c, token) =>
290           new Elem(null, token, Null, TopScope, c)
291         }
292       } else if (value != null) {
293         value match {
294           // a list so create a parent element called key with children labeled
295           // key in singular form
296           case l: ConfigList =>
297             val childElems = l.asScala.map { it => makeElem(singular(key), it) }
298             new Elem(null, key, Null, TopScope, childElems: _*)
299             // a map so look for special cases and if they don't match convert
300             // as a generic map built using recursion
301           case m: ConfigObject =>
302             val mS = m.asScala
303             var elemKey = key
304             var attrs: MetaData = Null
305             val childElems =
306               if (mS.keySet.find {
307                 _ == SL.PropertiesStr.toString }.isDefined) {
308                 // a Properties object
309                 mS.filter{
310                   _._1 != SL.PropertiesStr.toString }.map { case(k, v) =>
311                     new Elem(
312                       null, SL.PropertyStr, Null, TopScope,
313                       new Elem(null, SL.Name, Null, TopScope, new Text(k)),
314                       new Elem(
315                         null, SL.ValueStr, Null, TopScope,
316                         new Text(removeQuotes(v.render()))))
317                 }.toSeq
318               } else if (mS.values.forall { v =>
319                 v.valueType() == ConfigValueType.STRING }) {
320                 if (mS.keySet.forall { isDependencyProperty(_) }) {
321                   // a Dependency object (for configurations)
322                   elemKey = SL.DependencyStr
323                   mS.map { case(k, v) => new Elem(
324                     null, k, Null, TopScope, Text(removeQuotes(v.render())))
325                   }.toSeq
326                 } else if (mS.keySet.find {
327                   _ == SL.Implementation.toString }.isDefined) {
328                   // a resource transformer
329                   elemKey = SL.Transformer
330                   val impl =
331                     mS.find { case(k, v) =>
332                       k == SL.Implementation.toString }.map { _._2 }.get
333                   attrs = new UnprefixedAttribute(
334                     SL.Implementation, removeQuotes(impl.render()), Null)
335                   mS.filter { case(k, _) =>
336                     k != SL.Implementation.toString }.map { case(k, v) =>
337                       new Elem(
338                         null, k, Null, TopScope, Text(removeQuotes(v.render())))
339                   }.toSeq
340                 } else if (mS.keySet.find {
341                   _ == SL.Archive.toString }.isDefined) {
342                   // a Maven archiver
343                   val arch = HoconReader.readArchiver(m.toConfig())
344                   Seq(arch.xml) ++
345                   mS.filter { _._1 != SL.Archive.toString }.map { case(k, v) =>
346                     makeElem(k, v) }
347                 } else {
348                   // a normal Map (ie non-string value type)
349                   mS.map { case(k, v) => makeElem(k, v) }.toSeq
350                 }
351               } else {
352                 // a normal Map (ie non-string value type)
353                 mS.map { case(k, v) => makeElem(k, v) }.toSeq
354               }
355             new Elem(null, elemKey, attrs, TopScope, childElems: _*)
356             // a simple value so convert to a string and wrap into a Element
357             // labeled key
358           case v: ConfigValue =>
359             val value = removeQuotes(v.render())
360             new Elem(null, key, Null, TopScope, new Text(value))
361         }
362       } else {
363         null
364       }
365 
366     // start the recurse by visiting the top level children of the config
367     if (config != null) {
368       val children: Seq[(String, ConfigValue)] = config.root().asScala.toSeq
369       val childElems: Seq[Elem] =
370         children.map { entry => makeElem(entry._1, entry._2) }
371       new Elem(null, SL.Configuration, Null, TopScope, childElems: _*)
372     } else {
373       null
374     }
375   }

377   /**
378     * Converts a Json4s object into a Typesafe config in a very direct way
379     * Handles no special cases as there is an isomorphism between Json and
380     * Hocon.  Hocon is really just a special case of Json afterall.
381     */
382   def jsonToConfig(jobject: JObject): Config = {
383     import scala.collection.JavaConverters._
384 
385     def fromJson(v: JValue): Any =
386       if (v != null) {
387         v match {
388           case JNull => null
389           case s: JString => s.s
390           case b: JBool => b.value
391           case d: JDecimal => d.num
392           case d: JDouble => d.num
393           case i: JInt => i.num
394           case o: JObject => ConfigValueFactory.fromMap(fromJsonObject(o))
395           case a: JArray =>
396             val iter = a.arr.map { fromJson(_) }.asJava
397             ConfigValueFactory.fromIterable(iter)
398           case v: JValue => assert(false); null
399         }
400       } else {
401         null
402       }
403 
404     def fromJsonObject(jobj: JObject): java.util.Map[String, Any] =
405       jobj.obj.map { case((key, v)) =>
406         v match {
407           case arr: JArray => (key, fromJson(arr))
408           case obj: JObject => (key, fromJsonObject(obj))
409           case v: JValue => (key, fromJson(v))
410         }
411       }.toMap.asJava
412 
413     if (jobject != null)
414       ConfigValueFactory.fromMap(fromJsonObject(jobject)).toConfig()
415     else null
416   }

418   /**
419     * Converts a Typesafe Config object to a Json4s Json object usually for
420     * serialization.  Does a direct translation.
421     */
422   def configToJson(conf: Config): JObject = {
423     import scala.collection.JavaConverters._
424 
425     def makeJValuePrimitive(value: ConfigValue): JValue =
426       value.unwrapped() match {
427         case null => JNull
428         case b: java.lang.Boolean => JBool(b.booleanValue())
429         case b: java.lang.Byte => JInt(b.toInt)
430         case s: java.lang.Short => JInt(s.toInt)
431         case i: java.lang.Integer => JInt(i.toInt)
432         case l: java.lang.Long => JDecimal(new java.math.BigDecimal(l.toLong))
433         case f: java.lang.Float => JDouble(f.toDouble)
434         case d: java.lang.Double => JDouble(d.toDouble)
435         case s: String => JString(s)
436         case _ => assert(false); null
437       }
438 
439     def makeJValue(value: ConfigValue): JValue = {
440       value match {
441         case l: ConfigList =>
442           JArray(l.asScala.map { it => makeJValue(it) }.toList)
443         case m: ConfigObject =>
444           JObject(m.asScala.map { it => (it._1, makeJValue(it._2)) }.toList)
445         case _ => makeJValuePrimitive(value)
446       }
447     }
448 
449     if (conf != null) {
450       val root = conf.root()
451       val childKeys: Seq[String] = root.keySet().asScala.toSeq
452       val childElems: Seq[JField] =
453         childKeys.map { key => (key, makeJValue(root.get(key))) }
454       new JObject(childElems.toList)
455     } else {
456       null
457     }
458   }
