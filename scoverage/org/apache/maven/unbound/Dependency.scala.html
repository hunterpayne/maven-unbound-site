106   def makeModelObject(): org.apache.maven.model.Dependency = {
107     val dependency = new org.apache.maven.model.Dependency()
108     if (groupId != null) dependency.setGroupId(groupId)
109     if (artifactId != null) dependency.setArtifactId(artifactId)

116     dependency.setOptional(optional)
117     dependency
118   }
