32 /**
33   * Conversion code that translates Json4s objects to and from Project case
34   * class instances.  Unfortunately because Json4s serialization doesn't handle
35   * case classes with default values well, often there are case objects
36   * littered throughout the code to provide explicit serialization to/from
37   * Json4s objects.  Those case objects extend this trait to get all the
38   * utility code needed to read/write Json4s objects
39   */
40 trait CommonJsonReader extends Labels {
41 
42   implicit val formats: Formats
43   implicit val boolReader = DefaultReaders.BooleanReader
44   implicit val strReader = DefaultReaders.StringReader
45 
46   // utility methods for reading various types of data from Json4s objects
47   // and into Option[T]s
48   protected def readBool(fields: List[JField], key: String): Option[Boolean] =
49     fields.filter { _._1 == key }.headOption.map { case(_, v) => v.as[Boolean] }
50 
51   // for strings
52   protected def readStr(fields: List[JField], key: String): Option[String] =
53     fields.filter { _._1 == key }.headOption.map { case(_, v) => v match {
54       case JString(s) => s.replaceAllLiterally("\n", scala.compat.Platform.EOL)
55       case v =>
56         v.as[String].replaceAllLiterally("\n", scala.compat.Platform.EOL)
57     }}
58 
59   // for objects
60   protected[unbound] def readObject[T](
61     obj: JObject, key: String, defVa: T = null)(implicit m: Manifest[T]): T =
62     (obj \ key) match {
63       case obj2: JObject => Extraction.extract[T](obj2)
64       case _ => defVa
65     }
66 
67   // for reading Properties objects
68   protected def readProperties(
69     obj: JObject, key: String = SL.PropertiesStr): Map[String, String] =
70     (obj \ key) match {
71       case JObject(fields) =>
72         fields.map { case((key, v)) => (key, v.as[String]) }.toMap
73       case _ => Map[String, String]()
74     }
75 
76   // for reading lists of objects
77   protected def readObjectSequence[T](
78     fields: List[JField], key: String,
79     defVal: Seq[T] = Seq[T]())(implicit m: Manifest[T]): Seq[T] =
80     fields.filter { _._1 == key }.headOption.map { e =>
81       e._2.children.map { Extraction.extract[T](_) }}.getOrElse(defVal)
82 
83   // for reading lists of strings
84   protected def readStringSequence(
85     fields: List[JField], key: String): Seq[String] =
86     fields.filter { _._1 == key }.headOption.map { goals =>
87       goals._2.children.map { goal => Extraction.extract[String](goal) }
88     }.getOrElse(Seq[String]())
89 
90   // utility methods for converting a specific data type to a Json4s field
91   protected def writeBool(
92     name: String, b: Boolean, defVal: Boolean): Option[JField] =
93     if (b != defVal) Some((name, JBool(b)))
94     else None
95 
96   // for writing a string
97   protected def writeStr(
98     name: String, s: String, defVal: String = null): Option[JField] = {
99     if (s != null && s != defVal) Some((name, JString(s)))
100     else None
101   }
102 
103   // for writing a generic object of type T
104   protected def writeObject[T](
105     name: String, t: T, defVal: T = null): Option[JField] =
106     if (t != null && t != defVal) Some((name, Extraction.decompose(t)))
107     else None
108 
109   // for writing a Properties object
110   protected def writeProperties(
111     name: String, p: Map[String, String]): Option[JField] =
112     if (!p.isEmpty)
113       Some((name, JObject(p.map { case(k, v) => (k, JString(v)) }.toList)))
114     else None
115 
116   // for writing a list of objects of type T
117   protected def writeObjectSequence[T](
118     name: String, arr: Seq[T], defVal: Seq[T] = Seq[T]()): Option[JField] =
119     if (arr != null && arr != defVal)
120       Some((name, JArray(arr.map { Extraction.decompose(_) }.toList)))
121     else None
122 
123   // for writing a list of strings
124   protected def writeStringSequence(
125     name: String, v: Seq[String]): Option[JField] =
126     if (v != null && !v.isEmpty)
127       Some((name, JArray(v.map { JString(_) }.toList)))
128     else None
129 }

178 /**
179   * Static Json reader for Project objects.  Reads from Strings, Readers or
180   * InputStreams and returns Project case classes
181   */
182 object JsonReader extends JsonProjectAPI {
183 
184   /** Reads a string into a Project case class */
185   def readPOM(jsonSrc: String): Project = read[Project](jsonSrc)
186 
187   /** Reads the contents of a Reader into a Project case class */
188   def readPOM(reader: Reader): Project = read[Project](reader)
189 
190   /**
191     * Reads the contents of an InputStream in a specific character encoding
192     * into a Project case class
193     */
194   def readPOM(in: InputStream, enc: String = "UTF-8"): Project =
195     read[Project](new InputStreamReader(in, enc))
196 }
197 
198 /**
199   * Static Json writer for Project objects.  Writes Project case classes into
200   * Strings, Writers or OutputStreams
201   */
202 object JsonWriter extends JsonProjectAPI {
203 
204   /** Writes a condensed Json representing project and returns it as a String */
205   def writeConcisePOM(project: Project): String = write[Project](project)
206 
207   /**
208     * Writes a human readable Json representing project and returns it as a
209     * String
210     */
211   def writePOM(project: Project): String = writePretty(project)
212 
213   /**
214     * Writes a human readable Json representing project into the provided
215     * Writer
216     */
217   def writePOM[W <: Writer](project: Project, writer: W): Unit =
218     writePretty[Project, W](project, writer)
219 
220   /**
221     * Writes a human readable Json representing project into the provided
222     * OutputStream using encoding enc
223     */
224   def writePOM(
225     project: Project, os: OutputStream, enc: String = "UTF-8"): Unit =
226     writePretty[Project, OutputStreamWriter](
227       project, new OutputStreamWriter(os, enc))
228 
229   // used to write an Archiver as Json
230   protected[unbound] def writeArchiver(archiver: Archiver): String =
231     writePretty(archiver)
