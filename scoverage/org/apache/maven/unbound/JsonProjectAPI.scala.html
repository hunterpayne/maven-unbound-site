32 /**
33   * Conversion code that translates Json4s objects to and from Project case
34   * class instances.  Unfortunately because Json4s serialization doesn't handle
35   * case classes with default values well, often there are case objects
36   * littered throughout the code to provide explicit serialization to/from
37   * Json4s objects.  Those case objects extend this trait to get all the
38   * utility code needed to read/write Json4s objects
39   */
40 trait CommonJsonReader extends Labels {
41 
42   implicit val formats: Formats
43   implicit val boolReader = DefaultReaders.BooleanReader
44   implicit val strReader = DefaultReaders.StringReader
45 
46   // utility methods for reading various types of data from Json4s objects
47   // and into Option[T]s
48   protected def readBool(fields: List[JField], key: String): Option[Boolean] =
49     fields.filter { _._1 == key }.headOption.map { case(_, v) => v.as[Boolean] }
50 
51   // for strings
52   protected def readStr(fields: List[JField], key: String): Option[String] =
53     fields.filter { _._1 == key }.headOption.map { case(_, v) => v match {
54       case JString(s) => unencodeJson(s)
55       case v => unencodeJson(v.as[String])
56     }}
57 
58   // for objects
59   protected[unbound] def readObject[T](
60     obj: JObject, key: String, defVa: T = null)(implicit m: Manifest[T]): T =
61     (obj \ key) match {
62       case obj2: JObject => Extraction.extract[T](obj2)
63       case _ => defVa
64     }
65 
66   // for reading Properties objects
67   protected def readProperties(
68     obj: JObject, key: String = SL.PropertiesStr): Map[String, String] =
69     (obj \ key) match {
70       case JObject(fields) =>
71         fields.map { case((key, v)) => (key, v.as[String]) }.toMap
72       case _ => Map[String, String]()
73     }
74 
75   // for reading lists of objects
76   protected def readObjectSequence[T](
77     fields: List[JField], key: String,
78     defVal: Seq[T] = Seq[T]())(implicit m: Manifest[T]): Seq[T] =
79     fields.filter { _._1 == key }.headOption.map { e =>
80       e._2.children.map { Extraction.extract[T](_) }}.getOrElse(defVal)
81 
82   // for reading lists of strings
83   protected def readStringSequence(
84     fields: List[JField], key: String): Seq[String] =
85     fields.filter { _._1 == key }.headOption.map { goals =>
86       goals._2.children.map { goal => goal.as[String] }
87     }.getOrElse(Seq[String]())
88 
89   // utility methods for converting a specific data type to a Json4s field
90   protected def writeBool(
91     name: String, b: Boolean, defVal: Boolean): Option[JField] =
92     if (b != defVal) Some((name, JBool(b)))
93     else None
94 
95   // for writing a string
96   protected def writeStr(
97     name: String, s: String, defVal: String = null): Option[JField] = {
98     if (s != null && s != defVal) Some((name, JString(s)))
99     else None
100   }
101 
102   // for writing a generic object of type T
103   protected def writeObject[T](
104     name: String, t: T, defVal: T = null): Option[JField] =
105     if (t != null && t != defVal) Some((name, Extraction.decompose(t)))
106     else None
107 
108   // for writing a Properties object
109   protected def writeProperties(
110     name: String, p: Map[String, String]): Option[JField] =
111     if (!p.isEmpty)
112       Some((name, JObject(p.map { case(k, v) => (k, JString(v)) }.toList)))
113     else None
114 
115   // for writing a list of objects of type T
116   protected def writeObjectSequence[T](
117     name: String, arr: Seq[T], defVal: Seq[T] = Seq[T]()): Option[JField] =
118     if (arr != null && arr != defVal)
119       Some((name, JArray(arr.map { Extraction.decompose(_) }.toList)))
120     else None
121 
122   // for writing a list of strings
123   protected def writeStringSequence(
124     name: String, v: Seq[String]): Option[JField] =
125     if (v != null && !v.isEmpty)
126       Some((name, JArray(v.map { JString(_) }.toList)))
127     else None
128 }

177 /**
178   * Static Json reader for Project objects.  Reads from Strings, Readers or
179   * InputStreams and returns Project case classes
180   */
181 object JsonReader extends JsonProjectAPI {
182 
183   /** Reads a string into a Project case class */
184   def readPOM(jsonSrc: String): Project = read[Project](jsonSrc)
185 
186   /** Reads the contents of a Reader into a Project case class */
187   def readPOM(reader: Reader): Project = read[Project](reader)
188 
189   /**
190     * Reads the contents of an InputStream in a specific character encoding
191     * into a Project case class
192     */
193   def readPOM(in: InputStream, enc: String = "UTF-8"): Project =
194     read[Project](new InputStreamReader(in, enc))
195 }
196 
197 /**
198   * Static Json writer for Project objects.  Writes Project case classes into
199   * Strings, Writers or OutputStreams
200   */
201 object JsonWriter extends JsonProjectAPI {
202 
203   /** Writes a condensed Json representing project and returns it as a String */
204   def writeConcisePOM(project: Project): String = write[Project](project)
205 
206   /**
207     * Writes a human readable Json representing project and returns it as a
208     * String
209     */
210   def writePOM(project: Project): String = writePretty(project)
211 
212   /**
213     * Writes a human readable Json representing project into the provided
214     * Writer
215     */
216   def writePOM[W <: Writer](project: Project, writer: W): Unit = {
217     writePretty[Project, W](project, writer)
218     writer.flush()
219   }
220 
221   /**
222     * Writes a human readable Json representing project into the provided
223     * OutputStream using encoding enc
224     */
225   def writePOM(
226     project: Project, os: OutputStream, enc: String = "UTF-8"): Unit = {
227     val writer = new OutputStreamWriter(os, enc)
228     writePretty(project, writer)
229     writer.flush()
230   }
231 
232   // used to write an Archiver as Json
233   protected[unbound] def writeArchiver(archiver: Archiver): String =
234     writePretty(archiver)
235 
236   // used to write an Fileset as Json
237   protected[unbound] def writeFileset(fileset: Fileset): String =
238     writePretty(fileset)
