25 /**
26   * A trait case classes can extend to get the ability to write themselves
27   * as XML provided they have a field named xml of type Elem
28   */
29 trait Writeable {
30 
31   val xml: Elem
32 
33   // max width: 80 chars
34   // indent:     2 spaces
35   /**
36     * returns a string containing a pretty printed XML document represented
37     * by the field xml
38     */
39   def toXmlString: String = {
40     val printer = new PrettyPrinter(80, 2)
41     val sb = new StringBuilder()
42     printer.format(xml, sb)
43     sb.toString()
44   }
45 
46   /** writes this object into the provided Writer */
47   def writePOM(writer: Writer): Unit = writer.write(toXmlString)
48 
49   /** writes this object to the specified file using UTF-8 character encoding */
50   def writePOM(filename: String): Unit = {
51     val writer =
52       new OutputStreamWriter(new FileOutputStream(new File(filename)), "UTF-8")
53     try { writePOM(writer) } finally { writer.close() }
54   }
