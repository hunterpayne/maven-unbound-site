272       insertComments(currPath)
273 
274       method.returnType.toString match {
275         case "Boolean" =>
276           writeKey(key)

422       // get default value for argument at index i with name p
423       def valueFor(p: Symbol, i: Int): Any = {
424         val defarg = ts.member(newTermName(s"$name$$default$$${i + 1}"))

426           defarg != NoSymbol,

428         (im.reflectMethod(defarg.asMethod))()
429       }
430 
431       (for (ps <- method.paramss; p <- ps) yield p).zipWithIndex.map { p =>
432         valueFor(p._1, p._2) }
