26 /**
27   * A trait case classes can extend to get the ability to write themselves
28   * as XML provided they have a field named xml of type Elem
29   */
30 trait Writeable {
31 
32   val xml: Elem
33 
34   // max width: 80 chars
35   // indent:     2 spaces
36   /**
37     * returns a string containing a pretty printed XML document represented
38     * by the field xml
39     */
40   def toXmlString: String = {
41     val printer = new PrettyPrinter(80, 2)
42     val sb = new StringBuilder()
43     printer.format(xml, sb)
44     sb.toString()
45   }
46 
47   /** writes this object into the provided Writer */
48   def writePOM(writer: Writer): Unit = writer.write(toXmlString)
49 
50   /** writes this object to the specified file using UTF-8 character encoding */
51   def writePOM(filename: String): Unit = {
52     val writer =
53       new OutputStreamWriter(new FileOutputStream(new File(filename)), "UTF-8")
54     try {
55       writePOM(writer)
56       writer.flush()
57     } finally { writer.close() }
58   }
59 
60   /** writes this object to the specified stream using the specified encoding */
61   def writePOM(os: OutputStream, enc: String): Unit = {
62     val writer = new OutputStreamWriter(os, enc)
63     try { writePOM(writer) } finally { writer.close() }
64   }
