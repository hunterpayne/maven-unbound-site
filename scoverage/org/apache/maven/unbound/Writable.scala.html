270     def outputValue(
271       method: MethodSymbol, key: String, value: Any): Unit = {
272 
273       // inserts comments from parsed from XML here
274       val currPath = path ::: List(ElementLabel(key))
275       insertComments(currPath)
276 
277       method.returnType.toString match {
278         case "Boolean" =>
279           writeKey(key)
280           sb.append(" = ")
281           if (value.asInstanceOf[Boolean]) sb.append("true")
282           else sb.append("false")
283           sb.append(nl)
284 
285         case "Int" | "Long" | "Float" | "Double" =>
286           writeKey(key)
287           sb.append(" = ")
288           sb.append(value.toString)
289           sb.append(nl)
290 
291         case "String" =>
292           writeKey(key)
293           sb.append(" = ")
294           sb.append("\"")
295           sb.append(encodeJson(value.toString))
296           sb.append("\"")
297           sb.append(nl)
298 
299         case "Seq[String]" =>
300           val lst = value.asInstanceOf[Seq[String]]
301           // get comments that are interspearsed in the list of strings in XML
302           comments.comments.filter { comment =>
303             comment.path.elems.toList.startsWith(currPath) &&
304             currPath.size < comment.path.elems.size }.foreach {
305             case comment: Comments =>
306               comment.s.foreach { str =>
307                 sb.append(spacer * (depth + 1))
308                 sb.append("# ")
309                 sb.append(str)
310                 sb.append(nl)
311               }
312           }
313           if (!lst.isEmpty) {
314             writeKey(key)
315             sb.append(" = [ ")
316 
317             val last = lst.last
318             value.asInstanceOf[Seq[String]].foreach { c =>
319               sb.append("\"")
320               sb.append(encodeJson(c))
321               sb.append("\"")
322               if (c != last) sb.append(", ")
323             }
324             sb.append(" ]")
325             sb.append(nl)
326           }
327 
328         case lstClz if (lstClz.startsWith("Seq[")) =>
329           val lst = value.asInstanceOf[Seq[_]]
330           if (!lst.isEmpty) {
331             val innerClassname = lstClz.substring(4, lstClz.length - 1)

333             sb.append(" = [")
334             sb.append(nl)
335 
336             val last = lst.last

338               val idxPath = currPath ::: List(ListIndex(idx))
339               insertComments(idxPath, depth + 2)

344             sb.append(nl)
345           }
346 
347         case "Properties" =>

352             sb.append(nl)

356 
357               val v = props.getProperty(k)

360               sb.append(" = \"")
361               sb.append(encodeJson(v))
362               sb.append("\"")
363               sb.append(nl)
364               idx = idx + 1
365             }
366             sb.append(spacer * (depth + 1))
367             sb.append("}")
368             sb.append(nl)
369           }
370 
371         case "Map[String,String]" =>
372           val strMap = value.asInstanceOf[Map[String, String]]

376             sb.append(nl)

378             strMap.foreach { case(k, v) =>
379               insertComments(currPath ::: List(ElementLabel(k)), depth + 2)
380               insertComments(currPath ::: List(ListIndex(idx)), depth + 2)

382               sb.append("\"")

384               sb.append("\" = \"")
385               sb.append(encodeJson(v))
386               sb.append("\"")

403               val keyPath = currPath ::: List(ElementLabel(k))
404               insertComments(keyPath, depth + 2)

406               sb.append(" = ")

409             sb.append(spacer * (depth + 1))
410             sb.append("}")
411             sb.append(nl)
412           }
413 
414         case "com.typesafe.config.Config" =>
415           val conf = value.asInstanceOf[Config]
416           if (!conf.isEmpty) {
417             writeKey(key)
418             sb.append(" ")
419             val options =
420               ConfigRenderOptions.defaults().setOriginComments(false)
421             // val rendered = conf.root().render(options)
422             // need to take off the first currPath.size PathElements off the
423             // paths b/c otherwise the comment insertion doesn't work

439

441           writeKey(key)

447     def defaut(im: InstanceMirror, name: String): Seq[Any] = {
448       val at = newTermName(name)
449       val ts = im.symbol.typeSignature
450       val method = (ts.member(at)).asMethod
451 
452       // get default value for argument at index i with name p
453       def valueFor(p: Symbol, i: Int): Any = {
454         val defarg = ts.member(newTermName(s"$name$$default$$${i + 1}"))
455         require(
456           defarg != NoSymbol,
457           p.toString + " of " + instance + " has no default value")
458         (im.reflectMethod(defarg.asMethod))()
459       }
460 
461       (for (ps <- method.paramss; p <- ps) yield p).zipWithIndex.map { p =>
462         valueFor(p._1, p._2) }
463     }

483     sb.append(nl)

486       val value = instMirror.reflectMethod(method.asMethod)()
487       require(defValues.hasNext)
488       val defValue = defValues.next
489       if (value != null && value != defValue)
490         outputValue(method, method.name.toString, value)
491     }
492 
493     if (depth > 0) sb.append(spacer * depth)
494     sb.append("}")
495     sb.append(nl)
