67 case class DeploymentRepository(
68   uniqueVersion: Boolean = true,
69   releases: RepositoryPolicy = null, snapshots: RepositoryPolicy = null,
70   id: String = null, name: String = null, url: String = null,
71   layout: String = SL.DefaultStr) {
72 
73   def this(elem: Elem) = this(
74     emptyToDefault(
75       (elem \ SL.UniqueVersion).text.toLowerCase(Locale.ROOT), SL.TrueStr) ==
76       SL.TrueStr.toString(),
77     (elem \ SL.Releases).map { case e: Elem =>
78       new RepositoryPolicy(e) }.headOption.getOrElse(null),
79     (elem \ SL.Snapshots).map { case e: Elem =>
80       new RepositoryPolicy(e) }.headOption.getOrElse(null),
81     emptyToNull((elem \ SL.Id).text),
82     emptyToNull((elem \ SL.Name).text),
83     emptyToNull((elem \ SL.UrlStr).text),
84     emptyToDefault((elem \ SL.Layout).text, SL.DefaultStr))
85 
86   lazy val repositoryXml =
87     <repository>
88       { if (!uniqueVersion) <uniqueVersion>false</uniqueVersion> }
89       { if (releases != null) releases.releasesXml }
90       { if (snapshots != null) snapshots.snapshotsXml }
91       { if (id != null) <id>{id}</id> }
92       { if (name != null) <name>{name}</name> }
93       { if (url != null) <url>{url}</url> }
94       { if (layout != null && layout != SL.DefaultStr.toString)
95         <layout>{layout}</layout> }
96     </repository>
97 
98   lazy val snapshotXml =
99     <snapshotRepository>
100       { if (!uniqueVersion) <uniqueVersion>false</uniqueVersion> }
101       { if (releases != null) releases.releasesXml }
102       { if (snapshots != null) snapshots.snapshotsXml }
103       { if (id != null) <id>{id}</id> }
104       { if (name != null) <name>{name}</name> }
105       { if (url != null) <url>{url}</url> }
106       { if (layout != null && layout != SL.DefaultStr.toString)
107         <layout>{layout}</layout> }
108     </snapshotRepository>
109 
110   def makeModelObject(): org.apache.maven.model.DeploymentRepository = {
111     val repo = new org.apache.maven.model.DeploymentRepository()
112     repo.setUniqueVersion(uniqueVersion)
113     if (releases != null) repo.setReleases(releases.makeModelObject())
114     if (snapshots != null) repo.setSnapshots(snapshots.makeModelObject())
115     // else if (releases != null) repo.setSnapshots(releases.makeModelObject())
116     repo.setId(id)
117     repo.setName(name)
118     repo.setUrl(url)
119     repo.setLayout(layout)
120     repo
121   }
122 }
