69 /**
70   * This class handles the special case of the Maven Archiver
71   * which is a piece of shared infrastructure provided by Maven to Plugins
72   * to allow compressing of build products into a single file.
73   * @see See [[http://maven.apache.org/shared/maven-archiver/index.html]] for
74   * more details
75   * @author Hunter Payne
76   */
77 case class Archiver(
78   addMavenDescriptor: Boolean = true,
79   compress: Boolean = true,
80   forced: Boolean = true,
81   index: Boolean = false,
82   manifest: ManifestObj = null,
83   manifestEntries: Map[String, String] = Map[String, String](),
84   manifestFile: String = null,
85   manifestSections: Seq[ManifestSection] = Seq[ManifestSection](),
86   pomPropertiesFile: String = null
87 ) {
88 
89   def this(elem: Elem) = this(
90     emptyToDefaultBool((elem \ SL.AddMavenDescriptor).text, true),
91     emptyToDefaultBool((elem \ SL.Compress).text, true),
92     emptyToDefaultBool((elem \ SL.Forced).text, true),
93     emptyToDefaultBool((elem \ SL.Index).text, false),
94     (elem \ SL.ManifestStr).map { case e: Elem =>
95       new ManifestObj(e) }.headOption.getOrElse(null),
96     (elem \ SL.ManifestEntries).headOption.map(
97       _.child.filter(_.isInstanceOf[Elem]).map { e =>
98         (e.label, e.text.trim) }.toMap).getOrElse(Map[String, String]()),
99     emptyToNull((elem \ SL.ManifestFile).text.trim),
100     (elem \ SL.ManifestSections \ SL.ManifestSectionStr).map { case e: Elem =>
101       new ManifestSection(e) },
102     emptyToNull((elem \ SL.PomPropertiesFile).text)
103   )
104 
105   lazy val xml =
106     <archive>
107       { if (!addMavenDescriptor) <addMavenDescriptor>false</addMavenDescriptor>}
108       { if (!compress) <compress>false</compress> }
109       { if (!forced) <forced>false</forced> }
110       { if (index) <index>true</index> }
111       { if (manifest != null) manifest.xml }
112       { if (!manifestEntries.isEmpty) <manifestEntries>
113         { manifestEntries.map { case(k, v) => PropertyValue(k, v).xml } }
114         </manifestEntries> }
115       { if (manifestFile != null) <manifestFile>{manifestFile}</manifestFile> }
116       { if (!manifestSections.isEmpty) <manifestSections>
117         { manifestSections.map { _.xml } } </manifestSections> }
118       { if (pomPropertiesFile != null)
119         <pomPropertiesFile>{pomPropertiesFile}</pomPropertiesFile> }
120     </archive>
121 }
