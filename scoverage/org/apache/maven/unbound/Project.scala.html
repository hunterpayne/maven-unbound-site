273     licenses.foreach { lic => model.addLicense(lic.makeModelObject()) }
274     developers.foreach { dev => model.addDeveloper(dev.makeModelObject()) }
275     contributors.foreach { c => model.addContributor(c.makeModelObject()) }
276     mailingLists.foreach { ml => model.addMailingList(ml.makeModelObject()) }
277     modules.foreach { module => model.addModule(module) }
278     if (issueManagement != null)
279       model.setIssueManagement(issueManagement.makeModelObject())
280     if (ciManagement != null)
281       model.setCiManagement(ciManagement.makeModelObject())
282     if (distributionManagement != null)
283       model.setDistributionManagement(distributionManagement.makeModelObject())
284     properties.foreach { case(k, v) => model.addProperty(k, v) }
285     val mgmt = new org.apache.maven.model.DependencyManagement()
286     dependencyManagement.foreach { d =>
287       mgmt.addDependency(d.makeModelObject()) }
288     model.setDependencyManagement(mgmt)
289     dependencies.foreach { d => model.addDependency(d.makeModelObject()) }
290     repositories.foreach { r => model.addRepository(r.makeModelObject()) }
291     pluginRepositories.foreach { repo =>
292       model.addPluginRepository(repo.makeModelObject()) }
293     if (build != null) model.setBuild(build.makeModelObject())
294     if (reporting != null) model.setReporting(reporting.makeModelObject())
295     profiles.foreach { pro => model.addProfile(pro.makeModelObject()) }
296     model
