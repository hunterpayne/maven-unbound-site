28 /**
29   * The Maven Unbound Command Line Interface(CLI).  It converts pom.json
30   * and pom.conf files to pom.xml files and visa-versa.  It reads the parsed
31   * files to extract the list of sub-modules to recurse down a heirarchy
32   * performing the same conversions on all necessary modules of a multi-module
33   * project.  Normally this class is only used when building a command line
34   * interface but can be used in other cases as well.
35   */
36 object Cli {
37 
38   val xmlFileName = "/pom.xml"
39   val hoconFileName = "/pom.conf"
40   val jsonFileName = "/pom.json"
41 
42   /**
43     * Converts a project at path at from Hocon to XML by writing a new pom.xml
44     * in at
45     * @param at the path where we are writing the new pom.xml file
46     * @param project the parsed form of the Project to write to disk
47     */
48   def createPomHoconFiles(at: String, project: Project): Unit = {
49 
50     val hocon = ConfigFactory.parseString(JsonWriter.writeConcisePOM(project))
51     val writer = new OutputStreamWriter(
52       new FileOutputStream(new File(at + hoconFileName)), "UTF-8")
53     val options = ConfigRenderOptions.defaults().setOriginComments(false)
54     try {
55       writer.write(hocon.root().render(options))
56       writer.flush()
57       println(s"generated ${at}${hoconFileName} from ${at}${xmlFileName}")
58     } finally {
59       writer.close()
60     }
61     project.modules.foreach { module =>
62       recurseXml(at + File.separator + module, (s, p) =>
63         createPomHoconFiles(s, p)) }
64   }
65 
66   /**
67     * Converts a project at path at from Json to XML by writing a new pom.xml
68     * in at
69     * @param at the path where we are writing the new pom.xml file
70     * @param project the parsed form of the Project to write to disk
71     */
72   def createPomJsonFiles(at: String, project: Project): Unit = {
73 
74     val writer = new FileWriter(new File(at + jsonFileName))
75     try {
76       JsonWriter.writePOM(project, writer)
77       writer.flush()
78       println(s"generated ${at}${jsonFileName} from ${at}${xmlFileName}")
79     } finally {
80       writer.close()
81 
82       project.modules.foreach { module =>
83         recurseXml(at + File.separator + module, (s, p) =>
84           createPomJsonFiles(s, p)) }
85     }
86   }
87 
88   /**
89     * Recurses down a multi-module Maven project converting XML
90     * POM files into Hocon or Json
91     * @param s the path where to look for POM files
92     * @param createFile a function to converts XML POM files into something
93     * else and writing those new files to disk in the directory specified by s
94     */
95   def recurseXml(s: String, createFile: (String, Project) => Unit): Unit = {
96 
97     val xml = new File(s + xmlFileName)
98 
99     if (xml.exists()) {
100 
101       val xmlIn = new FileInputStream(xml)
102       try {
103         val project = new Project(XML.load(xmlIn))
104         createFile(s, project)
105       } catch {
106         case e: Exception => e.printStackTrace()
107       } finally {
108         xmlIn.close()
109       }
110     } else {
111 
112       println(s"no pom.json or pom.conf in ${s}")
113     }
114   }
115 
116   /**
117     * Converts a project at path at from XML to something else by writing a
118     * new POM file in some other format
119     * @param at the path where we are writing the new POM file
120     * @param project the parsed form of the Project to write to disk
121     * @param form the type of file to write
122     */
123   def createPomXmlFiles(at: String, project: Project, from: String): Unit = {
124 
125     try {
126       project.writePOM(at + xmlFileName)
127       println(s"generated ${at}${xmlFileName} from ${at}${from}")
128     } finally {
129       project.modules.foreach { mod => recurse(at + File.separator + mod) }
130     }
131   }
132 
133   /**
134     * Recurses down a multi-module Maven project converting Hocon or Json
135     * POM files into XML.
136     * @param s the path where to look for POM files
137     */
138   def recurse(s: String): Unit = {
139 
140     val conf = new File(s + hoconFileName)
141     val json = new File(s + jsonFileName)
142 
143     if (json.exists()) {
144 
145       val jsonStr = Source.fromFile(json).getLines.mkString
146       val project = JsonReader.readPOM(jsonStr)
147       createPomXmlFiles(s, project, jsonFileName)
148 
149     } else if (conf.exists()) {
150 
151       val project = HoconReader.readPOM(loadConfig(conf))
152       createPomXmlFiles(s, project, hoconFileName)
153 
154     } else {
155 
156       println(s"no pom.json or pom.conf in ${s}")
157     }
158   }
159 
160   /** entry point of the CLI */
161   def main(args: Array[String]): Unit = {
162 
163     // generate json or hocon files
164     if (args.forall { _.startsWith("--generate") }) {
165 
166       // read xml and generate json
167       if (args.find(_ == "--generate-json").isDefined)
168         recurseXml(".", (s, p) => createPomJsonFiles(s, p))
169 
170       // read xml and generate hocon
171       if (args.find(_ == "--generate-hocon").isDefined)
172         recurseXml(".", (s, p) => createPomHoconFiles(s, p))
173 
174       val unkn = args.find { s =>
175         s != "--generate-hocon" && s != "--generate-json" }
176       if (unkn.isDefined) {
177         println(
178           "can't generate with ${unkn.get}, supported types are hocon and json")
179         println("use please use --generate-hocon or --generate-json")
180       }
181     } else {
182 
183       recurse(".")
184     }
185   }
