29 /**
30   * The Maven Unbound Command Line Interface(CLI).  It converts pom.json
31   * and pom.conf files to pom.xml files and visa-versa.  It reads the parsed
32   * files to extract the list of sub-modules to recurse down a heirarchy
33   * performing the same conversions on all necessary modules of a multi-module
34   * project.  Normally this class is only used when building a command line
35   * interface but can be used in other cases as well.
36   */
37 object Cli {
38 
39   val xmlFileName = "/pom.xml"
40   val hoconFileName = "/pom.conf"
41   val jsonFileName = "/pom.json"
42 
43   /**
44     * Converts a project at path at from Hocon to XML by writing a new pom.xml
45     * in at
46     * @param at the path where we are writing the new pom.xml file
47     * @param project the parsed form of the Project to write to disk
48     */
49   def createPomHoconFiles(
50     at: String, comments: DocComments, project: Project): Unit = {
51 
52     val hocon = ConfigFactory.parseString(JsonWriter.writeConcisePOM(project))
53     val hocon2 = comments.insertConf(hocon)
54     val writer = new OutputStreamWriter(
55       new FileOutputStream(new File(at + hoconFileName)), "UTF-8")
56     val options = ConfigRenderOptions.defaults().setOriginComments(false)
57     try {
58       writer.write(hocon2.root().render(options))
59       writer.flush()
60       println(s"generated ${at}${hoconFileName} from ${at}${xmlFileName}")
61     } finally {
62       writer.close()
63     }
64     project.modules.foreach { module =>
65       recurseXml(at + File.separator + module, (s, c, p) =>
66         createPomHoconFiles(s, c, p)) }
67   }
68 
69   /**
70     * Converts a project at path at from Json to XML by writing a new pom.xml
71     * in at
72     * @param at the path where we are writing the new pom.xml file
73     * @param project the parsed form of the Project to write to disk
74     */
75   def createPomJsonFiles(at: String, project: Project): Unit = {
76 
77     val fos = new FileOutputStream(new File(at + jsonFileName))
78     try {
79       JsonWriter.writePOM(project, fos)
80       fos.flush()
81       println(s"generated ${at}${jsonFileName} from ${at}${xmlFileName}")
82     } finally {
83       fos.close()
84 
85       project.modules.foreach { module =>
86         recurseXml(at + File.separator + module, (s, _, p) =>
87           createPomJsonFiles(s, p)) }
88     }
89   }

91   /**
92     * Recurses down a multi-module Maven project converting XML
93     * POM files into Hocon or Json
94     * @param s the path where to look for POM files
95     * @param createFile a function to converts XML POM files into something
96     * else and writing those new files to disk in the directory specified by s
97     */
98   def recurseXml(s: String, createFile: (String, DocComments, Project) => Unit):
99       Unit = {
100 
101     val xml = new File(s + xmlFileName)
102 
103     if (xml.exists()) {
104 
105       val xmlIn = new FileInputStream(xml)
106       try {
107         // val root = scala.xml.XML.load(xmlIn)
108         val parser =
109           ConstructingParser.fromSource(Source.fromInputStream(xmlIn), true)
110         val root = parser.document.docElem.asInstanceOf[Elem]
111         // extract comments from root
112         val comments = CommentExtractor(root)
113         val project = new Project(root)
114         // pass them to createFile which can insert them back into the
115         // Config object before its rendered
116         createFile(s, comments, project)
117       } catch {
118         case e: Exception => e.printStackTrace()
119       } finally {
120         xmlIn.close()
121       }
122     } else {
123 
124       println(s"no pom.json or pom.conf in ${s}")
125     }
126   }
127 
128   /**
129     * Converts a project at path at from XML to something else by writing a
130     * new POM file in some other format
131     * @param at the path where we are writing the new POM file
132     * @param project the parsed form of the Project to write to disk
133     * @param form the type of file to write
134     */
135   def createPomXmlFiles(at: String, project: Project, from: String): Unit = {
136 
137     try {
138       project.writePOM(at + xmlFileName)
139       println(s"generated ${at}${xmlFileName} from ${at}${from}")
140     } finally {
141       project.modules.foreach { mod => recurse(at + File.separator + mod) }
142     }
143   }
144 
145   /**
146     * Recurses down a multi-module Maven project converting Hocon or Json
147     * POM files into XML.
148     * @param s the path where to look for POM files
149     */
150   def recurse(s: String): Unit = {
151 
152     val conf = new File(s + hoconFileName)
153     val json = new File(s + jsonFileName)
154 
155     if (json.exists()) {
156 
157       val jsonStr = Source.fromFile(json).getLines.mkString
158       val project = JsonReader.readPOM(jsonStr)
159       createPomXmlFiles(s, project, jsonFileName)
160 
161     } else if (conf.exists()) {
162 
163       val config = loadConfig(conf)
164       val project = HoconReader.readPOM(config)
165       if (false) { // if to write comments or not
166         val comments = CommentExtractor(config)
167         try {
168           (new Writeable {
169             val xml = comments.insertXml(project.xml)
170           }).writePOM(s + xmlFileName)
171           println(s"generated ${s}${xmlFileName} from ${s}${hoconFileName}")
172         } finally {
173           project.modules.foreach { mod => recurse(s + File.separator + mod) }
174         }
175       } else {
176         createPomXmlFiles(s, project, hoconFileName)
177       }
178     } else {
179 
180       println(s"no pom.json or pom.conf in ${s}")
181     }
182   }
183 
184   /** entry point of the CLI */
185   def main(args: Array[String]): Unit = {
186 
187     // generate json or hocon files
188     if (args.forall { _.startsWith("--generate") }) {
189 
190       // read xml and generate json
191       if (args.find(_ == "--generate-json").isDefined)
192         recurseXml(".", (s, _, p) => createPomJsonFiles(s, p))
193 
194       // read xml and generate hocon
195       if (args.find(_ == "--generate-hocon").isDefined)
196         recurseXml(".", (s, c, p) => createPomHoconFiles(s, c, p))
197 
198       val unkn = args.find { s =>
199         s != "--generate-hocon" && s != "--generate-json" }
200       if (unkn.isDefined) {
201         println(
202           "can't generate with ${unkn.get}, supported types are hocon and json")
203         println("use please use --generate-hocon or --generate-json")
204       }
205     } else {
206 
207       recurse(".")
208     }
209   }
