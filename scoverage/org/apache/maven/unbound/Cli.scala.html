27 /**
28   * The Maven Unbound Command Line Interface(CLI).  It converts pom.json
29   * and pom.conf files to pom.xml files and visa-versa.  It reads the parsed
30   * files to extract the list of sub-modules to recurse down a heirarchy
31   * performing the same conversions on all necessary modules of a multi-module
32   * project.  Normally this class is only used when building a command line
33   * interface but can be used in other cases as well.
34   */
35 object Cli {
36 
37   val xmlFileName = "/pom.xml"
38   val hoconFileName = "/pom.conf"
39   val jsonFileName = "/pom.json"
40 
41   /**
42     * Converts a project at path at from Hocon to XML by writing a new pom.xml
43     * in at
44     * @param at the path where we are writing the new pom.xml file
45     * @param project the parsed form of the Project to write to disk
46     */
47   def createPomHoconFiles(at: String, project: Project): Unit = {
48 
49     val hocon = ConfigFactory.parseString(JsonWriter.writeConcisePOM(project))
50     val writer = new OutputStreamWriter(
51       new FileOutputStream(new File(at + hoconFileName)), "UTF-8")
52     val options = ConfigRenderOptions.defaults().setOriginComments(false)
53     try {
54       writer.write(hocon.root().render(options))
55       writer.flush()
56       println(s"generated ${at}${hoconFileName} from ${at}${xmlFileName}")
57     } finally {
58       writer.close()
59     }
60     project.modules.foreach { module =>
61       recurseXml(at + File.separator + module, (s, p) =>
62         createPomHoconFiles(s, p)) }
63   }
64 
65   /**
66     * Converts a project at path at from Json to XML by writing a new pom.xml
67     * in at
68     * @param at the path where we are writing the new pom.xml file
69     * @param project the parsed form of the Project to write to disk
70     */
71   def createPomJsonFiles(at: String, project: Project): Unit = {
72 
73     val jsonStr = JsonWriter.writePOM(project)
74     val writer = new OutputStreamWriter(
75       new FileOutputStream(new File(at + jsonFileName)), "UTF-8")
76     try {
77       writer.write(jsonStr)
78       writer.flush()
79       println(s"generated ${at}${jsonFileName} from ${at}${xmlFileName}")
80     } finally {
81       writer.close()

83       project.modules.foreach { module =>
84         recurseXml(at + File.separator + module, (s, p) =>
85           createPomJsonFiles(s, p)) }
86     }
87   }
88 
89   /**
90     * Recurses down a multi-module Maven project converting XML
91     * POM files into Hocon or Json
92     * @param s the path where to look for POM files
93     * @param createFile a function to converts XML POM files into something
94     * else and writing those new files to disk in the directory specified by s
95     */
96   def recurseXml(s: String, createFile: (String, Project) => Unit): Unit = {
97 
98     val xml = new File(s + xmlFileName)
99 
100     if (xml.exists()) {
101 
102       val xmlIn = new FileInputStream(xml)
103       try {
104         val project = new Project(XML.load(xmlIn))
105         createFile(s, project)
106       } catch {
107         case e: Exception => e.printStackTrace()
108       } finally {
109         xmlIn.close()
110       }
111     } else {
112 
113       println(s"no pom.json or pom.conf in ${s}")
114     }
115   }
116 
117   /**
118     * Converts a project at path at from XML to something else by writing a
119     * new POM file in some other format
120     * @param at the path where we are writing the new POM file
121     * @param project the parsed form of the Project to write to disk
122     * @param form the type of file to write
123     */
124   def createPomXmlFiles(at: String, project: Project, from: String): Unit = {
125 
126     val xmlStr = project.toXmlString
127     val writer = new OutputStreamWriter(
128       new FileOutputStream(new File(at + xmlFileName)), "UTF-8")
129     try {
130       writer.write(xmlStr)
131       writer.flush()
132       println(s"generated ${at}${xmlFileName} from ${at}${from}")
133     } finally {
134       writer.close()
135 
136       project.modules.foreach { module =>
137         recurse(at + File.separator + module) }
138     }
139   }
140 
141   /**
142     * Recurses down a multi-module Maven project converting Hocon or Json
143     * POM files into XML.
144     * @param s the path where to look for POM files
145     */
146   def recurse(s: String): Unit = {
147 
148     val conf = new File(s + hoconFileName)
149     val json = new File(s + jsonFileName)
150 
151     if (json.exists()) {
152 
153       val jsonStr = Source.fromFile(json).getLines.mkString
154       val project = JsonReader.readPOM(jsonStr)
155       createPomXmlFiles(s, project, jsonFileName)
156 
157     } else if (conf.exists()) {
158 
159       val project = HoconReader.readPOM(loadConfig(conf))
160       createPomXmlFiles(s, project, hoconFileName)
161 
162     } else {
163 
164       println(s"no pom.json or pom.conf in ${s}")
165     }
166   }
167 
168   /** entry point of the CLI */
169   def main(args: Array[String]): Unit = {
170 
171     // generate json or hocon files
172     if (args.forall { _.startsWith("--generate") }) {
173 
174       // read xml and generate json
175       if (args.find(_ == "--generate-json").isDefined)
176         recurseXml(".", (s, p) => createPomJsonFiles(s, p))
177 
178       // read xml and generate hocon
179       if (args.find(_ == "--generate-hocon").isDefined)
180         recurseXml(".", (s, p) => createPomHoconFiles(s, p))
181 
182       val unkn = args.find { s =>
183         s != "--generate-hocon" && s != "--generate-json" }
184       if (unkn.isDefined) {
185         println(
186           "can't generate with ${unkn.get}, supported types are hocon and json")
187         println("use please use --generate-hocon or --generate-json")
188       }
189     } else {
190 
191       recurse(".")
192     }
193   }
