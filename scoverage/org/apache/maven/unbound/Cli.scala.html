29 /**
30   * The Maven Unbound Command Line Interface(CLI).  It converts pom.json
31   * and pom.conf files to pom.xml files and visa-versa.  It reads the parsed
32   * files to extract the list of sub-modules to recurse down a heirarchy
33   * performing the same conversions on all necessary modules of a multi-module
34   * project.  Normally this class is only used when building a command line
35   * interface but can be used in other cases as well.
36   */
37 object Cli {
38 
39   val xmlFileName = "/pom.xml"
40   val hoconFileName = "/pom.conf"
41   val jsonFileName = "/pom.json"
42 
43   /**
44     * Converts a project at path at from Hocon to XML by writing a new pom.xml
45     * in at
46     * @param at the path where we are writing the new pom.xml file
47     * @param project the parsed form of the Project to write to disk
48     */
49   def createPomHoconFiles(
50     at: String, comments: DocComments, project: Project): Unit = {
51 
52     val hocon = ConfigFactory.parseString(JsonWriter.writeConcisePOM(project))
53     // val hocon2 = comments.insertConf(hocon)
54     val writer = new OutputStreamWriter(
55       new FileOutputStream(new File(at + hoconFileName)), "UTF-8")
56     val options = ConfigRenderOptions.defaults().setOriginComments(false)
57     try {
58       // writer.write(hocon2.root().render(options))
59       writer.write(project.generateConf(comments))
60       writer.flush()
61       println(s"generated ${at}${hoconFileName} from ${at}${xmlFileName}")
62     } finally {
63       writer.close()
64     }
65     project.modules.foreach { module =>
66       recurseXml(at + File.separator + module, (s, c, p) =>
67         createPomHoconFiles(s, c, p)) }
68   }

70   /**
71     * Converts a project at path at from Json to XML by writing a new pom.xml
72     * in at
73     * @param at the path where we are writing the new pom.xml file
74     * @param project the parsed form of the Project to write to disk
75     */
76   def createPomJsonFiles(at: String, project: Project): Unit = {
77 
78     val fos = new FileOutputStream(new File(at + jsonFileName))
79     try {
80       JsonWriter.writePOM(project, fos)
81       fos.flush()
82       println(s"generated ${at}${jsonFileName} from ${at}${xmlFileName}")
83     } finally {
84       fos.close()

86       project.modules.foreach { module =>
87         recurseXml(at + File.separator + module, (s, _, p) =>
88           createPomJsonFiles(s, p)) }
89     }
90   }

92   /**
93     * Recurses down a multi-module Maven project converting XML
94     * POM files into Hocon or Json
95     * @param s the path where to look for POM files
96     * @param createFile a function to converts XML POM files into something
97     * else and writing those new files to disk in the directory specified by s
98     */
99   def recurseXml(s: String, createFile: (String, DocComments, Project) => Unit):
100       Unit = {
101 
102     val xml = new File(s + xmlFileName)
103 
104     if (xml.exists()) {
105 
106       val xmlIn = new FileInputStream(xml)
107       try {
108         // val root = scala.xml.XML.load(xmlIn)
109         val parser =
110           ConstructingParser.fromSource(Source.fromInputStream(xmlIn), true)
111         val root = parser.document.docElem.asInstanceOf[Elem]
112         // extract comments from root
113         val comments = CommentExtractor(root)
114         val project = new Project(root)
115         // pass them to createFile which can insert them back into the
116         // Config object before its rendered
117         createFile(s, comments, project)
118       } catch {
119         case e: Exception => e.printStackTrace()
120       } finally {
121         xmlIn.close()
122       }
123     } else {
124 
125       println(s"no pom.json or pom.conf in ${s}")
126     }
127   }
128 
129   /**
130     * Converts a project at path at from XML to something else by writing a
131     * new POM file in some other format
132     * @param at the path where we are writing the new POM file
133     * @param project the parsed form of the Project to write to disk
134     * @param form the type of file to write
135     */
136   def createPomXmlFiles(at: String, project: Project, from: String): Unit = {
137 
138     try {
139       project.writePOM(at + xmlFileName)
140       println(s"generated ${at}${xmlFileName} from ${at}${from}")
141     } finally {
142       project.modules.foreach { mod => recurse(at + File.separator + mod) }
143     }
144   }
145 
146   /**
147     * Recurses down a multi-module Maven project converting Hocon or Json
148     * POM files into XML.
149     * @param s the path where to look for POM files
150     */
151   def recurse(s: String): Unit = {
152 
153     val conf = new File(s + hoconFileName)
154     val json = new File(s + jsonFileName)
155 
156     if (json.exists()) {
157 
158       val jsonStr = Source.fromFile(json).getLines.mkString
159       val project = JsonReader.readPOM(jsonStr)
160       createPomXmlFiles(s, project, jsonFileName)
161 
162     } else if (conf.exists()) {
163 
164       val config = loadConfig(conf)
165       val project = HoconReader.readPOM(config)
166       if (false) { // if to write comments or not
167         val comments = CommentExtractor(config)
168         try {
169           (new Writable {
170             val xml = comments.insertXml(project.xml)
171           }).writePOM(s + xmlFileName)
172           println(s"generated ${s}${xmlFileName} from ${s}${hoconFileName}")
173         } finally {
174           project.modules.foreach { mod => recurse(s + File.separator + mod) }
175         }
176       } else {
177         createPomXmlFiles(s, project, hoconFileName)
178       }
179     } else {
180 
181       println(s"no pom.json or pom.conf in ${s}")
182     }
183   }
184 
185   /** entry point of the CLI */
186   def main(args: Array[String]): Unit = {
187 
188     // generate json or hocon files
189     if (args.forall { _.startsWith("--generate") }) {
190 
191       // read xml and generate json
192       if (args.find(_ == "--generate-json").isDefined)
193         recurseXml(".", (s, _, p) => createPomJsonFiles(s, p))
194 
195       // read xml and generate hocon
196       if (args.find(_ == "--generate-hocon").isDefined)
197         recurseXml(".", (s, c, p) => createPomHoconFiles(s, c, p))
198 
199       val unkn = args.find { s =>
200         s != "--generate-hocon" && s != "--generate-json" }
201       if (unkn.isDefined) {
202         println(
203           "can't generate with ${unkn.get}, supported types are hocon and json")
204         println("use please use --generate-hocon or --generate-json")
205       }
206     } else {
207 
208       recurse(".")
209     }
210   }
