44 case class ListIndex(idx: Int) extends PathElement {
45 
46   def matchNode(parent: Elem, n: Node): Boolean = n match {

52   import scala.collection.JavaConverters._

54   def matchConf(parent: ConfigValue, key: String, n: ConfigValue): Boolean =
55     parent match {
56       case l: ConfigList => if (l.size > idx) l.get(idx) == n else false
57       case o: ConfigObject =>
58         val keys = o.asScala.keys
59         if (keys.size > idx) {

62     }

65 case class CommentPath(elems: PathElement*) {
66 
67   def insertConf(s: Seq[String], conf: Config): Config = {
68     import scala.collection.JavaConverters._
69 
70     def findNext(
71       it: Iterator[PathElement], cv: ConfigValue): Seq[ConfigValue] =
72       if (it.hasNext) {
73         val pe = it.next
74         cv match {
75           case cur: ConfigObject =>
76             cur.asScala.find { case(k, v) => pe.matchConf(cur, k, v) } match {
77               case Some((_, c)) => findNext(it, c) :+ cur
78               case None =>
79                 println("no match " + cur + " looking for " + pe)
80                 Seq()
81             }
82           case cl: ConfigList =>
83             cl.asScala.find { v => pe.matchConf(cl, "", v) } match {
84               case Some(c) => findNext(it, c) :+ cl
85               case None =>
86                 println("no match list " + cl + " looking for " + pe)
87                 Seq()
88             }
89         }
90       } else Seq(cv)
91 
92     val it = elems.toIterator
93     val path = findNext(it, conf.root())
94     if (!path.isEmpty && elems.size <= path.size) {
95       val relems = elems.reverse.toIterator
96       // add comments to path.head
97       val origin = path.head.origin().withComments(s.asJava)
98       path.tail.foldLeft(path.head.withOrigin(origin)) { case(ch, par) =>
99         par match {
100           case cur: ConfigObject => relems.next match {
101             case ElementLabel(label) => cur.withValue(label, ch)
102             case ListIndex(idx) =>
103               val keys = cur.asScala.keys.toSeq
104               if (keys.size > idx) {
105                 val k = keys(idx)
106                 cur.withValue(k, ch)
107               } else {
108                 println("mismatch " + cur + " @ listidx=" + idx)
109                 cur
110               }
111           }
112           case cl: ConfigList => relems.next match {
113             case ListIndex(idx) =>
114               val lst = new java.util.ArrayList[ConfigValue](cl)
115               lst.set(idx, ch)
116               ConfigValueFactory.fromIterable(lst)
117             case ElementLabel(label) =>
118               println("mismatch " + label + " in " + cl)
119               cl
120           }
121         }}.asInstanceOf[ConfigObject].toConfig
122 
123     } else conf
124   }

126   def insertXml(s: Seq[String], root: Elem): Elem = {
127     import scala.collection.JavaConverters._
128 
129     def findNext(it: Iterator[PathElement], cur: Elem): Seq[Elem] =
130       if (it.hasNext) {
131         it.next match {
132           case ElementLabel(label) =>
133             cur.child.find {
134               case ch: Elem => ch.label == label
135               case _ => false
136             } match {
137               case Some(ch: Elem) => findNext(it, ch) :+ cur
138               case _ => Seq()
139             }
140           case ListIndex(idx) =>
141             findNext(
142               it,
143               cur.child.filter { _.isInstanceOf[Elem]
144               }(idx).asInstanceOf[Elem]) :+
145             cur
146         }
147       } else Seq(cur)
148 
149     val it = elems.tail.toIterator // skip project
150     val path = findNext(it, root)
151 
152     if (!path.isEmpty) {
153 
154       val relems = elems.tail.reverse.toIterator
155       val newChildren: Seq[Node] =
156         s.map { XmlComment(_) } ++ path.head.child.toSeq
157       val oldLeaf = path.head
158       val newLeaf = new Elem(
159         oldLeaf.prefix, oldLeaf.label, oldLeaf.attributes, oldLeaf.scope,
160         newChildren: _*)
161 
162       path.tail.foldLeft((newLeaf, oldLeaf)) { case((ch, oldCh), par) =>
163         val newChildren = par.child.updated(par.child.indexOf(oldCh), ch)
164         (new Elem(
165           par.prefix, par.label, par.attributes, par.scope, newChildren: _*),
166          par)
167       }._1
168     } else root
169   }

188 object CommentExtractor {
189 
190   def apply(conf: Config): DocComments = {
191     import scala.collection.JavaConverters._
192 
193     def traverse(curr: ConfigValue, key: Any, path: List[PathElement]):
194         Seq[Comments] = {
195 
196       val newPath = path :+ (key match {
197         case s: String => ElementLabel(s)
198         case i: Int => ListIndex(i)
199       })
200 
201       val comments = curr.origin().comments().asScala.toSeq
202       val thisComments =
203         if (comments.isEmpty) Seq()
204         else Seq(Comments(comments, CommentPath(newPath: _*)))
205       val childComments: Seq[Comments] = curr match {
206         case cur: ConfigObject =>
207           cur.asScala.flatMap { case(k, v) => traverse(v, k, newPath) }.toSeq
208         case cl: ConfigList =>
209           cl.asScala.zipWithIndex.flatMap { case(v, idx) =>
210             traverse(v, idx, newPath) }
211         case _ => Seq()
212       }
213       thisComments ++ childComments
214     }
215 
216     val root = conf.root()
217     val buf = List()
218     DocComments(
219       root.asScala.flatMap { case(k, v) => traverse(v, k, buf) }.toSeq)
220   }
221 
222   protected[unbound] def isListElem(cur: Elem): Boolean = {
223     val elemChildren = cur.child.filter(_.isInstanceOf[Elem])
224     if (elemChildren.size > 1 ||
225       (1 == elemChildren.size &&
226         singular(cur.label) == elemChildren(0).label)) {
227       val head = elemChildren.head
228       elemChildren.tail.forall { _ match {
229         case e: Elem => e.label == head.label
230         case t: Text => t.text.trim.isEmpty
231         case c: XmlComment => true
232         case _ => false
233       }}
234     } else {
235       false
236     }
237   }
238 
239   def apply(root: Elem): DocComments = {
240 
241     def traverse(cur: Elem, path: List[Elem]): Seq[Comments] = {
242       val newPath = path :+ cur
243       var elemIdx = 0
244       val numElems = cur.child.filter(_.isInstanceOf[Elem]).size
245       cur.child.flatMap { n => n match {
246         case e: Elem =>
247           elemIdx = Math.min(elemIdx + 1, numElems - 1)
248           traverse(e, newPath)
249         case c: XmlComment =>
250           val newPathElems = elemPathToPath(newPath)
251           val comments = c.commentText.split(Array('\n', '\r'))
252           Seq(Comments(
253             comments.toSeq,
254             CommentPath((newPathElems ++ Seq(ListIndex(elemIdx))): _*)))
255         case _ => Seq()
256       }}
257     }
258 
259     def elemPathToPath(path: List[Elem]): Seq[PathElement] = path match {
260       case h :: h2 :: rest =>
261         if (isListElem(h))
262           Seq(
263             ElementLabel(h.label),
264             ListIndex(h.child.filter(_.isInstanceOf[Elem]).indexOf(h2))) ++
265             elemPathToPath(rest)
266         else Seq(ElementLabel(h.label)) ++ elemPathToPath(h2 :: rest)
267       case h :: Nil => Seq(ElementLabel(h.label))
268       case Nil => Seq()
269     }
270 
271     val buf = List(root)
272     DocComments(root.child.flatMap { _ match {
273       case e: Elem => traverse(e, buf)
274       case _ => Seq()
275     }})
276   }
