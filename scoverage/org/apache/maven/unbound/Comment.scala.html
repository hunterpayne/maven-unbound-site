43 case class ListIndex(idx: Int) extends PathElement {
44 
45   def matchNode(parent: Elem, n: Node): Boolean = n match {

48     case _ => false

51   import scala.collection.JavaConverters._

53   def matchConf(parent: ConfigValue, key: String, n: ConfigValue): Boolean =
54     parent match {
55       case l: ConfigList => if (l.size > idx) l.get(idx) == n else false

58         if (keys.size > idx) keys.toSeq(idx) == n

60     }

63 case class CommentPath(elems: PathElement*) {
64 
65   def insertConf(s: Seq[String], conf: Config): Config = {
66     import scala.collection.JavaConverters._
67 
68     def findNext(
69       it: Iterator[PathElement], cv: ConfigValue): Seq[ConfigValue] =
70       if (it.hasNext) {
71         val pe = it.next
72         cv match {
73           case cur: ConfigObject =>
74             cur.asScala.find { case(k, v) => pe.matchConf(cur, k, v) } match {
75               case Some((_, c)) => findNext(it, c) :+ cur
76               case None => Seq()
77             }
78           case cl: ConfigList =>
79             cl.asScala.find { v => pe.matchConf(cl, "", v) } match {
80               case Some(c) => findNext(it, c) :+ cl
81               case None => Seq()
82             }
83         }
84       } else Seq(cv)
85 
86     val path = findNext(elems.toIterator, conf.root())
87     if (!path.isEmpty) {
88       val relems = elems.reverse.toIterator
89       // add comments to path.head
90       val origin = path.head.origin().withComments(s.asJava)
91       path.tail.foldLeft(path.head.withOrigin(origin)) { case(ch, par) =>
92         par match {
93           case cur: ConfigObject => relems.next match {
94             case ElementLabel(label) => cur.withValue(label, ch)
95             case ListIndex(idx) => cur.withValue(s"[$idx]", ch)
96           }
97           case cl: ConfigList =>
98             val idx = relems.next.asInstanceOf[ListIndex].idx
99             val lst = new java.util.ArrayList[ConfigValue](cl)
100             lst.set(idx, ch)
101             ConfigValueFactory.fromIterable(lst)
102         }}.asInstanceOf[ConfigObject].toConfig
103 
104     } else conf
105   }
106 
107   def insertXml(s: Seq[String], root: Elem): Elem = {
108     import scala.collection.JavaConverters._
109 
110     def findNext(it: Iterator[PathElement], cur: Elem): Seq[Elem] =
111       if (it.hasNext) {
112         it.next match {
113           case ElementLabel(label) =>
114             cur.child.find {
115               case ch: Elem => ch.label == label
116               case _ => false
117             } match {
118               case Some(ch: Elem) => findNext(it, ch) :+ cur
119               case _ => Seq()
120             }
121           case ListIndex(idx) =>
122             findNext(
123               it,
124               cur.child.filter { _.isInstanceOf[Elem]
125               }(idx).asInstanceOf[Elem]) :+
126             cur
127         }
128       } else Seq(cur)
129 
130     val it = elems.tail.toIterator // skip project
131     val path = findNext(it, root)
132 
133     if (!path.isEmpty) {
134 
135       val relems = elems.tail.reverse.toIterator
136       val newChildren: Seq[Node] =
137         s.map { XmlComment(_) } ++ path.head.child.toSeq
138       val oldLeaf = path.head
139       val newLeaf = new Elem(
140         oldLeaf.prefix, oldLeaf.label, oldLeaf.attributes, oldLeaf.scope,
141         newChildren: _*)
142 
143       path.tail.foldLeft((newLeaf, oldLeaf)) { case((ch, oldCh), par) =>
144         val newChildren = par.child.updated(par.child.indexOf(oldCh), ch)
145         (new Elem(
146           par.prefix, par.label, par.attributes, par.scope, newChildren: _*),
147          par)
148       }._1
149     } else root
150   }
151 }

169 object CommentExtractor {
170 
171   def apply(conf: Config): DocComments = {
172     import scala.collection.JavaConverters._
173 
174     def traverse(curr: ConfigValue, key: Any, path: List[PathElement]):
175         Seq[Comments] = {
176 
177       val newPath = path :+ (key match {
178         case s: String => ElementLabel(s)
179         case i: Int => ListIndex(i)
180       })
181 
182       val comments = curr.origin().comments().asScala.toSeq
183       val thisComments =
184         if (comments.isEmpty) Seq()
185         else Seq(Comments(comments, CommentPath(newPath: _*)))
186       val childComments: Seq[Comments] = curr match {
187         case cur: ConfigObject =>
188           cur.asScala.flatMap { case(k, v) => traverse(v, k, newPath) }.toSeq
189         case cl: ConfigList =>
190           cl.asScala.zipWithIndex.flatMap { case(v, idx) =>
191             traverse(v, idx, newPath) }
192         case _ => Seq()
193       }
194       thisComments ++ childComments
195     }
196 
197     val root = conf.root()
198     val buf = List()
199     DocComments(
200       root.asScala.flatMap { case(k, v) => traverse(v, k, buf) }.toSeq)
201   }
202 
203   protected[unbound] def isListElem(cur: Elem): Boolean = {
204     val head = cur.child.filter(_.isInstanceOf[Elem]).head
205     !cur.child.isEmpty && cur.child.tail.forall { _ match {
206       case e: Elem => e.label == head.label
207       case t: Text => t.text.trim.isEmpty
208       case c: XmlComment => true
209       case _ => false
210     }}
211   }
212 
213   def apply(root: Elem): DocComments = {
214 
215     def traverse(cur: Elem, path: List[Elem]): Seq[Comments] = {
216       val newPath = path :+ cur
217       cur.child.zipWithIndex.flatMap { case(n, idx) => n match {
218         case e: Elem => traverse(e, newPath)
219         case c: XmlComment =>
220           val newPathElems = elemPathToPath(newPath)
221           Seq(Comments(
222             Seq(c.commentText),
223             CommentPath((newPathElems ++ Seq(ListIndex(idx))): _*)))
224         case _ => Seq()
225       }}
226     }
227 
228     def elemPathToPath(path: List[Elem]): Seq[PathElement] = path match {
229       case h :: h2 :: rest =>
230         if (isListElem(h))
231           Seq(
232             ElementLabel(h.label),
233             ListIndex(h.child.filter(_.isInstanceOf[Elem]).indexOf(h2))) ++
234             elemPathToPath(rest)
235         else Seq(ElementLabel(h.label)) ++ elemPathToPath(h2 :: rest)
236       case h :: Nil => Seq(ElementLabel(h.label))
237       case Nil => Seq()
238     }
239 
240     val buf = List(root)
241     DocComments(root.child.flatMap { _ match {
242       case e: Elem => traverse(e, buf)
243       case _ => Seq()
244     }})
245   }
