227 
228     def traverse(curr: ConfigValue, key: Any, path: List[PathElement]):
229         Seq[Comments] = {
230 
231       val newPath = path :+ (key match {
232         case s: String => ElementLabel(s)
233         case i: Int => ListIndex(i)
234       })
235 
236       val comments = curr.origin().comments().asScala.toSeq
237       val thisComments =
238         if (comments.isEmpty) Seq()

241         case cur: ConfigObject =>
242           cur.asScala.flatMap { case(k, v) => traverse(v, k, newPath) }.toSeq

244           cl.asScala.zipWithIndex.flatMap { case(v, idx) =>

246         case _ => Seq()

252     val buf = List()

257   protected[unbound] def isListElem(cur: Elem): Boolean = {
258     val elemChildren = cur.child.filter(_.isInstanceOf[Elem])
259     if (elemChildren.size > 1 ||
260       (1 == elemChildren.size &&
261         singular(cur.label) == elemChildren(0).label)) {
262       val head = elemChildren.head
263       elemChildren.tail.forall { _ match {
264         case e: Elem => e.label == head.label
265         case t: Text => t.text.trim.isEmpty
266         case c: XmlComment => true
267         case _ => false
268       }}
269     } else {
270       false
271     }
272   }

274   def apply(root: Elem): DocComments = {
275 
276     def traverse(cur: Elem, path: List[Elem]): Seq[Comments] = {
277       val newPath = path :+ cur
278       var elemIdx = 0
279       var lastLabel: String = null
280       val elements = cur.child.filter { _.isInstanceOf[Elem] }
281       val nextLabels: Map[String, String] =
282         if (elements.size > 1)
283           elements.map { _.label }.sliding(2).map { s => (s(0), s(1)) }.toMap
284         else Map[String, String]()
285       val isList = isListElem(cur)
286       val numElems = cur.child.filter(_.isInstanceOf[Elem]).size
287       cur.child.flatMap { n => n match {
288         case e: Elem =>
289           elemIdx = Math.min(elemIdx + 1, numElems - 1)
290           lastLabel = e.label
291           traverse(e, newPath)
292         case c: XmlComment =>
293           val newPathElems = elemPathToPath(newPath)
294           val comments = c.commentText.split(Array('\n', '\r'))
295           // dependencies are always in lists and the string dependency is never
296           // a key
297           if (null == lastLabel || isList ||
298             lastLabel == SL.DependencyStr.toString)
299             Seq(Comments(
300               comments.toSeq,
301               CommentPath((newPathElems ++ Seq(ListIndex(elemIdx))): _*)))
302           else Seq(Comments(
303             comments.toSeq,
304             CommentPath((newPathElems ++ Seq(ElementLabel(
305               nextLabels.getOrElse(lastLabel, lastLabel)))): _*)))
306         case _ => Seq()
307       }}
308     }
309 
310     def elemPathToPath(path: List[Elem]): Seq[PathElement] = path match {
311       case h :: h2 :: rest =>
312         if (isListElem(h))
313           Seq(
314             ElementLabel(h.label),
315             ListIndex(h.child.filter(_.isInstanceOf[Elem]).indexOf(h2))) ++
316             elemPathToPath(rest)
317         else Seq(ElementLabel(h.label)) ++ elemPathToPath(h2 :: rest)
318       case h :: Nil => Seq(ElementLabel(h.label))
319       case Nil => Seq()
320     }
321 
322     val buf = List(root)
323     DocComments(root.child.flatMap { _ match {
324       case e: Elem => traverse(e, buf)
325       case _ => Seq()
326     }})
327   }
