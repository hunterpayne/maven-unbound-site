228   def apply(conf: Config): DocComments = {
229     import scala.collection.JavaConverters._
230 
231     def traverse(curr: ConfigValue, key: Any, path: List[PathElement]):
232         Seq[Comments] = {
233 
234       val newPath = path :+ (key match {
235         case s: String => ElementLabel(s)
236         case i: Int => ListIndex(i)
237       })
238 
239       val comments = curr.origin().comments().asScala.toSeq.map { strip(_) }
240       val thisComments =
241         if (comments.isEmpty) Seq()
242         else Seq(Comments(comments, CommentPath(newPath: _*)))
243       val childComments: Seq[Comments] = curr match {
244         case cur: ConfigObject =>
245           cur.asScala.flatMap { case(k, v) => traverse(v, k, newPath) }.toSeq
246         case cl: ConfigList =>
247           cl.asScala.zipWithIndex.flatMap { case(v, idx) =>
248             traverse(v, idx, newPath) }
249         case _ => Seq()
250       }
251       thisComments ++ childComments
252     }

255     val buf = List()
256     DocComments(
257       root.asScala.flatMap { case(k, v) => traverse(v, k, buf) }.toSeq)

260   protected[unbound] def isListElem(cur: Elem): Boolean = {
261     val elemChildren = cur.child.filter(_.isInstanceOf[Elem])
262     if (elemChildren.size > 1 ||
263       (1 == elemChildren.size &&
264         singular(cur.label) == elemChildren(0).label)) {
265       val head = elemChildren.head
266       elemChildren.tail.forall { _ match {
267         case e: Elem => e.label == head.label
268         case t: Text => t.text.trim.isEmpty
269         case c: XmlComment => true
270         case _ => false
271       }}
272     } else {
273       false
274     }
275   }

277   def apply(root: Elem): DocComments = {
278 
279     def traverse(cur: Elem, path: List[Elem]): Seq[Comments] = {
280       val newPath = path :+ cur
281       var elemIdx = 0
282       var lastLabel: String = null
283       val elements = cur.child.filter { _.isInstanceOf[Elem] }
284       val nextLabels: Map[String, String] =
285         if (elements.size > 1)
286           elements.map { _.label }.sliding(2).map { s => (s(0), s(1)) }.toMap
287         else Map[String, String]()
288       val isList = isListElem(cur)
289       val numElems = cur.child.filter(_.isInstanceOf[Elem]).size
290       cur.child.flatMap { n => n match {
291         case e: Elem =>
292           elemIdx = Math.min(elemIdx + 1, numElems - 1)
293           lastLabel = e.label
294           traverse(e, newPath)
295         case c: XmlComment =>
296           val newPathElems = elemPathToPath(newPath)
297           val comments = c.commentText.split(Array('\n', '\r'))
298           // dependencies are always in lists and the string dependency is
299           // never a key
300           if (null == lastLabel || isList ||
301             lastLabel == SL.DependencyStr.toString)
302             Seq(Comments(
303               comments.toSeq,
304               CommentPath((newPathElems ++ Seq(ListIndex(elemIdx))): _*)))
305           else Seq(Comments(
306             comments.toSeq,
307             CommentPath((newPathElems ++ Seq(ElementLabel(
308               nextLabels.getOrElse(lastLabel, lastLabel)))): _*)))
309         case _ => Seq()
310       }}
311     }
312 
313     def elemPathToPath(path: List[Elem]): Seq[PathElement] = path match {
314       case h :: h2 :: rest =>
315         if (isListElem(h))
316           Seq(
317             ElementLabel(h.label),
318             ListIndex(h.child.filter(_.isInstanceOf[Elem]).indexOf(h2))) ++
319             elemPathToPath(rest)
320         else Seq(ElementLabel(h.label)) ++ elemPathToPath(h2 :: rest)
321       case h :: Nil => Seq(ElementLabel(h.label))
322       case Nil => Seq()
323     }
324 
325     val buf = List(root)
326     DocComments(root.child.flatMap { _ match {
327       case e: Elem => traverse(e, buf)
328       case _ => Seq()
329     }})
330   }
