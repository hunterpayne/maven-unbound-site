32 case class ElementLabel(label: String) extends PathElement {

34   def matchNode(parent: Elem, n: Node): Boolean = n match {
35     case e: Elem => e.label == label

43 case class ListIndex(idx: Int) extends PathElement {
44 
45   def matchNode(parent: Elem, n: Node): Boolean = n match {
46     case e: Elem =>
47       parent.child.toSeq.filter(_.isInstanceOf[Elem]).indexOf(e) == idx
48     case _ => false

51   import scala.collection.JavaConverters._
52 
53   def matchConf(parent: ConfigValue, key: String, n: ConfigValue): Boolean =
54     parent match {
55       case l: ConfigList => if (l.size > idx) l.get(idx) == n else false
56       case o: ConfigObject =>
57         val keys = o.asScala.keys
58         if (keys.size > idx) {
59           o.get(keys.toSeq(idx)) == n
60         } else false
61     }

64 case class CommentPath(elems: PathElement*) {
65 
66   def insertConf(s: Seq[String], conf: Config): Config = {
67     import scala.collection.JavaConverters._
68 
69     def findNext(
70       it: Iterator[PathElement], cv: ConfigValue): Seq[ConfigValue] =
71       if (it.hasNext) {
72         val pe = it.next
73         cv match {
74           case cur: ConfigObject =>
75             cur.asScala.find { case(k, v) => pe.matchConf(cur, k, v) } match {
76               case Some((_, c)) => findNext(it, c) :+ cur
77               case None => println("no match " + cur); Seq()
78             }
79           case cl: ConfigList =>
80             cl.asScala.find { v => pe.matchConf(cl, "", v) } match {
81               case Some(c) => findNext(it, c) :+ cl
82               case None =>
83                 println("no match list " + cl + " looking for " + pe)
84                 Seq()
85             }
86         }
87       } else Seq(cv)
88 
89     val it = elems.toIterator
90     val path = findNext(it, conf.root())
91     if (!path.isEmpty && elems.size <= path.size) {
92       val relems = elems.reverse.toIterator
93       // add comments to path.head
94       val origin = path.head.origin().withComments(s.asJava)
95       path.tail.foldLeft(path.head.withOrigin(origin)) { case(ch, par) =>
96         par match {
97           case cur: ConfigObject => relems.next match {
98             case ElementLabel(label) => cur.withValue(label, ch)
99             case ListIndex(idx) =>
100               val keys = cur.asScala.keys.toSeq
101               if (keys.size > idx) {
102                 val k = keys(idx)
103                 cur.withValue(k, ch)
104               } else {
105                 println("mismatch " + cur + " @ listidx=" + idx)
106                 cur
107               }
108           }
109           case cl: ConfigList => relems.next match {
110             case ListIndex(idx) =>
111               val lst = new java.util.ArrayList[ConfigValue](cl)
112               lst.set(idx, ch)
113               ConfigValueFactory.fromIterable(lst)
114             case ElementLabel(label) =>
115               println("mismatch " + label + " in " + cl)
116               cl
117           }
118         }}.asInstanceOf[ConfigObject].toConfig
119 
120     } else conf
121   }

123   def insertXml(s: Seq[String], root: Elem): Elem = {
124     import scala.collection.JavaConverters._
125 
126     def findNext(it: Iterator[PathElement], cur: Elem): Seq[Elem] =
127       if (it.hasNext) {
128         it.next match {
129           case ElementLabel(label) =>
130             cur.child.find {
131               case ch: Elem => ch.label == label
132               case _ => false
133             } match {
134               case Some(ch: Elem) => findNext(it, ch) :+ cur
135               case _ => Seq()
136             }
137           case ListIndex(idx) =>
138             findNext(
139               it,
140               cur.child.filter { _.isInstanceOf[Elem]
141               }(idx).asInstanceOf[Elem]) :+
142             cur
143         }
144       } else Seq(cur)
145 
146     val it = elems.tail.toIterator // skip project
147     val path = findNext(it, root)
148 
149     if (!path.isEmpty) {
150 
151       val relems = elems.tail.reverse.toIterator
152       val newChildren: Seq[Node] =
153         s.map { XmlComment(_) } ++ path.head.child.toSeq
154       val oldLeaf = path.head
155       val newLeaf = new Elem(
156         oldLeaf.prefix, oldLeaf.label, oldLeaf.attributes, oldLeaf.scope,
157         newChildren: _*)

159       path.tail.foldLeft((newLeaf, oldLeaf)) { case((ch, oldCh), par) =>
160         val newChildren = par.child.updated(par.child.indexOf(oldCh), ch)
161         (new Elem(
162           par.prefix, par.label, par.attributes, par.scope, newChildren: _*),
163          par)
164       }._1
165     } else root
166   }

185 object CommentExtractor {
186 
187   def apply(conf: Config): DocComments = {
188     import scala.collection.JavaConverters._
189 
190     def traverse(curr: ConfigValue, key: Any, path: List[PathElement]):
191         Seq[Comments] = {
192 
193       val newPath = path :+ (key match {
194         case s: String => ElementLabel(s)
195         case i: Int => ListIndex(i)
196       })
197 
198       val comments = curr.origin().comments().asScala.toSeq
199       val thisComments =
200         if (comments.isEmpty) Seq()
201         else Seq(Comments(comments, CommentPath(newPath: _*)))
202       val childComments: Seq[Comments] = curr match {
203         case cur: ConfigObject =>
204           cur.asScala.flatMap { case(k, v) => traverse(v, k, newPath) }.toSeq
205         case cl: ConfigList =>
206           cl.asScala.zipWithIndex.flatMap { case(v, idx) =>
207             traverse(v, idx, newPath) }
208         case _ => Seq()
209       }
210       thisComments ++ childComments
211     }
212 
213     val root = conf.root()
214     val buf = List()
215     DocComments(
216       root.asScala.flatMap { case(k, v) => traverse(v, k, buf) }.toSeq)
217   }
218 
219   protected[unbound] def isListElem(cur: Elem): Boolean = {
220     val elemChildren = cur.child.filter(_.isInstanceOf[Elem])
221     if (elemChildren.size > 1 ||
222       (1 == elemChildren.size &&
223         singular(cur.label) == elemChildren(0).label)) {
224       val head = elemChildren.head
225       elemChildren.tail.forall { _ match {
226         case e: Elem => e.label == head.label
227         case t: Text => t.text.trim.isEmpty
228         case c: XmlComment => true
229         case _ => false
230       }}
231     } else {
232       false
233     }
234   }
235 
236   def apply(root: Elem): DocComments = {
237 
238     def traverse(cur: Elem, path: List[Elem]): Seq[Comments] = {
239       val newPath = path :+ cur
240       var elemIdx = 0
241       cur.child.flatMap { n => n match {
242         case e: Elem => elemIdx = elemIdx + 1; traverse(e, newPath)
243         case c: XmlComment =>
244           val newPathElems = elemPathToPath(newPath)
245           Seq(Comments(
246             Seq(c.commentText),
247             CommentPath((newPathElems ++ Seq(ListIndex(elemIdx))): _*)))
248         case _ => Seq()
249       }}
250     }
251 
252     def elemPathToPath(path: List[Elem]): Seq[PathElement] = path match {
253       case h :: h2 :: rest =>
254         if (isListElem(h))
255           Seq(
256             ElementLabel(h.label),
257             ListIndex(h.child.filter(_.isInstanceOf[Elem]).indexOf(h2))) ++
258             elemPathToPath(rest)
259         else Seq(ElementLabel(h.label)) ++ elemPathToPath(h2 :: rest)
260       case h :: Nil => Seq(ElementLabel(h.label))
261       case Nil => Seq()
262     }
263 
264     val buf = List(root)
265     DocComments(root.child.flatMap { _ match {
266       case e: Elem => traverse(e, buf)
267       case _ => Seq()
268     }})
269   }
