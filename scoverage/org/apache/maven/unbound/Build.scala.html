111 case class Build(
112   sourceDirectory: String = Build.defSourceDir,
113   scriptSourceDirectory: String = Build.defScriptDir,
114   testSourceDirectory: String = Build.defTestDir,
115   outputDirectory: String = Build.defOutputDir,
116   testOutputDirectory: String = Build.defTestOutputDir,
117   extensions: Seq[Extension] = Seq[Extension](),
118   defaultGoal: String = null,
119   resources: Seq[Resource] = Seq[Resource](new Resource(Build.defResourcesDir)),
120   testResources: Seq[Resource] =
121     Seq[Resource](new Resource(Build.defTestResourcesDir)),
122   directory: String = SL.Target,
123   finalName: String = null, /* ${artifactId}-${version} */
124   filters: Seq[String] = Seq[String](),
125   pluginManagement: Seq[Plugin] = Seq[Plugin](),
126   plugins: Seq[Plugin] = Seq[Plugin]()) {
127 
128   def this(elem: Elem) = this(
129     emptyToDefault((elem \ SL.SourceDirectory).text, Build.defSourceDir),
130     emptyToDefault((elem \ SL.ScriptSourceDirectory).text, Build.defScriptDir),
131     emptyToDefault((elem \ SL.TestSourceDirectory).text, Build.defTestDir),
132     emptyToDefault((elem \ SL.OutputDirectory).text, Build.defOutputDir),
133     emptyToDefault(
134       (elem \ SL.TestOutputDirectory).text, Build.defTestOutputDir),
135     (elem \ SL.Extensions \ SL.ExtensionStr).map { case e: Elem =>
136       new Extension(e)},
137     emptyToNull((elem \ SL.DefaultGoal).text),
138     ensureDefault(
139       (elem \ SL.Resources \ SL.ResourceStr).map { case e: Elem =>
140         new Resource(e) },
141       new Resource(Build.defResourcesDir)),
142     ensureDefault(
143       (elem \ SL.TestResources \ SL.TestResource).map { case e: Elem =>
144         new Resource(e) },
145       new Resource(Build.defTestResourcesDir)),
146     emptyToDefault((elem \ SL.DirectoryStr).text, SL.Target),
147     emptyToNull((elem \ SL.FinalName).text),
148     (elem \ SL.Filters \ SL.FilterStr).map { _.text },
149     (elem \ SL.PluginManagement \ SL.Plugins \ SL.PluginStr).map {
150       case e: Elem => new Plugin(e) },
151     (elem \ SL.Plugins \ SL.PluginStr).map { case e: Elem => new Plugin(e) })
152 
153   lazy val xml =
154     <build>
155       { if (sourceDirectory != null && sourceDirectory != Build.defSourceDir)
156         <sourceDirectory>{sourceDirectory}</sourceDirectory> }
157       { if (scriptSourceDirectory != null &&
158         scriptSourceDirectory != Build.defScriptDir)
159         <scriptSourceDirectory>{scriptSourceDirectory}</scriptSourceDirectory> }
160       { if (testSourceDirectory != null &&
161         testSourceDirectory != Build.defTestDir)
162         <testSourceDirectory>{testSourceDirectory}</testSourceDirectory> }
163       { if (outputDirectory != null && outputDirectory != Build.defOutputDir)
164         <outputDirectory>{outputDirectory}</outputDirectory> }
165       { if (testOutputDirectory != null &&
166         testOutputDirectory != Build.defTestOutputDir)
167         <testOutputDirectory>{testOutputDirectory}</testOutputDirectory> }
168       { if (!extensions.isEmpty) <extensions>
169         { extensions.map { _.xml } }
170         </extensions> }
171       { if (defaultGoal != null) <defaultGoal>{defaultGoal}</defaultGoal> }
172       { if (!resources.isEmpty && !Build.isDefaultResources(resources))
173         <resources> { resources.map { _.xml } } </resources> }
174       { if (!testResources.isEmpty &&
175         !Build.isDefaultTestResources(testResources))
176         <testResources> { testResources.map { _.testXml } } </testResources> }
177       { if (directory != null && directory != SL.Target.toString)
178         <directory>{directory}</directory> }
179       { if (finalName != null) <finalName>{finalName}</finalName> }
180       { if (!filters.isEmpty) <filters>
181         { filters.map { Filter(_).xml } }
182         </filters> }
183       { if (!pluginManagement.isEmpty) <pluginManagement><plugins>
184         { pluginManagement.map { _.xml } }
185         </plugins></pluginManagement> }
186       { if (!plugins.isEmpty) <plugins>
187         { plugins.map { _.xml } }
188         </plugins> }
189     </build>
190 
191   def makeModelObject(): org.apache.maven.model.Build = {
192     val build = new org.apache.maven.model.Build()
193     build.setSourceDirectory(sourceDirectory)
194     build.setScriptSourceDirectory(scriptSourceDirectory)
195     build.setTestSourceDirectory(testSourceDirectory)
196     build.setOutputDirectory(outputDirectory)
197     build.setTestOutputDirectory(testOutputDirectory)
198     extensions.foreach { ex => build.addExtension(ex.makeModelObject()) }
199     if (defaultGoal != null) build.setDefaultGoal(defaultGoal)
200     resources.foreach { r => build.addResource(r.makeModelObject()) }
201     testResources.foreach { r => build.addTestResource(r.makeModelObject()) }
202     if (directory != null) build.setDirectory(directory)
203     if (finalName != null) build.setFinalName(finalName)
204     filters.foreach { f => build.addFilter(f) }
205     val pm = new org.apache.maven.model.PluginManagement()
206     pluginManagement.foreach { p => pm.addPlugin(p.makeModelObject()) }
207     build.setPluginManagement(pm)
208     plugins.foreach { p => build.addPlugin(p.makeModelObject()) }
209     build
210   }

305 case class Resource(
306   targetPath: String = null, filtering: Boolean = false,
307   directory: String = SL.Dot,
308   includes: Seq[String], excludes: Seq[String] = Seq[String]()) {
309 
310   def this(name: String) = this(null, false, SL.Dot, Seq[String](name))
311 
312   def this(elem: Elem) = this(
313     emptyToNull((elem \ SL.TargetPath).text),
314     emptyToDefault(
315       (elem \ SL.Filtering).text.toLowerCase(Locale.ROOT), SL.FalseStr) ==
316       SL.TrueStr.toString,
317     emptyToDefault((elem \ SL.DirectoryStr).text, SL.Dot),
318     (elem \ SL.Includes \ SL.IncludeStr).map { _.text },
319     (elem \ SL.Excludes \ SL.ExcludeStr).map { _.text })
320 
321   lazy val xml =
322     <resource>
323       { if (targetPath != null) <targetPath>{targetPath}</targetPath> }
324       <filtering>{filtering}</filtering>
325       <directory>{directory}</directory>
326       { if (!includes.isEmpty) <includes>
327         { includes.map { Include(_).xml } }
328         </includes> }
329       { if (!excludes.isEmpty) <excludes>
330         { excludes.map { Exclude(_).xml } }
331         </excludes> }
332     </resource>
333 
334   lazy val testXml =
335     <testResource>
336       { if (targetPath != null) <targetPath>{targetPath}</targetPath> }
337       <filtering>{filtering}</filtering>
338       <directory>{directory}</directory>
339       { if (!includes.isEmpty) <includes>
340         { includes.map { Include(_).xml } }
341         </includes> }
342       { if (!excludes.isEmpty) <excludes>
343         { excludes.map { Exclude(_).xml } }
344         </excludes> }
345     </testResource>
346 
347   def makeModelObject(): org.apache.maven.model.Resource = {
348     val resource = new org.apache.maven.model.Resource()
349     if (targetPath != null) resource.setTargetPath(targetPath)
350     resource.setFiltering(filtering)
351     resource.setDirectory(directory)
352     includes.foreach { inc => resource.addInclude(inc) }
353     excludes.foreach { exc => resource.addExclude(exc) }
354     resource
355   }
356 }
