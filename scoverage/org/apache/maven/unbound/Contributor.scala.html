70   organization: String = null, organizationUrl: String = null,
71   roles: Seq[String] = Seq[String](), timezone: String = null,

74   def this(elem: Elem) = this(
75     emptyToNull((elem \ SL.Name).text),
76     emptyToNull((elem \ SL.Email).text),
77     emptyToNull((elem \ SL.UrlStr).text),
78     emptyToNull((elem \ SL.OrganizationStr).text),
79     emptyToNull((elem \ SL.OrganizationUrl).text),
80     (elem \ SL.Roles \ SL.RoleStr).map { _.text },
81     emptyToNull((elem \ SL.TimezoneStr).text),
82     (elem \ SL.PropertiesStr).flatMap(_.map { e => (e.label, e.text) }).toMap)
83 
84   lazy val xml =
85     <contributor>
86       { if (name != null) <name>{name}</name> }
87       { if (email != null) <email>{email}</email> }
88       { if (url != null) <url>{url}</url> }
89       { if (organization != null) <organization>{organization}</organization> }
90       { if (organizationUrl != null)
91         <organizationUrl>{organizationUrl}</organizationUrl> }
92       { if (!roles.isEmpty) <roles> { roles.map { Role(_).xml } } </roles> }
93       { if (timezone != null) <timezone>{timezone}</timezone> }
94       { if (properties != null && !properties.isEmpty) <properties>
95         { properties.map { case(k, v) => PropertyValue(k, v).xml } }
96         </properties> }
97     </contributor>
98 
99   def makeModelObject(): org.apache.maven.model.Contributor = {
100     val contributor = new org.apache.maven.model.Contributor()
101     if (name != null) contributor.setName(name)
102     if (email != null) contributor.setEmail(email)
103     if (url != null) contributor.setUrl(url)
104     if (organization != null) contributor.setOrganization(organization)
105     if (organizationUrl != null) contributor.setOrganizationUrl(organizationUrl)
106     roles.foreach { role => contributor.addRole(role) }
107     if (timezone != null) contributor.setTimezone(timezone)
108     if (properties != null)
109       properties.foreach { case(k, v) => contributor.addProperty(k, v) }
110     contributor
111   }
