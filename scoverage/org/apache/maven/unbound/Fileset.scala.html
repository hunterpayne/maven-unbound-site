84   def this(elem: Elem) = this(

94     (elem \ SL.Mapper).map { case e: Elem =>
95       new Mapper(e) }.headOption.getOrElse(null)
96   )
97 
98   lazy val xml =
99     <fileset>
100       { if (directory != null) <directory>{directory}</directory> }
101       { if (lineEnding != null) <lineEnding>{lineEnding}</lineEnding> }
102       { if (followSymlinks) <followSymlinks>true</followSymlinks> }
103       { if (outputDirectory != null)
104         <outputDirectory>{outputDirectory}</outputDirectory> }
105       { if (!useDefaultExcludes)
106         <useDefaultExcludes>false</useDefaultExcludes> }
107       { if (!includes.isEmpty) <includes>
108         { includes.map { Include(_).xml } }
109         </includes> }
110       { if (!excludes.isEmpty) <excludes>
111         { excludes.map { Exclude(_).xml } }
112         </excludes> }
113       { if (fileMode != null && fileMode != SL.DefaultFileMode.toString)
114         <fileMode>{fileMode}</fileMode> }
115       { if (directoryMode != null &&
116         directoryMode != SL.DefaultDirectoryMode.toString)
117         <directoryMode>{directoryMode}</directoryMode> }
118       { if (mapper != null) mapper.xml }
119     </fileset>
120 }
